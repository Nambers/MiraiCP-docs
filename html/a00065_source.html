715 class MIRAICP_EXPORT Event {
716 private: // typedefs
717 class eventNode {
718 private:
720 std::shared_ptr<NodeHandle> _handle;
721
722 public:
723 std::function<bool(MiraiCPEvent *)> func;
724
725 public:
726 eventNode() : _handle(), func(nullptr) {}
727
728 explicit eventNode(std::function<bool(MiraiCPEvent *)> f) : _handle(new NodeHandle(true)), func(std::move(f)) {}
729
730 eventNode(const eventNode &_o) = default; // for MSVC compatible, or you will get an error
731 eventNode(eventNode &&_o) noexcept : _handle(std::move(_o._handle)), func(std::move(_o.func)) {}
732
733 public:
735 bool run(MiraiCPEvent *a) const {
736 return _handle->isEnable() && func(a);
737 }
738
739 std::shared_ptr<NodeHandle> getHandle() {
740 return _handle;
741 }
742 };
743
744 using priority_level = unsigned char;
745 using event_vector = std::vector<eventNode>;
746 using eventNodeTable = std::vector<std::map<priority_level, event_vector>>;
747
748 private: // member
749 eventNodeTable _all_events_;
750
751 private:
752 Event() : _all_events_(int(eventTypes::Types::count)){};
753
754 public: // singleton mode
755 static Event processor;
756
757 private:
758 template<typename EventClass>
759 constexpr static int id() {
760 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持广播继承MiraiCPEvent的事件");
761 return static_cast<int>(EventClass::get_event_type());
762 }
763
764 public:
765 static bool noRegistered(int index) {
766 return processor._all_events_[index].empty();
767 }
768
770 static void clear() {
771 for (auto &a: processor._all_events_) a.clear();
772 }
773
774 static void incomingEvent(nlohmann::json j, int type);
776 template<typename EventClass>
777 static void broadcast(EventClass &&val) {
778 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持广播MiraiCPEvent的派生类");
779 MiraiCPEvent *p = &val;
780 for (auto &&[k, v]: processor._all_events_[id<EventClass>()]) {
781 for (auto &&a: v) {
782 if (a.run(p)) return;
783 }
784 }
785 }

794 template<typename EventClass>
795 static std::shared_ptr<NodeHandle> registerEvent(std::function<void(EventClass)> callback, priority_level level = 100) {
796 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持注册MiraiCPEvent的派生类事件");
797 std::function<bool(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
798 callback(*static_cast<EventClass *>(p));
799 return false;
800 };
801 auto t = eventNode(tmp);
802 auto ans = t.getHandle();
803 // 先获得shared_ptr才可以emplace_back
804 processor._all_events_[id<EventClass>()][level].emplace_back(std::move(t));
805 return ans;
806 }

816 template<typename EventClass>

818 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持注册MiraiCPEvent的派生类事件");

820 return callback(*dynamic_cast<EventClass *>(p));
821 };
