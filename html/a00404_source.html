51 void push_task(size_t id, functionPointerType func) {
52 bool insert;
53 {
54 std::unique_lock lk(get_mutex());
55 insert = get_function_storer().insert({id, std::move(func)}).second;
56 }
57
58 if (!insert) {
59 Logger::logger.error("Fatal: task id " + std::to_string(id) + " is already taken by another task!");
60 return;
61 }

79 void task_interface(size_t id) {
80 static_assert(std::is_same_v<LibLoader::LoaderApi::task_func_with_id, decltype(&task_interface)>);
81 storerIteratorType iter;
82 functionPointerType fPtr;
83 {
84 std::shared_lock lck(get_mutex());
85 iter = get_function_storer().find(id);
86 if (iter == get_function_storer().end()) {
87 Logger::logger.error("Invalid task id " + std::to_string(id) + " is requested to run");
88 return;
89 }
90 fPtr = iter->second;
91 }
92 // running task!
93 (*fPtr)();
94 }
