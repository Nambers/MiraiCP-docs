42 int loadCore(const std::string &corePath) {
43 //java虚拟机启动时接收的参数
44 JavaVMOption vmOption[2];
45
46 vmOption[0].optionString = (char *) "-Xmx2048M";
47#ifdef _WIN32
48 std::string tmp = "-Djava.class.path=.\\;" + corePath + ";";
49#else
50 std::string tmp = "-Djava.class.path=./:" + corePath;
51#endif
52 vmOption[1].optionString = (char *) tmp.c_str();
53
54 // for debugging, add "-verbose:jni"
55 // note: change length of vmOption to 3
56 // vmOption[2].optionString = (char *) "-verbose:jni";
57
58 JavaVMInitArgs vmInitArgs;
59 vmInitArgs.version = JNI_VERSION_1_8;
60 vmInitArgs.options = vmOption;
61 // 参数数
62 vmInitArgs.nOptions = 2;
63 // 忽略无法识别jvm的情况
64 // vmInitArgs.ignoreUnrecognized = JNI_TRUE;
65 // 不忽略
66 vmInitArgs.ignoreUnrecognized = JNI_FALSE;
67 //加载JVM动态库
68 JNIEnv *env = nullptr;
69#ifdef _WIN32
70 jvmLib = ::LoadLibraryA(_JVM_DLL_PATH);
71 if (jvmLib == nullptr) {
72 // 加载JVM动态库错误
73 return 1;
74 }
75 //获取JVM函数地址
76 JNICREATEPROC jvmProcAddress = (JNICREATEPROC) GetProcAddress((HINSTANCE) jvmLib, "JNI_CreateJavaVM");
77 if (jvmLib == nullptr) {
78 freeLibrary(jvmLib);
79 // 加载JVM动态库错误
80 return 2;
81 }
82 jint jvmProc = (jvmProcAddress) (&MiraiCP::ThreadManager::gvm, (void **) &env, &vmInitArgs);
83 if (jvmProc < 0 || MiraiCP::ThreadManager::gvm == nullptr || env == nullptr) {
84 freeLibrary(jvmLib);
85 // 创建JVM错误
86 return 3;
87 }
88#else
89 long flag = JNI_CreateJavaVM(&MiraiCP::ThreadManager::gvm, (void **) &env, &vmInitArgs);
90 if (flag != 0) {
91 // Error creating VM. Exiting...
92 freeLibrary(nullptr);
93 return 3;
94 }
95#endif
96 //加载启动类
97 coreClaz = env->FindClass("tech/eritquearcus/miraicp/core/Core");
98 if (env->ExceptionCheck() == JNI_TRUE || coreClaz == nullptr) {
99 env->ExceptionDescribe();
100 env->ExceptionClear();
101 freeLibrary(jvmLib);
102 // 加载启动类失败
103 return 4;
104 }
105 //加载 main 启动方法
106 jmethodID methodID = env->GetStaticMethodID(coreClaz, "load", "()V");
107 if (env->ExceptionCheck() == JNI_TRUE || methodID == nullptr) {
108 env->ExceptionDescribe();
109 env->ExceptionClear();
110 freeLibrary(jvmLib);
111 // 加载启动方法失败
112 return 5;
113 }
114 JNINativeMethod e[] = {{(char *) "enroll", (char *) "()Ljava/lang/Void;", (jstring *) enroll}};
115 MiraiCP::JNIApi::registerMethods(env, "tech/eritquearcus/miraicp/core/Core", e, 1);
116 env->CallStaticVoidMethod(coreClaz, methodID, nullptr);
117 return 0;
118 }

126 Bot login(QQID id, std::string_view password, bool md5, std::string_view protocol, std::string_view heartBeat, JNIEnv *env) {
127 nlohmann::json tmp;
128 tmp["id"] = id;
129 tmp["passwords"] = password;
130 tmp["md5"] = md5;
131 tmp["protocol"] = protocol;
132 tmp["heartBeat"] = heartBeat;
133 if (env == nullptr)
134 env = MiraiCP::ThreadManager::getEnv();
135 jmethodID loginMethodId = env->GetStaticMethodID(coreClaz, "login", "(Ljava/lang/String;)Ljava/lang/String;");
136 std::string re = MiraiCP::Tools::jstring2str((jstring) env->CallStaticObjectMethod(coreClaz, loginMethodId, MiraiCP::Tools::str2jstring(tmp.dump().c_str(), env)));
137 if (re != "200")
138 throw BotException("登录异常:" + re, MIRAICP_EXCEPTION_WHERE);
139 return Bot(id);
140 }
