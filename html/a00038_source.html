52 inline std::shared_ptr<InternalBot> get_bot(QQID id) {
53 static std::unordered_map<QQID, std::shared_ptr<InternalBot>> BotPool;
54 static std::mutex mtx;
55 std::lock_guard<std::mutex> lck(mtx);
56 auto &Ptr = BotPool.try_emplace(id).first->second;
57 if (!Ptr) Ptr = std::make_shared<InternalBot>(id);
58 return Ptr;
