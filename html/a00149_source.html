33 template<typename Callable, typename... Args>
34 explicit MiraiCPNewThread(Callable &&func, Args &&...args)
35 : std::thread(
36 [lambda_func = std::forward<Callable>(func)](auto &&...argss) {
37 try {
38 lambda_func(std::forward<decltype(argss)>(argss)...);
39 } catch (MiraiCPExceptionBase &e) {
40 e.raise();
41 Event::broadcast(MiraiCPExceptionEvent(&e));
42 } catch (const std::exception &e) {
43 MiraiCPThreadException exNew(std::string(e.what()), std::this_thread::get_id(), MIRAICP_EXCEPTION_WHERE);
44 exNew.raise();
45 Event::broadcast(MiraiCPExceptionEvent(&exNew));
46 } catch (...) {
47 MiraiCPThreadException exNew("unknown exception type", std::this_thread::get_id(), MIRAICP_EXCEPTION_WHERE);
48 exNew.raise();
49 Event::broadcast(MiraiCPExceptionEvent(&exNew));
50 }
51 },
52 std::forward<Args>(args)...) {}
