41 class SingleMessage : public MiraiCodeable {
42 public:
43 virtual ~SingleMessage() = default;
44 static std::map<int, std::string> messageType;
45
49 static int getKey(const std::string &value) {
50 for (const auto &a: messageType) {
51 if (Tools::iequal(a.second, value)) return a.first;
52 }
53 return -1;
54 }
55
58 int type;
59 std::string content;
60 std::string prefix;
61
62 std::string toMiraiCode() const override {
63 Logger::logger.info("base");
64 if (type > 0)
65 if (type == 1)
66 return "[mirai:at:" + content + "] ";
67 else if (type == 2)
68 return "[mirai:atall] ";
69 else
70 return "[mirai:" + messageType[type] + this->prefix + Tools::escapeToMiraiCode(content) + "]";
71 else
72 return content;
73 }
74
75 bool operator==(const SingleMessage &m) const {
76 return this->type == m.type && this->toMiraiCode() == m.toMiraiCode();
77 }
78
79 bool operator==(SingleMessage *m) const {
80 return this->type == m->type && this->toMiraiCode() == m->toMiraiCode();
81 }
82
87 SingleMessage(int type, std::string content, std::string prefix = ":") : type(type),
88 content(std::move(content)),
89 prefix(std::move(prefix)) {}
90 };
