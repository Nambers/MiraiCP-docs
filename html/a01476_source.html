58 bool ForwardedMessage::operator==(const ForwardedMessage &m) const {
59 if (this->nodes.size() != m.nodes.size()) return false;
60 int i = 0;
61 if (!(this->display == m.display)) return false;
62 return std::all_of(this->nodes.begin(), this->nodes.end(), [&i, &m](const auto &n) {
63 return n.message == m[i++].message;
64 });

90 ForwardedMessage ForwardedMessage::deserializationFromMessageJson(const json &msg) {
91 if (!(msg.contains("type") && msg["type"] == "ForwardMessage") && msg.contains("nodeList"))
92 throw std::invalid_argument("json is not a ForwardMessage");
93 ForwardedMessageDisplayStrategy display;
94 if (msg.contains("title") && msg.contains("brief") && msg.contains("source") && msg.contains("summary") && msg.contains("preview"))
95 display = ForwardedMessageDisplayStrategy::fromJson(msg);
96 else
97 display = ForwardedMessageDisplayStrategy::defaultStrategy();
98 std::vector<ForwardedNode> nodes;
99 auto &j = msg["nodeList"];
100 if (!j.is_array())
101 throw IllegalArgumentException("OnlineForwardedMessage参数异常, 应为 array", MIRAICP_EXCEPTION_WHERE);
102 try {
103 for (auto &&a: j) {
104 if (a["messageChain"][0].contains("kind") && a["messageChain"][0]["kind"] == "FORWARD") {
105 nodes.emplace_back(a["senderId"], a["senderName"], ForwardedMessage::deserializationFromMessageJson(a["messageChain"][1]["nodeList"]), a["time"]);
106 } else {
107 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageJson(a["messageChain"]), a["time"]);
108 }
109 }
110
111 } catch (const json::parse_error &) {
112 throw APIException("OnlineForwardedMessage格式化异常", MIRAICP_EXCEPTION_WHERE);
113 }
114 return ForwardedMessage(std::move(nodes), display);
115 }
