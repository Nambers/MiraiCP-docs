49 static auto &deleteQueue() {
50 static std::deque<std::unique_ptr<PolyM::Msg>> queue;
51 return queue;

54 PayLoadInfo _try_get_payload() {
55 PayLoadInfo ret{nullptr, -1};
56 auto queue_ptr = MiraiCP::getMsgQueue();
57 auto msg = queue_ptr->tryGet();
58 if (!msg) return ret;
59 ret.payload = msg->getRawPayLoad();
60 ret.payload_id = msg->getMsgId();
61 deleteQueue().emplace_back(std::move(msg));
62 return ret;
63 }
64
65 void _delete_one_msg() {
66 deleteQueue().pop_front();
67 }

70 auto queue_ptr = MiraiCP::getMsgQueue();
71 queue_ptr->respondTo(msg_uid, PolyM::DataMsg<MiraiCPString>(MessageType::RET, std::move(msg)));
72 }

84MIRAICP_EXPORT PluginMessageHandles_C FUNC_ENTRANCE(PluginArgv_C argv) {
85 static_assert(std::is_same_v<decltype(&FUNC_ENTRANCE), LibLoader::plugin_entrance_func_ptr>);
86 using namespace MiraiCP;
87 PluginInterface::PluginMessageHandles ret{};
88
89 Event::clear();
90 getMsgQueue()->clear();
91 register_plugin_argv(argv);
92 // logger api is set
93
94 try {
95 Logger::logger.info("开始启动插件: " + MiraiCP::CPPPlugin::config.getId());
96 enrollPlugin();
97 // plugin == nullptr 无插件实例加载
98 if (CPPPlugin::plugin != nullptr) {
99 CPPPlugin::plugin->onEnable();
100 }
101 } catch (const MiraiCPExceptionBase &e) {
102 try {
103 std::cerr.flush();
104 e.raise();
105 Logger::logger.error("插件(id=" + CPPPlugin::config.getId() + ", name=" + CPPPlugin::config.name + ")启动失败");
106 } catch (...) {}
107 return ret.toC();
108 } catch (const std::exception &e) {
109 try {
110 std::cerr.flush();
111 Logger::logger.error(e.what());
112 Logger::logger.error("插件(id=" + CPPPlugin::config.getId() + ", name=" + CPPPlugin::config.name + ")启动失败");
113 } catch (...) {}
114 return ret.toC();
115 } catch (...) {
116 try {
117 std::cerr.flush();
118 Logger::logger.error("插件(id=" + CPPPlugin::config.getId() + ", name=" + CPPPlugin::config.name + ")启动失败");
119 } catch (...) {}
120 return ret.toC();
121 }
122
123 ret.try_get_payload = PluginInterface::_try_get_payload;
124 ret.delete_one_msg = PluginInterface::_delete_one_msg;
125 ret.make_response = MiraiCP::PluginInterface::_make_response;
126 return ret.toC();

149MIRAICP_EXPORT int FUNC_EVENT(const MiraiCP::MiraiCPString &c) {
150 static_assert(std::is_same_v<decltype(&FUNC_EVENT), LibLoader::plugin_event_func_ptr>);
151 using json = nlohmann::json;
152 using namespace MiraiCP;
153 std::string content = c;
154 json j;
155
156 try {
157 j = json::parse(content);
158 } catch (json::parse_error &e) {
159 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("消息解析分流：格式化json错误");
160 Logger::logger.error("For debug: " + content);
161 Logger::logger.error(e.what());)
162 return PLUGIN_NORMAL;
163 }
164
165 int type;
166
167 try {
168 if (!j.is_object() || !j.contains("eventId") || !j["eventId"].is_number()) {
169 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("Json格式错误：没有eventId这一field或这一field不是数字类型");)
170 return PLUGIN_NORMAL;
171 }
172 type = j["eventId"].get<int>();
173 if (type != eventTypes::Command && Event::noRegistered(type)) return PLUGIN_NORMAL;
174 } catch (...) {
175 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("Json格式错误：解析json时遇到无法预料的异常");)
176 return PLUGIN_NORMAL;
177 }
178
179 try {
180 // core logic
181 Event::incomingEvent(BaseEventData(std::move(j)), type);
182 } catch (json::type_error &e) {
183 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("json格式化异常,位置：FUNC_EVENT");
184 Logger::logger.error(e.what());
185 Logger::logger.error(content);)
186 return PLUGIN_NORMAL;
187 } catch (MiraiCPExceptionBase &e) {
188 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Event::broadcast<MiraiCPExceptionEvent>(MiraiCPExceptionEvent(&e));
189 e.raise();)
190 return PLUGIN_NORMAL;
191 } catch (const std::exception &e) {
192 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error(e.what());
193 Logger::logger.error("info:", content);)
194 return PLUGIN_NORMAL;
195 } catch (...) {
196 // 如果产生了无法处理的异常，直接退出插件，return 非0值
197 // loader端将处理这个异常并直接unload绑定的插件
198 // 如果存在其他线程，可能导致段错误
199 return PLUGIN_ERROR;
200 }
201 return PLUGIN_NORMAL;
202}
