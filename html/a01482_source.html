46 auto GetGroupFromPool(QQID groupid, QQID botid) noexcept {
47 static std::unordered_map<QQID, std::unordered_map<QQID, std::shared_ptr<Group::DataType>>> Pool;
48 auto &Val = Pool[botid][groupid];
49 if (!Val) {
50 Val = std::make_shared<Group::DataType>(groupid);
51 Val->_botId = botid;
52 Val->_id = groupid;
53 Val->_type = MIRAI_GROUP;
54 }
55 return Val;
56 }

246 void GroupData::refreshInfo() {
247 std::string re = LowLevelAPI::getInfoSource(internalToJson());
248 LowLevelAPI::info tmp = LowLevelAPI::info0(re);
249 this->_nickOrNameCard = std::move(tmp.nickOrNameCard);
250 this->_avatarUrl = std::move(tmp.avatarUrl);
251 nlohmann::json j = nlohmann::json::parse(re)["setting"];
252 this->_setting.name = json_stringmover(j, "name");
253 this->_setting.isMuteAll = j["isMuteAll"];
254 this->_setting.isAllowMemberInvite = j["isAllowMemberInvite"];
255 this->_setting.isAutoApproveEnabled = j["isAutoApproveEnabled"];
256 this->_setting.isAnonymousChatEnabled = j["isAnonymousChatEnabled"];
257 }
