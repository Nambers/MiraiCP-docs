28class Main : public CPPPlugin {
29public:
30 Main() : CPPPlugin() {}
31 void onEnable() override {
32 Event::registerEvent<GroupMessageEvent>([](GroupMessageEvent a) {
33 // 一起传递过去的自定义信息
34 nlohmann::json j;
35 j["type"] = 1;
36 j["id"] = 111;
37 j["bid"] = 111;
38 schedule(100, j.dump());
39 Event::processor.registerEvent<TimeOutEvent>([](TimeOutEvent e) {
40 nlohmann::json j = nlohmann::json::parse(e.msg);
41 if (j["type"] == 1)
42 // 发送群消息
43 Group(j["id"], j["bid"]).sendMessage("");
44 });
45 });
46 }
47};
