28class Main : public CPPPlugin {
29public:
30 Main() : CPPPlugin() {}
31 void onEnable() override {
32 Event::registerEvent<GroupMessageEvent>([](GroupMessageEvent e) {
33 auto style1 = ForwardedMessageDisplayStrategy();
34 style1.summary = "Summarya";
35 style1.preview.emplace_back("preview1");
36 style1.source = "SourceA";
37 style1.brief = "BriefA";
38 e.group.sendMessage("Origin:");
39 ForwardedMessage(
40 {ForwardedNode(12, "a", MessageChain(PlainText("test")), 1),
41 ForwardedNode(12, "b", ForwardedMessage({ForwardedNode(12, "a", MessageChain(PlainText("test")), 1)}), 1)},
42 ForwardedMessageDisplayStrategy::defaultStrategy())
43 .sendTo(&e.group);
44
45 if (e.message[0].type() == OnlineForwardedMessage::type())
46 e.message[0].get<OnlineForwardedMessage>().toForwardedMessage(ForwardedMessageDisplayStrategy::defaultStrategy()).sendTo(&e.group);
47 e.group.sendMessage("style1");
48 ForwardedMessage(
49 {ForwardedNode(12, "a", MessageChain(PlainText("test")), 1),
50 ForwardedNode(12, "b", ForwardedMessage({ForwardedNode(12, "a", MessageChain(PlainText("test")), 1)}), 1)},
51 style1)
52 .sendTo(&e.group);
53
54 if (e.message[0].type() == OnlineForwardedMessage::type())
55 e.message[0].get<OnlineForwardedMessage>().toForwardedMessage(style1).sendTo(&e.group);
56 });
57 }
58};

61 MiraiCP::enrollPlugin(new Main());
