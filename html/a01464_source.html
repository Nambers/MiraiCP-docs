38 template<typename T>
39 MiraiCPString send_payload(T &&payload) {
40 using PayloadClass = typename std::decay_t<T>;
41 static_assert(std::is_base_of_v<MessageTraits<PayloadClass, true>, PayloadClass> || std::is_base_of_v<MessageTraits<PayloadClass, false>, PayloadClass>,
42 "PayloadClass must be a subclass of MessageTraits");
43 if constexpr (PayloadClass::blocking) {
44 auto msg = DataMsg<PayloadClass>(PayloadClass::static_payload_class_id(),
45 std::forward<T>(payload));
46 msg.getPayload().uid = msg.getUniqueId();
47 auto ret_raw = getMsgQueue()->request(
48 std::move(msg),
49 0,
50 LibLoader::try_wake_loader);
51 auto ret = static_cast<PolyM::DataMsg<MiraiCPString> *>(ret_raw.get());
52 return std::move(ret->getPayload());
53 } else {
54 auto msg = DataMsg<PayloadClass>(PayloadClass::static_payload_class_id(), std::forward<T>(payload));
55 msg.getPayload().uid = msg.getUniqueId();
56 getMsgQueue()->put(std::move(msg));
57 LibLoader::try_wake_loader();
58 return {};
59 }
60 }
