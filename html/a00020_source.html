83 MessageSource Contact::sendVoice0(const std::string &path, JNIEnv *env) {
84 json j;
85 json source;
86 source["path"] = path;
87 j["source"] = source.dump();
88 j["contactSource"] = this->toString();
89 std::string re = Config::koperation(Config::Voice, j, env);
90 if (re == "E1") {
91 throw UploadException("上传语音文件格式不对(必须为.amr/.silk)或文件不存在", MIRAICP_EXCEPTION_WHERE);
92 } else if (re == "E2") {
93 throw UploadException("上传语音文件大小超过服务器限制，一般限制在1MB上下", MIRAICP_EXCEPTION_WHERE);
94 }
95 return MessageSource::deserializeFromString(re);
