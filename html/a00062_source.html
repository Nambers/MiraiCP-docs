38 MessageChain GroupMessageEvent::nextMessage(long time, bool halt) const {
39 json j{{"contactSource", this->group.toString()}, {"time", time}, {"halt", halt}};
40 std::string r = KtOperation::ktOperation(KtOperation::NextMsg, std::move(j));
41 if (r == "E1")
42 throw TimeOutException("取下一条信息超时", MIRAICP_EXCEPTION_WHERE);
43 json re = json::parse(r);
44 return MessageChain::deserializationFromMessageSourceJson(json::parse(re["messageSource"].get<std::string>())).plus(MessageSource::deserializeFromString(re["messageSource"]));

169 void Event::incomingEvent(json j, int type) {
170 switch (type) {
171 case eventTypes::GroupMessageEvent: {
172 //GroupMessage
173 Event::broadcast(GroupMessageEvent(std::move(j)));
174 break;
175 }
176 case eventTypes::PrivateMessageEvent: {
177 //私聊消息
178 Event::broadcast(PrivateMessageEvent(std::move(j)));
179 break;
180 }
181 case eventTypes::GroupInviteEvent: {
182 //群聊邀请
183 Event::broadcast(GroupInviteEvent(std::move(j)));
184 break;
185 }
186 case eventTypes::NewFriendRequestEvent: {
187 //好友
188 Event::broadcast(NewFriendRequestEvent(std::move(j)));
189 break;
190 }
191 case eventTypes::MemberJoinEvent: { //新成员加入
192 Event::broadcast(MemberJoinEvent(std::move(j)));
193 break;
194 }
195 case eventTypes::MemberLeaveEvent: {
196 //群成员退出
197 Event::broadcast(MemberLeaveEvent(std::move(j)));
198 break;
199 }
200 case eventTypes::RecallEvent: {
201 Event::broadcast(RecallEvent(std::move(j)));
202 break;
203 }
204 case eventTypes::BotJoinGroupEvent: {
205 Event::broadcast(BotJoinGroupEvent(std::move(j)));
206 break;
207 }
208 case eventTypes::GroupTempMessageEvent: {
209 Event::broadcast(GroupTempMessageEvent(std::move(j)));
210 break;
211 }
212 case eventTypes::TimeOutEvent: {
213 Event::broadcast(TimeOutEvent(Tools::json_stringmover(j, "msg")));
214 break;
215 }
216 case eventTypes::BotOnlineEvent: {
217 Event::broadcast(BotOnlineEvent(j["botid"].get<QQID>()));
218 break;
219 }
220 case eventTypes::NudgeEvent: {
221 Event::broadcast(NudgeEvent(std::move(j)));
222 break;
223 }
224 case eventTypes::BotLeaveEvent: {
225 Event::broadcast(BotLeaveEvent(std::move(j)));
226 break;
227 }
228 case eventTypes::MemberJoinRequestEvent: {
229 Event::broadcast(MemberJoinRequestEvent(std::move(j)));
230 break;
231 }
232 case eventTypes::MessagePreSendEvent: {
233 Event::broadcast(MessagePreSendEvent(std::move(j)));
234 break;
235 }
236 case eventTypes::Command: {
237 // command
238 CommandManager::commandManager[j["bindId"]]->onCommand(
239 j.contains("contact") ? Contact::deserializeToPointer(Tools::json_jsonmover(j, "contact")) : nullptr,
240 Bot(j["botid"]),
241 MessageChain::deserializationFromMessageSourceJson(
242 j.contains("message") ? j["message"].get<std::string>() : "",
243 false));
244 break;
245 }
246 default: {
247 throw APIException("Unreachable code", MIRAICP_EXCEPTION_WHERE);
248 }
249 }
