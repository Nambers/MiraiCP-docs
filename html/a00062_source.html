116 static NodeHandle *internalRegister(std::function<bool(MiraiCPEvent *)> callback, size_t where, priority_level level) {
117 std::lock_guard lk(processor.eventsMtx);
118 auto &row = processor._all_events_[where][level];
119 row.emplace_back(std::make_unique<eventNode>(std::move(callback)));

124 template<typename EventClass>
125 static void broadcast(EventClass &&val) {
126 using UnderlyingClass = std::decay_t<EventClass>;
127 static_assert(std::is_base_of_v<MiraiCPEvent, UnderlyingClass>, "只支持广播MiraiCPEvent的派生类");
128 MiraiCPEvent *p = &val;
129 for (auto &&[k, v]: processor._all_events_[id<UnderlyingClass>()]) {
130 for (auto &&a: v) {
131 if (a->run(p)) return;
132 }
133 }
