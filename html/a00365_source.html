51 MessageChain MessageChain::deserializationFromMessageSourceJson(json j, bool origin) {
52 if (origin)
53 j = j["originalMessage"];
54 MessageChain mc;
55 if (j.empty()) return mc;
56
57 if (!j.is_array()) throw IllegalArgumentException(std::string(__func__) + "输入的json应当是数组类型", MIRAICP_EXCEPTION_WHERE);
58 json::array_t jArray = std::move(j);
59 if (jArray[0]["type"] == "MessageOrigin") {
60 if (jArray[0]["kind"] == "MUSIC_SHARE") {
61 if (jArray.size() < 2) throw IllegalArgumentException(std::string(__func__) + "数组长度应至少为2", MIRAICP_EXCEPTION_WHERE);
62
63 mc.add(MusicShare(jArray[1]["kind"], jArray[1]["title"], jArray[1]["summary"], jArray[1]["jumpUrl"], jArray[1]["pictureUrl"],
64 jArray[1]["musicUrl"], jArray[1]["brief"]));
65 return mc;
66 }
67 // todo del MessageSource deserialization methods
68 // mc.add(ForwardedMessage::deserializationFromMessageSourceJson(jArray));
69 return mc;
70 }
71
72 for (auto &node: jArray) {
73 if (node["type"] == "SimpleServiceMessage") {
74 mc.add(ServiceMessage(node["serviceId"], node["content"]));
75 continue;
76 }
77 if (node["type"] == "LightApp") {
78 mc.add(LightApp(node["content"]));
79 continue;
80 }
81 if (node["type"] == "OnlineAudio") {
82 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
83 node["urlForDownload"]));
84 continue;
85 }
86 if (node["type"] == "FileMessage") {
87 // note: RemoteFile::deserializeFromString() uses j["finfo"]["size"], which is different from here.
88 mc.add(RemoteFile(node["id"], node["internalId"], node["name"], node["size"]));
89 continue;
90 }
91 if (node["type"] == "MarketFace") {
92 mc.add(MarketFace(node["delegate"]["faceId"]));
93 continue;
94 }
95 switch (SingleMessage::getKey(node["type"])) {
96 // todo(Antares): change to enum in each case
97 case -2:
98 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
99 break;
100 case 0:
101 mc.add(PlainText(node["content"].get<std::string>()));
102 break;
103 case 1:
104 mc.add(At(node["target"]));
105 break;
106 case 2:
107 mc.add(AtAll());
108 break;
109 case 3:
110 mc.add(Image(node["imageId"]));
111 break;
112 case 7:
113 mc.add(Face(node["id"]));
114 break;
115 case 8:
116 mc.add(FlashImage(node["imageId"]));
117 break;
118 default:
119 Logger::logger.warning(
120 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: node:" +
121 node.dump());
122 mc.add(UnSupportMessage(node.dump()));
123 }
124 }
125 return mc;
126 }

128 MessageChain MessageChain::deserializationFromMessageJson(const json &j) {
129 MessageChain mc;
130 if (j.empty()) return mc;
131 if (!j.is_array())
132 throw IllegalArgumentException(std::string(__func__) + "输入的json应当是数组类型", MIRAICP_EXCEPTION_WHERE);
133 for (auto &node: j.get<json::array_t>()) {
134 switch (SingleMessage::getKey(node["type"])) {
135 case SingleMessageType::MusicShare_t:
136 mc.add(MusicShare(node["kind"], node["title"], node["summary"], node["jumpUrl"], node["pictureUrl"],
137 node["musicUrl"], node["brief"]));
138 break;
139 case SingleMessageType::ServiceMessage_t:
140 mc.add(ServiceMessage(node["serviceId"], node["content"]));
141 break;
142 case SingleMessageType::LightApp_t:
143 mc.add(LightApp(node["content"]));
144 break;
145 case SingleMessageType::OnlineAudio_t:
146 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"],
147 node["length"],
148 node["urlForDownload"]));
149 break;
150 case SingleMessageType::MarketFace_t:
151 mc.add(MarketFace(node["delegate"]["faceId"]));
152 break;
153 case SingleMessageType::RemoteFile_t:
154 // note: RemoteFile::deserializeFromString() uses j["finfo"]["size"], which is different from here.
155 mc.add(RemoteFile(node["id"], node["internalId"], node["name"], node["size"]));
156 break;
157 case SingleMessageType::MessageSource_t:
158 // mc.add(MessageSource::deserializeFromString(node.dump()));
159 break;
160 case SingleMessageType::QuoteReply_t:
161 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
162 break;
163 case SingleMessageType::UnsupportedMessage_t:
164 mc.add(UnSupportMessage(node["struct"].dump()));
165 break;
166 case SingleMessageType::PlainText_t:
167 mc.add(PlainText(node["content"].get<std::string>()));
168 break;
169 case SingleMessageType::At_t:
170 mc.add(At(node["target"]));
171 break;
172 case SingleMessageType::AtAll_t:
173 mc.add(AtAll());
174 break;
175 case SingleMessageType::Image_t:
176 mc.add(Image(node["imageId"], node["size"].get<size_t>(), node["width"].get<int>(),
177 node["height"].get<int>(), node["imageType"],
178 node["isEmoji"].get<bool>()));
179 break;
180 case SingleMessageType::Face_t:
181 mc.add(Face(node["id"]));
182 break;
183 case SingleMessageType::FlashImage_t:
184 mc.add(FlashImage(node["imageId"]));
185 break;
186 case SingleMessageType::OnlineForwardedMessage_t:
187 mc.add(ForwardedMessage::deserializationFromMessageJson(node));
188 break;
189 default:
190 Logger::logger.warning(
191 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: node:" +
192 node.dump());
193 mc.add(UnSupportMessage(node.dump()));
194 }
195 }
196 return mc;
197 }

236 size_t MessageChain::findEnd(const std::string &s, size_t start) {
237 size_t pos = start;
238 while (pos < s.length()) {
239 switch (s[pos]) {
240 case '\\':
241 pos += 2;
242 continue;
243 case ']':
244 return pos;
245 }
246 pos++;
247 }
248 return -1;
249 }
