74inline void platform_set_thread_name(decltype(platform_thread_self()) id, const char *name) {
75 if (checkPthreadLen(name)) {
76 pthread_setname_np(id, name);
77 } else {
78 std::string tNewName;
79 tNewName.reserve(15);
80 tNewName.append(std::string_view(name, 15));
81 pthread_setname_np(id, tNewName.c_str());
82 }
83}
