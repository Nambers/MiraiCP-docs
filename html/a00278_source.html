45 MessageChain MessageChain::deserializationFromMiraiCode(const std::string &m) {
46 size_t pos = 0;
47 size_t lastPos = -1;
48 MessageChain mc;
49 if (m.length() <= 7) {
50 return MessageChain(PlainText(m));
51 }
52 do {
53 if (m.length() - 7 - pos > 0 && m.substr(pos, 7) == "[mirai:") {
54 if (pos - lastPos > 1)
55 mc.add(PlainText(m.substr(lastPos + 1, pos - lastPos - 1))); // plain text
56 size_t back = MessageChain::findEnd(m, pos);
57 if (back == (size_t) -1) throw IllegalStateException("", MIRAICP_EXCEPTION_WHERE);
58 std::string tmp = m.substr(pos, back - pos);
59 tmp = Tools::replace(std::move(tmp), "[mirai:", "");
60 size_t i = tmp.find(':'); // first :
61 int t = SingleMessage::getMiraiCodeKey(tmp.substr(0, i));
62 switch (t) {
63 case SingleMessageType::PlainText_t:
64 // no miraiCode key is PlainText
65 Logger::logger.error("无法预料的错误, 信息: " + m);
66 break;
67 case SingleMessageType::At_t:
68 mc.add(At(std::stoll(tmp.substr(i + 1, tmp.length() - i - 1))));
69 break;
70 case SingleMessageType::AtAll_t:
71 mc.add(AtAll());
72 break;
73 case SingleMessageType::Image_t:
74 mc.add(Image(tmp.substr(i + 1, tmp.length() - i - 1)));
75 break;
76 case SingleMessageType::LightApp_t:
77 mc.add(LightApp(tmp.substr(i + 1, tmp.length() - i - 1)));
78 break;
79 case SingleMessageType::ServiceMessage_t: {
80 size_t comma = tmp.find(',');
81 mc.add(ServiceMessage(std::stoi(tmp.substr(i + 1, comma - i - 1)),
82 tmp.substr(comma + 1, tmp.length() - comma - 1)));
83 break;
84 }
85 case SingleMessageType::RemoteFile_t: {
86 //[mirai:file:/b53231e8-46dd-11ec-8ba5-5452007bd6c0,102,run.bat,55]
87 size_t comma1 = tmp.find(',');
88 size_t comma2 = tmp.find(',', comma1 + 1);
89 size_t comma3 = tmp.find(',', comma2 + 1);
90 mc.add(RemoteFile(tmp.substr(i + 1, comma1 - i - 1),
91 std::stoi(tmp.substr(comma1 + 1, comma2 - comma1 - 1)),
92 tmp.substr(comma2 + 1, comma3 - comma2 - 1),
93 std::stoll(tmp.substr(comma3 + 1, tmp.length() - comma3 - 1))));
94 break;
95 }
96 case SingleMessageType::Face_t:
97 mc.add(Face(std::stoi(tmp.substr(i + 1, tmp.length() - i - 1))));
98 break;
99 case SingleMessageType::FlashImage_t:
100 mc.add(FlashImage(tmp.substr(i + 1, tmp.length() - i - 1)));
101 break;
102 case SingleMessageType::MusicShare_t: {
103 //[mirai:musicshare:name,title,summary,jUrl,pUrl,mUrl,brief]
104 auto temp = Tools::split(tmp.substr(i + 1, tmp.length() - i - 1), ",");
105 mc.add(MusicShare(temp[0], temp[1], temp[2], temp[3], temp[4], temp[5], temp[6]));
106 break;
107 }
108 default:
109 Logger::logger.error(
110 "MiraiCP碰到了意料之中的错误(原因:部分SimpleMessage在MiraiCode解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MiraiCode:" +
111 m);
112 mc.add(UnSupportMessage("[mirai:" + tmp));
113 break;
114 }
115 pos = back;
116 lastPos = pos;
117 if (t == 1)
118 lastPos++;
119 }
120 pos++;
121 } while (pos < m.length());
122 if (lastPos + 1 < m.length())
123 mc.add(PlainText(m.substr(lastPos + 1, m.length() - lastPos - 1))); // plain text
124 return mc;
125 }

127 MessageChain MessageChain::deserializationFromMessageSourceJson(json j, bool origin) {
128 if (origin)
129 j = j["originalMessage"];
130 MessageChain mc;
131 if (j.empty()) return mc;
132
133 if (!j.is_array()) throw IllegalArgumentException(std::string(__func__) + "输入的json应当是数组类型", MIRAICP_EXCEPTION_WHERE);
134 json::array_t jArray = std::move(j);
135 if (jArray[0]["type"] == "MessageOrigin") {
136 if (jArray[0]["kind"] == "MUSIC_SHARE") {
137 if (jArray.size() < 2) throw IllegalArgumentException(std::string(__func__) + "数组长度应至少为2", MIRAICP_EXCEPTION_WHERE);
138
139 mc.add(MusicShare(jArray[1]["kind"], jArray[1]["title"], jArray[1]["summary"], jArray[1]["jumpUrl"], jArray[1]["pictureUrl"],
140 jArray[1]["musicUrl"], jArray[1]["brief"]));
141 return mc;
142 }
143 // todo del MessageSource deserialization methods
144// mc.add(ForwardedMessage::deserializationFromMessageSourceJson(jArray));
145 return mc;
146 }
147
148 for (auto &node: jArray) {
149 if (node["type"] == "SimpleServiceMessage") {
150 mc.add(ServiceMessage(node["serviceId"], node["content"]));
151 continue;
152 }
153 if (node["type"] == "LightApp") {
154 mc.add(LightApp(node["content"]));
155 continue;
156 }
157 if (node["type"] == "OnlineAudio") {
158 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
159 node["urlForDownload"]));
160 continue;
161 }
162 if (node["type"] == "FileMessage") {
163 // note: RemoteFile::deserializeFromString() uses j["finfo"]["size"], which is different from here.
164 mc.add(RemoteFile(node["id"], node["internalId"], node["name"], node["size"]));
165 continue;
166 }
167 if (node["type"] == "MarketFace") {
168 mc.add(MarketFace(node["delegate"]["faceId"]));
169 continue;
170 }
171 switch (SingleMessage::getKey(node["type"])) {
172 // todo(Antares): change to enum in each case
173 case -2:
174 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
175 break;
176 case 0:
177 mc.add(PlainText(node["content"].get<std::string>()));
178 break;
179 case 1:
180 mc.add(At(node["target"]));
181 break;
182 case 2:
183 mc.add(AtAll());
184 break;
185 case 3:
186 mc.add(Image(node["imageId"]));
187 break;
188 case 7:
189 mc.add(Face(node["id"]));
190 break;
191 case 8:
192 mc.add(FlashImage(node["imageId"]));
193 break;
194 default:
195 Logger::logger.warning(
196 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: node:" +
197 node.dump());
198 mc.add(UnSupportMessage(node.dump()));
199 }
200 }
201 return mc;
202 }

204 MessageChain MessageChain::deserializationFromMessageJson(const json &j) {
205 MessageChain mc;
206 if (j.empty()) return mc;
207 if (!j.is_array())
208 throw IllegalArgumentException(std::string(__func__) + "输入的json应当是数组类型", MIRAICP_EXCEPTION_WHERE);
209 for (auto &node: j.get<json::array_t>()) {
210 switch (SingleMessage::getKey(node["type"])) {
211 case SingleMessageType::MusicShare_t:
212 mc.add(MusicShare(node["kind"], node["title"], node["summary"], node["jumpUrl"], node["pictureUrl"],
213 node["musicUrl"], node["brief"]));
214 break;
215 case SingleMessageType::ServiceMessage_t:
216 mc.add(ServiceMessage(node["serviceId"], node["content"]));
217 break;
218 case SingleMessageType::LightApp_t:
219 mc.add(LightApp(node["content"]));
220 break;
221 case SingleMessageType::OnlineAudio_t:
222 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"],
223 node["length"],
224 node["urlForDownload"]));
225 break;
226 case SingleMessageType::MarketFace_t:
227 mc.add(MarketFace(node["delegate"]["faceId"]));
228 break;
229 case SingleMessageType::RemoteFile_t:
230 // note: RemoteFile::deserializeFromString() uses j["finfo"]["size"], which is different from here.
231 mc.add(RemoteFile(node["id"], node["internalId"], node["name"], node["size"]));
232 break;
233 case SingleMessageType::MessageSource_t:
234// mc.add(MessageSource::deserializeFromString(node.dump()));
235 break;
236 case SingleMessageType::QuoteReply_t:
237 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
238 break;
239 case SingleMessageType::UnsupportedMessage_t:
240 mc.add(UnSupportMessage(node["struct"].dump()));
241 break;
242 case SingleMessageType::PlainText_t:
243 mc.add(PlainText(node["content"].get<std::string>()));
244 break;
245 case SingleMessageType::At_t:
246 mc.add(At(node["target"]));
247 break;
248 case SingleMessageType::AtAll_t:
249 mc.add(AtAll());
250 break;
251 case SingleMessageType::Image_t:
252 mc.add(Image(node["imageId"], node["size"].get<size_t>(), node["width"].get<int>(),
253 node["height"].get<int>(), node["imageType"],
254 node["isEmoji"].get<bool>()));
255 break;
256 case SingleMessageType::Face_t:
257 mc.add(Face(node["id"]));
258 break;
259 case SingleMessageType::FlashImage_t:
260 mc.add(FlashImage(node["imageId"]));
261 break;
262 case SingleMessageType::OnlineForwardedMessage_t:
263 mc.add(ForwardedMessage::deserializationFromMessageJson(node));
264 break;
265 default:
266 Logger::logger.warning(
267 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: node:" +
268 node.dump());
269 mc.add(UnSupportMessage(node.dump()));
270 }
271 }
272 return mc;
273 }
