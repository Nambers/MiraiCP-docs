58 void admin_message(MessageProcessor::payload_ptr msg, const plugin_ptr &plugin) {
59 if (!plugin->isEnabled()) return;
60 using PayloadClass = MiraiCP::AdminMessage;
61 auto admin_msg = message_cast<PayloadClass>(msg);
62 switch (admin_msg->cmd_id) {
63 case MiraiCP::AdminMessage::ENABLE_ONE:
64 LoaderApi::enablePluginById(admin_msg->cmd_arg);
65 break;
66 case MiraiCP::AdminMessage::DISABLE_ONE:
67 LoaderApi::disablePluginById(admin_msg->cmd_arg);
68 break;
69 case MiraiCP::AdminMessage::ENABLE_ALL:
70 LoaderApi::enableAllPlugins();
71 break;
72 case MiraiCP::AdminMessage::DISABLE_ALL:
73 LoaderApi::disableAllPlugins();
74 break;
75 case MiraiCP::AdminMessage::RELOAD_ONE:
76 LoaderApi::reloadPluginById(admin_msg->cmd_arg);
77 break;
78 case MiraiCP::AdminMessage::UNLOAD_ONE:
79 LoaderApi::unloadPluginById(admin_msg->cmd_arg);
80 break;
81 case MiraiCP::AdminMessage::LOAD_NEW:
82 LoaderApi::loadNewPlugin(admin_msg->cmd_arg, admin_msg->immediate);
83 break;
84 default:
85 break;
