89 MessageSource Contact::sendVoiceImpl(std::string path) const {
90 json source{{"path", std::move(path)}};
91 json j{{"source", source.dump()},
92 {"contactSource", toString()}};
93 std::string re = KtOperation::ktOperation(KtOperation::Voice, std::move(j));
94 if (re == "E1") {
95 throw UploadException("上传语音文件格式不对(必须为.amr/.silk)或文件不存在", MIRAICP_EXCEPTION_WHERE);
96 } else if (re == "E2") {
97 throw UploadException("上传语音文件大小超过服务器限制，一般限制在1MB上下", MIRAICP_EXCEPTION_WHERE);
98 }
99 return MessageSource::deserializeFromString(re);
100 }

173 if (msg.empty()) {
