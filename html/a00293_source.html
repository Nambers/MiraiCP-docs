30 inline std::shared_ptr<IContactData> get_bot(QQID id) {
31 static std::unordered_map<QQID, std::shared_ptr<IContactData>> BotPool;
32 static std::mutex mtx;
33 std::lock_guard<std::mutex> lck(mtx);
34 auto &Ptr = BotPool.try_emplace(id).first->second;
35 if (!Ptr) {
36 Ptr = std::make_shared<IContactData>();
37 Ptr->_id = id;
38 Ptr->_type = MIRAI_BOT;
39 Ptr->_botId = id;
40 Ptr->forceRefreshNextTime();
41 }
42 return Ptr;
43 }
