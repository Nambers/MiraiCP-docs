46void VerifyImpl(JSTRING _version, JSTRING _cfgPath) {
47 //初始化日志模块
48 MiraiCP::Redirector::start();
49
50 LibLoader::logger.info("⭐libLoader 版本: " + MiraiCP::MiraiCPVersion);
51 auto version = "v" + J_TO_STD_STRING(_version);
52 if (version != MiraiCP::MiraiCPVersion) {
53 LibLoader::logger.warning(
54 "libLoader(" + MiraiCP::MiraiCPVersion + ")版本和MiraiCP启动器(" + version + ")不一致, 建议更新至最新");
55 }
56
57 // 测试有效的插件
58 LibLoader::registerAllPlugin(J_TO_STD_STRING(_cfgPath));
59
60 // 激活插件。创建loader thread。
61 // loader thread中创建多线程加载所有插件，调用入口函数
62 LibLoader::loaderThread = std::thread(LibLoader::LoaderMain::loaderMain);
63}

65void EventImpl(JSTRING content) {
66 std::string str = J_TO_STD_STRING(content);
67 auto stringPtr = MiraiCP::MiraiCPString(str);
68
69 if (str.find(R"("eventId":1000)") != std::string::npos) {
70 LibLoader::builtInCommand(str);
71 return;
72 }
73
74 LibLoader::PluginListManager::broadcastToAllEnabledPlugins(stringPtr);
75}

77void PluginDisableImpl() {
78 LibLoader::LoaderMain::loaderExit();
79 {
80 LibLoader::loaderWakeCV().notify_one();
81 }
82 LibLoader::loaderThread.join();
83}

144int registerMethods(JNIEnv *env, const char *className, const JNINativeMethod *gMethods, int numMethods) {
145 jclass clazz = env->FindClass(className);
146 if (clazz == nullptr) {
147 return JNI_FALSE;
148 }
149 //注册native方法
150 if (env->RegisterNatives(clazz, gMethods, numMethods) < 0) {
151 return JNI_FALSE;
152 }
153 return JNI_TRUE;
154}

158 JNIEnv *env = nullptr;
