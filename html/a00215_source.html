44

103 void LoaderMain::mainloop() noexcept {
104 static std::mutex fakeLock;
105 std::unique_lock fakeUniqueLock(fakeLock);
106 loaderWakeCV().wait(fakeUniqueLock, []() { return is_loader_exited() || shouldTick() || !loader_thread_task_queue.empty(); });
107
108 tick();
109 if (!loader_thread_task_queue.empty()) {
110 loadertask task;
111 {
112 std::lock_guard lk(task_mtx);
113 std::swap(task, loader_thread_task_queue.front());
114 loader_thread_task_queue.pop();
115 }
116 try {
117 switch (task.first) {
118 case LOADER_TASKS::ADD_THREAD:
119 loader_enablePluginById(task.second);
120 break;
121 case LOADER_TASKS::END_THREAD:
122 loader_disablePluginById(task.second);
123 break;
124 case LOADER_TASKS::ENABLE_ALL:
125 loader_enableAllPlugins();
126 break;
127 case LOADER_TASKS::DISABLE_ALL:
128 loader_disableAllPlugins();
129 break;
130 case LOADER_TASKS::LOAD_NEW_ACTIVATENOW:
131 loader_loadNewPlugin(task.second, true);
132 break;
133 case LOADER_TASKS::LOAD_NEW_DONTACTIVATE:
134 loader_loadNewPlugin(task.second, false);
135 break;
136 case LOADER_TASKS::UNLOAD:
137 loader_unloadPluginById(task.second);
138 break;
139 case LOADER_TASKS::RELOAD:
140 loader_reloadPluginById(task.second);
141 break;
142 case LOADER_TASKS::EXCEPTION_PLUGINEND:
143 loader_pluginEndByException(task.second);
144 break;
145 case LOADER_TASKS::RESET_THREAD:
146 loader_resetThreadByIndex(task.second);
147 break;
148 default:
149 throw std::exception();
150 }
151 } catch (LoaderBaseException &e) {
152 e.raise();
153 } catch (std::exception &e) {
154 logger.error("libLoader线程遇到无法预料的错误：" + std::string(e.what()) + "，程序将继续执行，如果发生意料之外的状况，请提交issue");
155 }
156 }
157 }
