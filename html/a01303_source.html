29 auto GetMemberFromPool(QQID id, QQID groupId, QQID botid) noexcept {
30 using Tools::idpair;
31 static std::unordered_map<idpair, std::unordered_map<QQID, std::shared_ptr<Member::DataType>>> Pool;
32 idpair pr{botid, groupId};
33 auto &Val = Pool[pr][id];
34 if (!Val) {
35 Val = std::make_shared<Member::DataType>(groupId);
36 Val->_id = id;
37 Val->_botId = botid;
38 Val->_type = MIRAI_MEMBER;
39 }
40 return Val;
41 }

43 auto GetMemberFromPool(const json &in_json) {

46 } catch (const nlohmann::detail::exception &) {
47 throw IllegalArgumentException("构造Member时传入的json异常", MIRAICP_EXCEPTION_WHERE);
48 }

56 Member::Member(nlohmann::json in_json) : Contact(GetMemberFromPool(in_json)) {
57 auto ActualDataPtr = GetDataInternal();
58 assert(ActualDataPtr != nullptr);
59 bool needrefresh = false;
60 if (in_json.contains("nickOrNameCard")) ActualDataPtr->_nickOrNameCard = Tools::json_stringmover(in_json, "nickOrNameCard");
61 else
62 needrefresh = true;
63 if (in_json.contains("avatarUrl")) ActualDataPtr->_avatarUrl = Tools::json_stringmover(in_json, "avatarUrl");
64 else
65 needrefresh = true;
66 if (in_json.contains("anonymous")) ActualDataPtr->_anonymous = in_json["anonymous"].get<bool>();
67 else
68 needrefresh = true;
69
70 if (needrefresh) forceRefreshNextTime();
71 }

113 void MemberData::refreshInfo() {
114 if (_anonymous) return;
115
116 std::string result = LowLevelAPI::getInfoSource(internalToJson());
117
118 if (result == "E1")
119 throw MemberException(1, MIRAICP_EXCEPTION_WHERE);
120 if (result == "E2")
121 throw MemberException(2, MIRAICP_EXCEPTION_WHERE);
122
123 {
124 LowLevelAPI::info tmp = LowLevelAPI::info0(result);
125 this->_nickOrNameCard = tmp.nickOrNameCard;
126 this->_avatarUrl = tmp.avatarUrl;
127 }
128
129 {
130 _permission = stoi(KtOperation::ktOperation(KtOperation::QueryM, internalToJson()));
131 }
132 }
