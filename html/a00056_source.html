40 void Group::OnlineAnnouncement::deleteThis() {
41 json j, i;
42 i["botid"] = this->botid;
43 i["groupid"] = this->groupid;
44 i["fid"] = this->fid;
45 i["type"] = 1;
46 j["identify"] = i.dump();
47 std::string re = Config::koperation(Config::Announcement, j);
48 if (re == "E1")
49 throw IllegalArgumentException("无法根据fid找到群公告(群公告不存在)", MIRAICP_EXCEPTION_WHERE);
50 if (re == "E3")
51 throw IllegalStateException("群公告状态异常", MIRAICP_EXCEPTION_WHERE);
52 }

147 RemoteFile Group::sendFile(const std::string &path, const std::string &filepath, JNIEnv *env) {
148 json tmp;
149 json source;
150 source["path"] = path;
151 source["filepath"] = filepath;
152 tmp["source"] = source.dump();
153 tmp["contactSource"] = this->toString();
154 std::string callback = Config::koperation(Config::SendFile, tmp, env);
155 if (callback == "E2") throw UploadException("找不到" + filepath + "位置:C-uploadfile", MIRAICP_EXCEPTION_WHERE);
156 if (callback == "E3")
157 throw UploadException("Upload error:路径格式异常，应为'/xxx.xxx'或'/xx/xxx.xxx'目前只支持群文件和单层路径, path:" + path, MIRAICP_EXCEPTION_WHERE);
158 return RemoteFile::deserializeFromString(callback);
159 }

161 RemoteFile Group::getFile(const std::string &path, const std::string &id, JNIEnv *env) {
162 // source 参数
163 if (path.empty() || path == "/")

165 json tmp;
166 json j;

168 tmp["path"] = path;
