32 void construct(JNIEnv *env) {
33 if (env == nullptr) env = ThreadManager::getEnv();
34 CPPLib = reinterpret_cast<jclass>(env->NewGlobalRef(
35 env->FindClass("tech/eritquearcus/miraicp/shared/CPPLib")));
36 if (CPPLib == nullptr) {
37 throw APIException("初始化错误，找不到CPPLib类", MIRAICP_EXCEPTION_WHERE);
38 }
39 KOperation = env->GetStaticMethodID(CPPLib, "KOperation", "(Ljava/lang/String;)Ljava/lang/String;");
40 }
41
42 void destruct() {
43 ThreadManager::getEnv()->DeleteGlobalRef(CPPLib);
44 }

46 std::string koperation(operation_set type, const nlohmann::json &data, JNIEnv *env, bool catchErr, const std::string &errorInfo) {
47 if (env == nullptr) env = ThreadManager::getEnv();
48 nlohmann::json j;
49 j["type"] = type;
50 j["data"] = data;
51 std::string re = Tools::jstring2str((jstring) env->CallStaticObjectMethod(CPPLib,
52 KOperation,
53 Tools::str2jstring(j.dump().c_str(), env)),
54 env);
55 if (catchErr) ErrorHandle(re, errorInfo);
56 return re;
57 }
