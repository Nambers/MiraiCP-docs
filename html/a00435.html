// 从id构造群成员
Member a(e.sender.id(), e.group.id(), e.bot.id);
// 踢出群成员

Member m = Member(e.sender.id(), e.group.id(), e.bot.id);
m.kick("this_is_reason");
} catch (BotException &err) {
//权限不足
Logger::logger.error(err.what());
} catch (MemberException &err) {
if (err.type == 1) {
//找不到群
}
if (err.type == 2) {
//找不到群成员
}
}
});
}
};
