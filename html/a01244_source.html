33 bool ForwardedMessage::operator==(const ForwardedMessage &m) const {
34 if (this->nodes.size() != m.nodes.size()) return false;
35 int i = 0;
36 if(!(this->display == m.display)) return false;
37 return std::all_of(this->nodes.begin(), this->nodes.end(), [&i, &m](const auto &n) {
38 return n.message == m[i++].message;
39 });
40 }

65 ForwardedMessage ForwardedMessage::deserializationFromMessageJson(const json &msg) {
66 if(!(msg.contains("type") && msg["type"]=="ForwardMessage") && msg.contains("nodeList"))
67 throw std::invalid_argument("json is not a ForwardMessage");
68 ForwardedMessageDisplayStrategy display;
69 if(msg.contains("title") && msg.contains("brief") && msg.contains("source") && msg.contains("summary") && msg.contains("preview"))
70 display = ForwardedMessageDisplayStrategy::fromJson(msg);
71 else
72 display = ForwardedMessageDisplayStrategy::defaultStrategy();
73 std::vector<ForwardedNode> nodes;
74 auto &j = msg["nodeList"];
75 if(!j.is_array())
76 throw IllegalArgumentException("OnlineForwardedMessage参数异常, 应为 array", MIRAICP_EXCEPTION_WHERE);
77 try {
78 for (auto &&a: j) {
79 if (a["messageChain"][0].contains("kind") && a["messageChain"][0]["kind"] == "FORWARD") {
80 nodes.emplace_back(a["senderId"], a["senderName"], ForwardedMessage::deserializationFromMessageJson(a["messageChain"][1]["nodeList"]), a["time"]);
81 } else {
82 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageJson(a["messageChain"]), a["time"]);
83 }
84 }
85
86 } catch (const json::parse_error &) {
87 throw APIException("OnlineForwardedMessage格式化异常", MIRAICP_EXCEPTION_WHERE);
88 }
89 return ForwardedMessage(std::move(nodes), display);
90 }
