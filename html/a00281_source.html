132 void LoaderMain::mainloop() const noexcept {
133 auto tick_time = std::chrono::microseconds(US_IN_SEC / tickRate);
134 using Clock = std::chrono::steady_clock;
135 auto loop_start = Clock::now();
136
137 tick();
138
139 auto loop_end = Clock::now();
140 auto loop_duration = std::chrono::duration_cast<std::chrono::milliseconds>(loop_end - loop_start);
141 if (loop_duration >= tick_time) {
142 logger.warning("libLoader线程运行时间过长：" + std::to_string(loop_duration.count()) + "ms");
143 } else {
144 std::unique_lock cvUniqueLock(loaderCVLock);
145 auto diff_time = tick_time - loop_duration;
146 loaderWakeCV().wait_for(cvUniqueLock, diff_time,
147 [this]() { return is_loader_exited() || Scheduler::timeup() || !loader_thread_task_queue.empty(); });
148 cvUniqueLock.unlock();
149 }
150 }

159 void LoaderMain::process_task_queue() const { // NOLINT(*-convert-member-functions-to-static)
160 if (!loader_thread_task_queue.empty()) {
161 loadertask task;
162 {
163 std::lock_guard lk(task_mtx);
164 std::swap(task, loader_thread_task_queue.front());
165 loader_thread_task_queue.pop();
166 }
167 try {
168 switch (task.first) {
169 case LOADER_TASKS::ADD_THREAD:
170 loader_enablePluginById(task.second);
171 break;
172 case LOADER_TASKS::END_THREAD:
173 loader_disablePluginById(task.second);
174 break;
175 case LOADER_TASKS::ENABLE_ALL:
176 loader_enableAllPlugins();
177 break;
178 case LOADER_TASKS::DISABLE_ALL:
179 loader_disableAllPlugins();
180 break;
181 case LOADER_TASKS::LOAD_NEW_ACTIVATENOW:
182 loader_loadNewPlugin(task.second, true);
183 break;
184 case LOADER_TASKS::LOAD_NEW_DONTACTIVATE:
185 loader_loadNewPlugin(task.second, false);
186 break;
187 case LOADER_TASKS::UNLOAD:
188 loader_unloadPluginById(task.second);
189 break;
190 case LOADER_TASKS::RELOAD:
191 loader_reloadPluginById(task.second);
192 break;
193 case LOADER_TASKS::EXCEPTION_PLUGINEND:
194 loader_pluginEndByException(task.second);
195 break;
196 case LOADER_TASKS::RESET_THREAD:
197 loader_resetThreadByIndex(task.second);
198 break;
199 default:
200 throw std::exception();
201 }
202 } catch (LoaderBaseException &e) {
203 e.raise();
204 } catch (std::exception &e) {
205 logger.error("libLoader线程遇到无法预料的错误：" + std::string(e.what()) + "，程序将继续执行，如果发生意料之外的状况，请提交issue");
206 }
207 }
208 }

210 void LoaderMain::readConfig() {
211 nlohmann::json j = readJsonFile(config_path);
212 if (j.contains("loaderConfigs")) {
213 auto &loader_conf = j["loaderConfigs"];
214 if (loader_conf.contains("threadPoolSize")) {
215 size_t thread_pool_size = loader_conf["threadPoolSize"];
216 Antares::pool = std::make_unique<Antares::ThreadPool<>>(thread_pool_size);
217 }
218 if (loader_conf.contains("tickRate")) {
219 size_t _tickRate = loader_conf["tickRate"];
220 if (_tickRate > US_IN_SEC) {
221 throw InvalidConfException("tickRate", MIRAICP_EXCEPTION_WHERE);
222 }
223 if (_tickRate == 0) {
224 tickRate = 100;
225 } else {
226 tickRate = _tickRate;
227 }
228 }
229 }
