37 template<class T>
38 Message(T &&Arg) { // NOLINT(google-explicit-constructor)
39 using NoCVRefType = typename std::remove_cv_t<typename std::remove_reference_t<T>>;
40
41 if constexpr (std::is_base_of_v<Super, NoCVRefType>) {
42 *this = std::forward<T>(Arg);

44 reset(new NoCVRefType(std::forward<T>(Arg)));
45 } else {
46 static_assert(
47 std::is_base_of_v<Super, NoCVRefType> || std::is_base_of_v<SingleMessage, NoCVRefType>,

49 }

63 template<class T>
64 T getVal() const {
65 // for dev: 不用 get 为了不和shared_ptr重叠
66 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
67 if (T::type() != this->type())
68 throw IllegalArgumentException(
69 "cannot convert from " + SingleMessage::messageType[this->type()] + " to " +
70 SingleMessage::messageType[T::type()], MIRAICP_EXCEPTION_WHERE);
71 T *re = static_cast<T *>(std::shared_ptr<SingleMessage>::get());
72 if (re == nullptr)
73 throw IllegalArgumentException(
74 "cannot convert from " + SingleMessage::messageType[this->type()] + " to " +
75 SingleMessage::messageType[T::type()], MIRAICP_EXCEPTION_WHERE);
76 return *re;
77 }

246 static size_t findEnd(const std::string &s, size_t start) {
247 size_t pos = start;
248 while (pos < s.length()) {
249 switch (s[pos]) {
250 case '\\':
251 pos += 2;
252 continue;
253 case ']':
254 return pos;
255 }
256 pos++;
257 }
258 return -1;
259 }
