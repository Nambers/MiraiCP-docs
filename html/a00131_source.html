234 static size_t findEnd(const std::string &s, size_t start) {
235 size_t pos = start;
236 while (pos < s.length()) {
237 switch (s[pos]) {
238 case '\\':
239 pos += 2;
240 continue;
241 case ']':
242 return pos;
243 }
244 pos++;
245 }
246 return -1;
247 }
