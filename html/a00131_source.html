37 template<class T>
38 Message(T &&Arg) { // NOLINT(google-explicit-constructor)
39 using NoCVRefType = typename std::remove_cv_t<typename std::remove_reference_t<T>>;
40
41 if constexpr (std::is_base_of_v<Super, NoCVRefType>) {
42 *this = std::forward<T>(Arg);
43 } else if constexpr (std::is_base_of_v<SingleMessage, NoCVRefType>) {
44 reset(new NoCVRefType(std::forward<T>(Arg)));
45 } else {
46 static_assert(
47 std::is_base_of_v<Super, NoCVRefType> || std::is_base_of_v<SingleMessage, NoCVRefType>,
48 "只支持SingleMessage的子类");
49 }
50 }

52 explicit Message(Super msgptr) : Super(std::move(msgptr)) {}

54 public:
57 int type() const {
58 return (*this)->internalType;
59 };

63 template<class T>
64 T getVal() const {
65 // for dev: 不用 get 为了不和shared_ptr重叠
66 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
67 if (T::type() != this->type())
68 throw IllegalArgumentException(
69 "cannot convert from " + SingleMessage::getTypeString(this->type()) + " to " +
70 SingleMessage::getTypeString(T::type()),
71 MIRAICP_EXCEPTION_WHERE);
72 T *re = static_cast<T *>(std::shared_ptr<SingleMessage>::get());
73 if (re == nullptr)
74 throw IllegalArgumentException(
75 "cannot convert from " + SingleMessage::getTypeString(this->type()) + " to " +
76 SingleMessage::getTypeString(T::type()),
77 MIRAICP_EXCEPTION_WHERE);
78 return *re;
79 }

82 return (*this)->toMiraiCode();
83 }

86 return (*this)->internalType == m->internalType && (*this)->toMiraiCode() == m->toMiraiCode();
87 }
88
89 bool operator!=(const Message &m) const {
90 return (*this)->internalType != m->internalType || (*this)->toMiraiCode() != m->toMiraiCode();

248 static size_t findEnd(const std::string &s, size_t start) {
249 size_t pos = start;
250 while (pos < s.length()) {
251 switch (s[pos]) {
252 case '\\':
253 pos += 2;
254 continue;
255 case ']':
256 return pos;
257 }
258 pos++;
259 }
260 return -1;
261 }
