void onEnable() override {

m.kick("this_is_reason");
} catch (BotException &err) {
//权限不足
Logger::logger.error(err.what());
} catch (MemberException &err) {
if (err.type == 1) {
//找不到群
}
if (err.type == 2) {
//找不到群成员
}
}
});
}
};
void MiraiCP::enrollPlugin() {
MiraiCP::enrollPlugin(new Main());
}
