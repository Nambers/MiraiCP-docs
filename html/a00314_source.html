40void VerifyImpl(JSTRING _version, JSTRING _cfgPath) {
41 //初始化日志模块
42 MiraiCP::Redirector::start();
43
44 LibLoader::logger.info("⭐libLoader 版本: " + MiraiCP::MiraiCPVersion);
45 auto version = "v" + J_TO_STD_STRING(_version);
46 if (version != MiraiCP::MiraiCPVersion) {
47 LibLoader::logger.warning(
48 "libLoader(" + MiraiCP::MiraiCPVersion + ")版本和MiraiCP启动器(" + version + ")不一致, 建议更新至最新");
49 }
50
51 // 测试有效的插件
52 std::string config_path = J_TO_STD_STRING(_cfgPath);
53 LibLoader::LoaderMain::get().setConfigPath(config_path);
54 LibLoader::registerAllPlugin(config_path);
55
56 // 激活插件。创建loader thread。
57 // loader thread中创建多线程加载所有插件，调用入口函数
58 LibLoader::loaderThread = std::thread([](){
59 LibLoader::LoaderMain::get().loaderMain();
60 });
61}

63void EventImpl(JSTRING content) {
64 std::string str = J_TO_STD_STRING(content);
65 auto stringPtr = MiraiCP::MiraiCPString(str);
66
67 if (str.find(R"("eventId":1000)") != std::string::npos) {
68 LibLoader::builtInCommand(str);
69 return;
70 }
71
72 LibLoader::PluginManager::Get().broadcastToAllEnabledPlugins(stringPtr);
73}

75void PluginDisableImpl() {
76 LibLoader::LoaderMain::get().loaderExit();
77 {
78 LibLoader::loaderWakeCV().notify_one();
79 }
80 LibLoader::loaderThread.join();
81}

116int registerMethods(JNIEnv *env, const char *className, const JNINativeMethod *gMethods, int numMethods) {
117 jclass clazz = env->FindClass(className);
118 if (clazz == nullptr) {
119 return JNI_FALSE;
120 }
121 //注册native方法
122 if (env->RegisterNatives(clazz, gMethods, numMethods) < 0) {
123 return JNI_FALSE;
124 }
125 return JNI_TRUE;
126}
