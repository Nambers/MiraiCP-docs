90 void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg) {
91 if (re == "EF")
92 throw FriendException(name, line);
93 if (re == "EG")
94 throw GroupException(name, line);
95 if (re == "EM")
96 throw MemberException(1, name, line);
97 if (re == "EMM")
98 throw MemberException(2, name, line);
99 if (re == "EB")
100 throw BotException("找不到bot:" + re, name, line);
101 if (re == "EA")
102 throw APIException(ErrorMsg, name, line);
103 if (re == "EC")
104 throw EventCancelledException("发送信息被取消", name, line);
105 if (re == "ET")
106 throw TimeOutException("发送信息超时", name, line);
107 if (re == "EP")
108 throw BotException(name, line);
109 // equal to Tools::start_with
110 if (re.rfind("EBM", 0) == 0)
111 throw BotIsBeingMutedException(std::stoi(re.substr(3)), name, line);
112 }
