60 MessageSource Contact::sendVoice0(const std::string &path, JNIEnv *env) {
61 json j;
62 json source;
63 source["path"] = path;
64 j["source"] = source.dump();
65 j["contactSource"] = this->serializationToString();
66 std::string re = Config::koperation(Config::Voice, j, env);
67 if (re == "E1")
68 MiraiCPThrow(UploadException("上传语音文件格式不对(必须为.amr/.silk)或文件不存在"));
69 else if (re == "E2")
70 MiraiCPThrow(UploadException("上传语音文件大小超过服务器限制，一般限制在1MB上下"));
71 return MessageSource::deserializeFromString(re);
