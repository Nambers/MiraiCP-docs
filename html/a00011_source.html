99 template<class T>
100 bool registerCommand(T command) {
101 static_assert(std::is_base_of_v<IRawCommand, T>, "只支持IRawCommand的派生类");
102 nlohmann::json j;
103 j["pluginId"] = Config::pluginId;
104 j["usage"] = command.config().usage;
105 j["primaryName"] = command.config().primaryName;
106 j["secondName"] = command.config().secondNames;
107 j["description"] = command.config().description;
108 j["override"] = command.config().override;
109 j["preFixOption"] = command.config().preFixOption;
110 size_t before = commandList.size();
111 std::shared_ptr<IRawCommand> c;
112 c.reset(new T(command));
113 commandList.push_back(c);
114 size_t now = commandList.size();
115 if (now - before == 1)
116 j["bindId"] = now - 1;
117 else {
118 auto i = std::find(commandList.begin(), commandList.end(), c);
119 if (i != commandList.end())
120 j["bindId"] = i - commandList.begin();
121 else
122 throw IllegalArgumentException("找不到合适的bindId", MIRAICP_EXCEPTION_WHERE);
123 }
124 nlohmann::json rej;
125 rej["command"] = j.dump();
126 std::string re = Config::koperation(Config::CommandReg, rej);
127 return re == "true";
128 }
129 static CommandManager commandManager;
130 };
