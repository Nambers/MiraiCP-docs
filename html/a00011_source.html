45// detect platform
46// ref: https://stackoverflow.com/questions/5919996/how-to-detect-reliably-mac-os-x-ios-linux-windows-in-c-preprocessor

48#undef MIRAICP_WINDOWS
49#define MIRAICP_WINDOWS 1
50#elif __APPLE__

54#define MIRAICP_IOS 1
55#elif TARGET_OS_MAC
56#undef MIRAICP_MACOS
57#define MIRAICP_MACOS 1
58#else
59static_assert(false, "Unknown apple platform");
60#endif
61#elif __ANDROID__
62#ifdef __TERMUX__
63#undef MIRAICP_TERMUX
64#define MIRAICP_TERMUX 1
65#else
66#undef MIRAICP_ANDROID

69#elif __linux__
70#undef MIRAICP_LINUX
71#define MIRAICP_LINUX 1
72#elif __unix__
73#undef MIRAICP_UNIX

189// getter
190// need to define macro LOC_CLASS_NAMESPACE to the class first!
191#define DECL_GETTER(attr) decltype(DataType::_##attr) attr();
192#define IMPL_GETTER(attr) \
193 decltype(LOC_CLASS_NAMESPACE::DataType::_##attr) LOC_CLASS_NAMESPACE::attr() { \
194 InternalData->requestRefresh(); \
195 MIRAICP_DATALOCK; \
196 return GetDataInternal()->_##attr; \
197 }
198#define INLINE_GETTER(attr) \
199 auto attr() { return GetDataInternal()->_##attr; }

206// operator++ of iterable enum
207#define MIRAICP_ITERABLE_ENUM_OPERATOR_PLUSPLUS \
208 inline Type &operator++(Type &value) { \
209 value = (Type) ((int) value + 1); \
210 return value; \
211 } \
212 inline Type operator++(Type &value, int) { \
213 Type result = value; \
214 value = (Type) ((int) value + 1); \
215 return result; \
216 }

225 Begin = BeginAt, \
226 End = Count, \
