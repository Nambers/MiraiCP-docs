36 MessageChain PrivateMessageEvent::nextMessage(long time, bool halt) const {
37 json j;
38 j["contactSource"] = this->sender.toString();
39 j["time"] = time;
40 j["halt"] = halt;
41 std::string r = KtOperation::ktOperation(KtOperation::NextMsg, j);
42 if (r == "E1")
43 throw TimeOutException("取下一条信息超时", MIRAICP_EXCEPTION_WHERE);
44 json re = json::parse(r);
45 return MessageChain::deserializationFromMessageSourceJson(json::parse(re["messageSource"].get<std::string>())).plus(MessageSource::deserializeFromString(re["messageSource"]));

91 void Event::incomingEvent(json j, int type) {
92 switch (type) {
93 case eventTypes::GroupMessageEvent: {
94 //GroupMessage
95 Event::broadcast<GroupMessageEvent>(
96 GroupMessageEvent(j["group"]["botid"],
97 Group(Group::deserialize(j["group"])),
98 Member(Member::deserialize(j["member"])),
99 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
100 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
101 break;
102 }
103 case eventTypes::PrivateMessageEvent: {
104 //私聊消息
105 Event::broadcast<PrivateMessageEvent>(
106 PrivateMessageEvent(j["friend"]["botid"],
107 Friend(Friend::deserialize(j["friend"])),
108 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
109 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
110 break;
111 }
112 case eventTypes::GroupInviteEvent:
113 //群聊邀请
114 Event::broadcast<GroupInviteEvent>(
115 GroupInviteEvent(
116 j["source"]["botid"],
117 j["request"],
118 j["source"]["inviternick"],
119 j["source"]["inviterid"],
120 j["source"]["groupname"],
121 j["source"]["groupid"]));
122 break;
123 case eventTypes::NewFriendRequestEvent:
124 //好友
125 Event::broadcast<NewFriendRequestEvent>(
126 NewFriendRequestEvent(
127 j["source"]["botid"],
128 j["request"],
129 j["source"]["fromid"],
130 j["source"]["fromgroupid"],
131 j["source"]["fromnick"],
132 j["source"]["message"]));
133 break;
134 case eventTypes::MemberJoinEvent:
135 //新成员加入
136 Event::broadcast<MemberJoinEvent>(
137 MemberJoinEvent(
138 j["group"]["botid"],
139 j["jointype"],
140 Member(Member::deserialize(j["member"])),
141 Group(Group::deserialize(j["group"])),
142 j["inviterid"]));
143 break;
144 case eventTypes::MemberLeaveEvent:
145 //群成员退出
146 Event::broadcast<MemberLeaveEvent>(MemberLeaveEvent(
147 j["group"]["botid"],
148 j["leavetype"],
149 j["memberid"],
150 Group(Group::deserialize(j["group"])),
151 j["operatorid"]));
152 break;
153 case eventTypes::RecallEvent:
154 Event::broadcast<RecallEvent>(RecallEvent(
155 j["botid"],
156 j["etype"],
157 j["time"],
158 j["authorid"],
159 j["operatorid"],
160 j["ids"],
161 j["internalids"],
162 j["groupid"]));
163 break;
164 case eventTypes::BotJoinGroupEvent:
165 Event::broadcast<BotJoinGroupEvent>(BotJoinGroupEvent(
166 j["group"]["botid"],
167 j["etype"],
168 Group(Group::deserialize(j["group"])),
169 j["inviterid"]));
170 break;
171 case eventTypes::GroupTempMessageEvent:
172 Event::broadcast<GroupTempMessageEvent>(GroupTempMessageEvent(
173 j["group"]["botid"],
174 Group(Group::deserialize(j["group"])),
175 Member(Member::deserialize(j["member"])),
176 MessageChain::deserializationFromMessageSourceJson(json::parse(j["message"].get<std::string>()))
177 .plus(MessageSource::deserializeFromString(j["source"]))));
178 break;
179 case eventTypes::TimeOutEvent:
180 Event::broadcast(TimeOutEvent(j["msg"]));
181 break;
182 case eventTypes::BotOnlineEvent:
183 Event::broadcast(BotOnlineEvent(j["botid"]));
184 break;
185 case eventTypes::NudgeEvent:
186 Event::broadcast(NudgeEvent(Contact::deserialize(j["from"]),
187 Contact::deserialize(j["target"]),
188 Contact::deserialize(j["subject"]),
189 j["botid"]));
190 break;
191 case eventTypes::BotLeaveEvent:
192 Event::broadcast(BotLeaveEvent(j["groupid"], j["botid"]));
193 break;
194 case eventTypes::MemberJoinRequestEvent: {
195 std::optional<Group> a;
196 std::optional<Member> b;
197 Contact temp = Contact::deserialize(j["group"]);
198 if (temp.id() == 0)
199 a = std::nullopt;
200 else
201 a = Group(temp);
202 temp = Contact::deserialize(j["inviter"]);
203 if (temp.id() == 0)
204 b = std::nullopt;
205 else
206 b = Member(temp);
207 Event::broadcast(MemberJoinRequestEvent(a, b, temp.botid(), j["requester"], j["requestData"]));
208 break;
209 }
210 case eventTypes::MessagePreSendEvent: {
211 Event::broadcast(MessagePreSendEvent(Contact::deserialize(j["target"]), MessageChain::deserializationFromMessageSourceJson(j["message"].get<std::string>(), false), j["botid"]));
212 break;
213 }
214 case eventTypes::Command: {
215 // command
216 std::optional<Contact> c = std::nullopt;
217 if (j.contains("contact")) c = Contact::deserialize(j["contact"]);
218 CommandManager::commandManager[j["bindId"]]->onCommand(c, Bot(j["botid"]), MessageChain::deserializationFromMessageSourceJson((j.contains("message") ? j["message"].get<std::string>() : ""), false));
219 break;
220 }
221 default:
222 throw APIException("Unreachable code", MIRAICP_EXCEPTION_WHERE);
223 }
224 }
