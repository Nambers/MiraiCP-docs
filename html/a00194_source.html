153 template<typename T1, typename T2, typename T = std::common_type_t<T1, T2>>
154 class [[nodiscard]] blocks {
155 public:
163 blocks(const T1 first_index_, const T2 index_after_last_, const size_t num_blocks_) : first_index(static_cast<T>(first_index_)), index_after_last(static_cast<T>(index_after_last_)), num_blocks(num_blocks_) {
164 if (index_after_last < first_index)
165 std::swap(index_after_last, first_index);
166 total_size = static_cast<size_t>(index_after_last - first_index);
167 block_size = static_cast<size_t>(total_size / num_blocks);
168 if (block_size == 0) {
169 block_size = 1;
170 num_blocks = (total_size > 1) ? total_size : 1;
171 }
172 }
173
180 [[nodiscard]] T start(const size_t i) const {
181 return static_cast<T>(i * block_size) + first_index;
182 }
183
190 [[nodiscard]] T end(const size_t i) const {
191 return (i == num_blocks - 1) ? index_after_last : (static_cast<T>((i + 1) * block_size) + first_index);
192 }
193
199 [[nodiscard]] size_t get_num_blocks() const {
200 return num_blocks;
201 }
202
208 [[nodiscard]] size_t get_total_size() const {
209 return total_size;
210 }

216 size_t block_size = 0;
217

222
226 T index_after_last = 0;
227
231 size_t num_blocks = 0;
232
236 size_t total_size = 0;
237 };

336 template<typename It, typename R = std::invoke_result_t<std::decay_t<decltype(*std::declval<It>())>>>
337 [[nodiscard]] multi_future<R> run_over(It &&begin, It &&end, size_t inSize) {
338 It ptr = begin;
339 if (inSize) {
340 multi_future<R> mf(inSize);
341 {
342 std::scoped_lock lk(tasks_mutex);
343 size_t i = 0;
344 while (ptr != end && i < inSize) {
345 mf[i++] = submit_nolock(*ptr);
346 }
347 }
348 return mf;
349 } else {
350 return multi_future<R>();
351 }

369 [[nodiscard]] multi_future<R> parallelize_loop(const T1 first_index, const T2 index_after_last, F &&loop, const size_t num_blocks = 0) {
370 blocks blks(first_index, index_after_last, num_blocks ? num_blocks : thread_count);
371 if (blks.get_total_size() > 0) {
372 multi_future<R> mf(blks.get_num_blocks());
373 {
374 std::scoped_lock lk(tasks_mutex);
375 for (size_t i = 0; i < blks.get_num_blocks(); ++i)
376 mf[i] = submit_nolock(std::forward<F>(loop), blks.start(i), blks.end(i));
377 }
378 return mf;
379 } else {
380 return multi_future<R>();
381 }

500 template<typename F, typename... A, typename R = std::invoke_result_t<std::decay_t<F>, std::decay_t<A>...>>
501 [[nodiscard]] std::future<R> submit(F &&task, A &&...args) {
502 auto task_promise = new std::promise<R>;
503 auto future = task_promise->get_future();
504 push_task(
505 [task_promise](auto &&task_inner, auto &&...argss) {
506 try {
507 if constexpr (std::is_void_v<R>) {
508 std::invoke(std::forward<decltype(task_inner)>(task_inner), std::forward<decltype(argss)>(argss)...);
509 task_promise->set_value();
510 } else {
511 task_promise->set_value(std::invoke(std::forward<decltype(task_inner)>(task_inner), std::forward<decltype(argss)>(argss)...));
512 }
513 } catch (...) {
514 try {
515 task_promise->set_exception(std::current_exception());
516 } catch (...) {
517 }
518 }
519 delete task_promise;
520 },
521 std::forward<F>(task), std::forward<A>(args)...);
522 return future;
523 }

592 void create_threads() {
593 running = true;
594 threadIndexCounter = 0;
595 for (concurrency_t i = 0; i < thread_count; ++i) {
596 while (threadIndexCounter != i) [[unlikely]] std::this_thread::sleep_for(std::chrono::milliseconds(10));
597 threads[i] = std::thread(&thread_pool::worker, this);
598 }
599 }

604 void destroy_threads() {
605 running = false;
606 task_available_cv.notify_all();
607 for (concurrency_t i = 0; i < thread_count; ++i) {
608 threads[i].join();
609 }

642 void worker() {
643 ThreadIdentify::IAmPoolThread();
644 setThreadIndex();
645 while (running) {
646 std::function<void()> task;
647 std::unique_lock<std::mutex> tasks_lock(tasks_mutex);
648 task_available_cv.wait(tasks_lock, [this] { return !tasks.empty() || !running; });
649 if (running && !paused) {
650 task = std::move(tasks.front());
651 tasks.pop();
652 tasks_lock.unlock();
653 task();
654 tasks_lock.lock();
655 --tasks_total;
656 if (waiting)
657 task_done_cv.notify_one();
658 }
659 }
