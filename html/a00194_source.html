69 void builtInCommand(const std::string &cmd) {
70 using nlohmann::json;
71 json j = json::parse(cmd);
72 auto it = actions.find(j["name"]);
73 if (it == actions.end()) {
74 logger.error("builtInCommand: unknown command: " + j["name"].get<std::string>());
75 return;
76 }
77 it->second(get_or_empty(j, "content"));
78 }
