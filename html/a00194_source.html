27 // avoid calling this if _size == 0
28 void MiraiCPString::construction() {
29 str = (char *) ::std::malloc(sizeof(char) * (_size + 1));
30 if (str == nullptr) {
31#ifdef MIRAICP_LIB_LOADER
32 LibLoader::logger.error("MiraiCPString::construction: malloc failed when trying to malloc size " + std::to_string(_size + 1));
33#endif
34 throw std::bad_alloc();
35 }
36 }

38 MiraiCPString::~MiraiCPString() { // do not inherit MiraiCPString!!
39 if (str != nullptr) {
40 // ensure deconstruction is paired to construction
41 free_this(str);
42 str = nullptr;
43 }
44 }

46 MiraiCPString::MiraiCPString(const MiraiCPString &other) {
47 _size = other._size;
48 if (_size == 0) return;
49 construction();
50 assert(str != nullptr);
51 memcpy(str, other.str, _size * sizeof(char));
52 str[_size] = 0;
53 }

59 MiraiCPString::MiraiCPString(const char *char_str) {
60 if (char_str == nullptr) return;
61 _size = strlen(char_str);
62 if (0 == _size) return;
63 construction();
64 assert(str != nullptr);
65 memcpy(str, char_str, _size * sizeof(char));
66 str[_size] = 0;
67 }

69 MiraiCPString::MiraiCPString(const std::string &string_str) {
70 _size = string_str.size();
71 if (!_size) return;
72 construction();
73 assert(str != nullptr);
74 memcpy(str, string_str.c_str(), _size * sizeof(char));
75 str[_size] = 0;
76 }

78 const char *MiraiCPString::copyToCharPtr() const {
79 if (str == nullptr || _size == 0) return new char[1]{0};
80 char *t = new char[_size + 1];
81 memcpy(t, str, (_size + 1) * sizeof(char));
82 return t;
83 }

85 bool MiraiCPString::operator==(const MiraiCPString &another) const {
86 return another._size == _size && (_size == 0 || strcmp(another.str, str) == 0);
