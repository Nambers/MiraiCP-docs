47 template<typename F, typename... Args, typename R = std::invoke_result_t<std::decay_t<F>, std::decay_t<Args>...>>
48 std::future<R> promiseTask(F &&func, Args &&...args) {
49 auto promise = std::make_shared<std::promise<R>>();
50 auto future = promise->get_future();
51 std::function<R()> taskfunction = std::bind(std::forward<F>(func), std::forward<Args>(args)...);
52 size_t function_id = internal::get_auto_incr_id();
53 auto to_push_func = std::make_shared<std::function<void()>>(
54 [promise = std::move(promise), function_id, taskfunction = std::move(taskfunction)]() {
55 try {
56 if constexpr (std::is_void_v<R>) {
57 std::invoke(taskfunction);
58 promise->set_value();
59 } else {
60 promise->set_value(std::invoke(taskfunction));
61 }
62 } catch (...) {
63 try {
64 promise->set_exception(std::current_exception());
65 internal::logerror("Failed to run task: " + std::to_string(function_id));
66 } catch (...) {
67 }
68 }
69 try {
70 internal::remove_task(function_id);
71 } catch (...) {
72 }
73 });
74 internal::push_task(function_id, std::move(to_push_func));
75 return future;
76 }

81 template<typename F, typename... Args, typename R = std::invoke_result_t<std::decay_t<F>, std::decay_t<Args>...>>
82 void addTask(F &&func, Args &&...args) {
83 if constexpr (std::is_same_v<std::decay_t<F>, void (*)()>) {
84 internal::raw_push_task(func);
85 } else {
86 auto function_id = internal::get_auto_incr_id();
87 std::function<void()> taskfunction = std::bind(std::forward<F>(func), std::forward<Args>(args)...);
88 auto to_push_func = std::make_shared<std::function<void()>>(
89 [function_id, taskfunction = std::move(taskfunction)]() {
90 try {
91 std::invoke(taskfunction);
92 } catch (...) {
93 try {
94 internal::logerror("Failed to run task: " + std::to_string(function_id));
95 } catch (...) {
96 }
97 }
98 try {
99 internal::remove_task(function_id);
100 } catch (...) {
101 }
102 });
103 internal::push_task(function_id, std::move(to_push_func));
