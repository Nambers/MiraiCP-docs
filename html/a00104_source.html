68 return "[mirai:atall] ";

194 RemoteFile RemoteFile::deserializeFromString(const std::string &source) {
195 json j;
196 try {
197 j = json::parse(source);
198 } catch (json::parse_error &e) {
199 Logger::logger.error("格式化json失败，RemoteFile::deserializeFromString");
200 Logger::logger.error(source);
201 Logger::logger.error(e.what());
202 throw e;
203 }
204 try {
205 auto re = RemoteFile(j["id"], j["internalid"], j["name"], j["finfo"]["size"]);
206 if (j.contains("dinfo")) {
207 struct Dinfo d {
208 j["dinfo"]["url"],
209 j["dinfo"]["md5"],
210 j["dinfo"]["sha1"]
211 };
212 re.dinfo = d;
213 }
214 if (j["finfo"].contains("uploaderid")) {
215 struct Finfo f {
216 j["finfo"]["size"],
217 j["finfo"]["uploaderid"],
218 j["finfo"]["expirytime"],
219 j["finfo"]["uploadtime"],
220 j["finfo"]["lastmodifytime"]
221 };
222 re.finfo = f;
223 }
224 if (j.contains("path"))
225 re.path = j["path"];
226 return re;
227 } catch (json::type_error &e) {
228 Logger::logger.error("json格式化失败，位置:RemoteFile");
229 Logger::logger.error(source);
230 Logger::logger.error(e.what());
231 throw e;
232 }
233 }

242 std::string RemoteFile::serializeToString() {
243 json j;
244 if (this->dinfo.has_value()) {
245 j["dinfo"]["url"] = this->dinfo->url;
246 j["dinfo"]["md5"] = this->dinfo->md5;
247 j["dinfo"]["shar1"] = this->dinfo->sha1;
248 }
249 if (this->finfo.has_value()) {
250 j["finfo"]["size"] = this->finfo->size;
251 j["finfo"]["uploaderid"] = this->finfo->uploaderid;
252 j["finfo"]["expirytime"] = this->finfo->expirytime;
253 j["finfo"]["uploadtime"] = this->finfo->uploadtime;
254 j["finfo"]["lastmodifytime"] = this->finfo->lastmodifytime;
255 }
256 j["id"] = this->id;
257 j["internalid"] = this->internalid;
258 j["name"] = this->name;
259 j["size"] = this->size;
260 if (this->path.has_value())
261 j["path"] = this->path.value();
262 return j.dump();
263 }

267 std::string re = Config::koperation(Config::QueryImgInfo, this->toJson(), env);
268 if (re == "E1")

271 this->url = j["url"];
272 this->md5 = j["md5"];
273 this->size = j["size"];
274 this->width = j["width"];
275 this->height = j["height"];
276 this->imageType = j["type"];
