43 MessageChain MessageChain::deserializationFromMiraiCode(const std::string &m) {
44 size_t pos = 0;
45 size_t lastPos = -1;
46 MessageChain mc;
47 if (m.length() <= 7) {
48 return MessageChain(PlainText(m));
49 }
50 do {
51 if (m.length() - 7 - pos > 0 && m.substr(pos, 7) == "[mirai:") {
52 if (pos - lastPos > 1)
53 mc.add(PlainText(m.substr(lastPos + 1, pos - lastPos - 1))); // plain text
54 size_t back = MessageChain::findEnd(m, pos);
55 if (back == (size_t) -1) throw IllegalStateException("", MIRAICP_EXCEPTION_WHERE);
56 std::string tmp = m.substr(pos, back - pos);
57 tmp = Tools::replace(std::move(tmp), "[mirai:", "");
58 size_t i = tmp.find(':'); // first :
59 int t = SingleMessage::getKey(tmp.substr(0, i));
60 switch (t) {
61 case 0:
62 // no miraiCode key is PlainText
63 Logger::logger.error("无法预料的错误, 信息: " + m);
64 break;
65 case 1:
66 mc.add(At(std::stoll(tmp.substr(i + 1, tmp.length() - i - 1))));
67 break;
68 case 2:
69 mc.add(AtAll());
70 break;
71 case 3:
72 mc.add(Image(tmp.substr(i + 1, tmp.length() - i - 1)));
73 break;
74 case 4:
75 mc.add(LightApp(tmp.substr(i + 1, tmp.length() - i - 1)));
76 break;
77 case 5: {
78 size_t comma = tmp.find(',');
79 mc.add(ServiceMessage(std::stoi(tmp.substr(i + 1, comma - i - 1)),
80 tmp.substr(comma + 1, tmp.length() - comma - 1)));
81 break;
82 }
83 case 6: {
84 //[mirai:file:/b53231e8-46dd-11ec-8ba5-5452007bd6c0,102,run.bat,55]
85 size_t comma1 = tmp.find(',');
86 size_t comma2 = tmp.find(',', comma1 + 1);
87 size_t comma3 = tmp.find(',', comma2 + 1);
88 mc.add(RemoteFile(tmp.substr(i + 1, comma1 - i - 1),
89 std::stoi(tmp.substr(comma1 + 1, comma2 - comma1 - 1)),
90 tmp.substr(comma2 + 1, comma3 - comma2 - 1),
91 std::stoll(tmp.substr(comma3 + 1, tmp.length() - comma3 - 1))));
92 break;
93 }
94 case 7:
95 mc.add(Face(std::stoi(tmp.substr(i + 1, tmp.length() - i - 1))));
96 break;
97 case 8:
98 mc.add(FlashImage(tmp.substr(i + 1, tmp.length() - i - 1)));
99 break;
100 case 9: {
101 //[mirai:musicshare:name,title,summary,jUrl,pUrl,mUrl,brief]
102 auto temp = Tools::split(tmp.substr(i + 1, tmp.length() - i - 1), ",");
103 mc.add(MusicShare(temp[0], temp[1], temp[2], temp[3], temp[4], temp[5], temp[6]));
104 break;
105 }
106 default:
107 Logger::logger.error(
108 "MiraiCP碰到了意料之中的错误(原因:部分SimpleMessage在MiraiCode解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MiraiCode:" +
109 m);
110 mc.add(UnSupportMessage("[mirai:" + tmp));
111 break;
112 }
113 pos = back;
114 lastPos = pos;
115 if (t == 1)
116 lastPos++;
117 }
118 pos++;
119 } while (pos < m.length());
120 if (lastPos + 1 < m.length())
121 mc.add(PlainText(m.substr(lastPos + 1, m.length() - lastPos - 1))); // plain text
122 return mc;
123 }

125 MessageChain MessageChain::deserializationFromMessageSourceJson(const json &tmp, bool origin) {
126 json j = tmp;
127 if (origin)
128 j = tmp["originalMessage"];
129 MessageChain mc;
130 if (j.empty()) return mc;
131 if (j[0]["type"] == "MessageOrigin") {
132 if (j[0]["kind"] == "MUSIC_SHARE") {
133 mc.add(MusicShare(j[1]["kind"], j[1]["title"], j[1]["summary"], j[1]["jumpUrl"], j[1]["pictureUrl"],
134 j[1]["musicUrl"], j[1]["brief"]));
135 return mc;
136 }
137 mc.add(OnlineForwardedMessage::deserializationFromMessageSourceJson(j));
138 return mc;
139 }
140 for (auto node: j) {
141 if (node["type"] == "SimpleServiceMessage") {
142 mc.add(ServiceMessage(node["serviceId"], node["content"]));
143 continue;
144 }
145 if (node["type"] == "LightApp") {
146 mc.add(LightApp(node["content"]));
147 continue;
148 }
149 if (node["type"] == "OnlineAudio") {
150 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
151 node["urlForDownload"]));
152 continue;
153 }
154 if (node["type"] == "FileMessage") {
155 mc.add(Group(tmp["targetId"].get<QQID>(), tmp["botId"].get<QQID>()).getFileById(node["id"]).plus(
156 node["internalId"]));
157 continue;
158 }
159 if (node["type"] == "MarketFace") {
160 mc.add(MarketFace(node["delegate"]["faceId"]));
161 break;
162 }
163 switch (SingleMessage::getKey(node["type"])) {
164 case -2:
165 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
166 break;
167 case 0:
168 mc.add(PlainText(node["content"].get<std::string>()));
169 break;
170 case 1:
171 mc.add(At(node["target"]));
172 break;
173 case 2:
174 mc.add(AtAll());
175 break;
176 case 3:
177 mc.add(Image(node["imageId"]));
178 break;
179 case 7:
180 mc.add(Face(node["id"]));
181 break;
182 case 8:
183 mc.add(FlashImage(node["imageId"]));
184 break;
185 default:
186 Logger::logger.warning(
187 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MessageSource:" +
188 j.dump());
189 mc.add(UnSupportMessage(node.dump()));
190 }
191 }
192 return mc;
193 }
