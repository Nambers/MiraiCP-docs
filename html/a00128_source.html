24 void MessageSource::recall(JNIEnv *env) const {

26 j["source"] = this->serializeToString();
27 std::string re = Config::koperation(Config::Recall, j, env);

43 MessageSource MessageSource::deserializeFromString(const std::string &source) {
44 json j = json::parse(source);
45 try {
46 return {j["ids"].dump(), j["internalIds"].dump(), source};
47 } catch (json::type_error &e) {
48 Logger::logger.error("消息源序列化出错，格式不符合(MessageSource::deserializeFromString)");
49 Logger::logger.error(source);
50 Logger::logger.error(e.what());
51 throw IllegalArgumentException(std::string("消息源序列化出错，格式不符合(MessageSource::deserializeFromString), ") + e.what(), MIRAICP_EXCEPTION_WHERE);
52 }
