60 std::string SingleMessage::toMiraiCode() const {
61 if (internalType > 0)
62 if (internalType == Types::At_t)
63 return "[mirai:at:" + content + "] ";
64 else if (internalType == Types::AtAll_t)
65 return "[mirai:atall] ";
66 else
67 return "[mirai:" + getTypeString(internalType) + prefix + Tools::escapeToMiraiCode(content) + "]";
68 else
69 return content;
70 }

127 void Image::refreshInfo() {

131 json j = json::parse(re);
132 this->url = json_stringmover(j, "url");
133 this->md5 = json_stringmover(j, "md5");
134 this->size = j["size"];
135 this->width = j["width"];
136 this->height = j["height"];
137 this->imageType = j["imageType"];
138 this->isEmoji = j["isEmoji"];

343 RemoteFile RemoteFile::deserializeFromString(const std::string &source) {
344 json j;
345 try {
346 j = json::parse(source);
347 } catch (json::parse_error &e) {
348 Logger::logger.error("格式化json失败，RemoteFile::deserializeFromString");
349 Logger::logger.error(source);
350 Logger::logger.error(e.what());
351 throw e;
352 }
353 try {
354 auto re = RemoteFile(j["id"], j["internalId"], j["name"], j["detailInfo"]["size"]);
355 if (j.contains("downloadInfo")) {
356 auto tmp = json_jsonmover(j, "downloadInfo");
357 struct Dinfo d {
358 json_jsonmover(tmp, "url"),
359 json_jsonmover(tmp, "md5"),
360 json_jsonmover(tmp, "sha1")
361 };
362 re.dinfo = d;
363 }
364 if (j["detailInfo"].contains("uploaderId")) {
365 auto tmp = json_jsonmover(j, "detailInfo");
366 struct Finfo f {
367 json_jsonmover(tmp, "size"),
368 json_jsonmover(tmp, "uploaderId"),
369 json_jsonmover(tmp, "expiryTime"),
370 json_jsonmover(tmp, "uploadTime"),
371 json_jsonmover(tmp, "lastModifyTime")
372 };
373 re.finfo = f;
374 }
375 if (j.contains("path"))
376 re.path = json_jsonmover(j, "path");
377 return re;
378 } catch (json::type_error &e) {
379 Logger::logger.error("json格式化失败，位置:RemoteFile");
380 Logger::logger.error(source);
381 Logger::logger.error(e.what());
382 throw e;
383 }
384 }

394 std::string RemoteFile::serializeToString() {
395 json j;
396 if (this->dinfo.has_value()) {
397 j["dinfo"]["url"] = this->dinfo->url;
398 j["dinfo"]["md5"] = this->dinfo->md5;
399 j["dinfo"]["shar1"] = this->dinfo->sha1;
400 }
401 if (this->finfo.has_value()) {
402 j["finfo"]["size"] = this->finfo->size;
403 j["finfo"]["uploaderid"] = this->finfo->uploaderid;
404 j["finfo"]["expirytime"] = this->finfo->expirytime;
405 j["finfo"]["uploadtime"] = this->finfo->uploadtime;
406 j["finfo"]["lastmodifytime"] = this->finfo->lastmodifytime;
407 }
408 j["id"] = this->id;
409 j["internalid"] = this->internalid;
410 j["name"] = this->name;
411 j["size"] = this->size;
412 if (this->path.has_value())
413 j["path"] = this->path.value();
414 return j.dump();
415 }
