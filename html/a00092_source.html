41 RemoteFile RemoteFile::deserializeFromString(const std::string &source) {
42 json j;
43 try {
44 j = json::parse(source);
45 } catch (json::parse_error &e) {
46 Logger::logger.error("格式化json失败，RemoteFile::deserializeFromString");
47 Logger::logger.error(source);
48 Logger::logger.error(e.what());
49 throw e;
50 }
51 try {
52 auto re = RemoteFile(j["id"], j["internalid"], j["name"], j["finfo"]["size"]);
53 if (j.contains("dinfo")) {
54 struct Dinfo d {
55 j["dinfo"]["url"],
56 j["dinfo"]["md5"],
57 j["dinfo"]["sha1"]
58 };
59 re.dinfo = d;
60 }
61 if (j["finfo"].contains("uploaderid")) {
62 struct Finfo f {
63 j["finfo"]["size"],
64 j["finfo"]["uploaderid"],
65 j["finfo"]["expirytime"],
66 j["finfo"]["uploadtime"],
67 j["finfo"]["lastmodifytime"]
68 };
69 re.finfo = f;
70 }
71 if (j.contains("path"))
72 re.path = j["path"];
73 return re;
74 } catch (json::type_error &e) {
75 Logger::logger.error("json格式化失败，位置:RemoteFile");
76 Logger::logger.error(source);
77 Logger::logger.error(e.what());
78 throw e;
79 }
80 }

82 std::string RemoteFile::serializeToString() {
83 json j;
84 if (this->dinfo.has_value()) {
85 j["dinfo"]["url"] = this->dinfo->url;
86 j["dinfo"]["md5"] = this->dinfo->md5;
87 j["dinfo"]["shar1"] = this->dinfo->sha1;
88 }
89 if (this->finfo.has_value()) {
90 j["finfo"]["size"] = this->finfo->size;
91 j["finfo"]["uploaderid"] = this->finfo->uploaderid;
92 j["finfo"]["expirytime"] = this->finfo->expirytime;
93 j["finfo"]["uploadtime"] = this->finfo->uploadtime;
94 j["finfo"]["lastmodifytime"] = this->finfo->lastmodifytime;
95 }
96 j["id"] = this->id;
97 j["internalid"] = this->internalid;
98 j["name"] = this->name;
99 j["size"] = this->size;
100 if (this->path.has_value())
101 j["path"] = this->path.value();
102 return j.dump();
103 }
