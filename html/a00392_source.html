53MIRAICP_EXPORT int FUNC_ENTRANCE(const LibLoader::LoaderApi::interface_funcs &funcs) {
54 static_assert(std::is_same_v<decltype(&FUNC_ENTRANCE), LibLoader::plugin_entrance_func_ptr>);
55 using namespace MiraiCP;
56
57 Event::clear();
58 LibLoader::LoaderApi::set_loader_apis(&funcs);
59 assert(LibLoader::LoaderApi::get_loader_apis() != nullptr);
60 // logger api is set
61
62 try {
63 Logger::logger.info("开始启动插件: " + MiraiCP::CPPPlugin::config.getId());
64 enrollPlugin();
65 // plugin == nullptr 无插件实例加载
66 if (CPPPlugin::plugin != nullptr) {
67 CPPPlugin::plugin->onEnable();
68 }
69 } catch (const MiraiCPExceptionBase &e) {
70 try {
71 std::cerr.flush();
72 e.raise();
73 Logger::logger.error("插件(id=" + CPPPlugin::config.getId() + ", name=" + CPPPlugin::config.name + ")启动失败");
74 } catch (...) {}
75 return PLUGIN_ERROR;
76 } catch (const std::exception &e) {
77 try {
78 std::cerr.flush();
79 Logger::logger.error(e.what());
80 Logger::logger.error("插件(id=" + CPPPlugin::config.getId() + ", name=" + CPPPlugin::config.name + ")启动失败");
81 } catch (...) {}
82 return PLUGIN_ERROR;
83 } catch (...) {
84 try {
85 std::cerr.flush();
86 Logger::logger.error("插件(id=" + CPPPlugin::config.getId() + ", name=" + CPPPlugin::config.name + ")启动失败");
87 } catch (...) {}
88 return PLUGIN_ERROR;
89 }
90 return PLUGIN_NORMAL;
91}

95MIRAICP_EXPORT int FUNC_EXIT() {
96 static_assert(std::is_same_v<decltype(&FUNC_EXIT), LibLoader::plugin_func_ptr>);
97
98 using namespace MiraiCP;
99
100 if (LibLoader::LoaderApi::get_loader_apis() == nullptr) return PLUGIN_NORMAL;
101
102 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.info("开始禁用插件：" + MiraiCP::CPPPlugin::config.getId());)
103 Event::clear();
104
105 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(if (CPPPlugin::plugin != nullptr) CPPPlugin::plugin->onDisable();)
106 CPPPlugin::plugin.reset();
107
108 // 无法保证用户插件析构函数是否调用api，在plugin.reset()之前不可reset loader api
109 LibLoader::LoaderApi::reset_loader_apis();
110
111 return PLUGIN_NORMAL;
112}

118MIRAICP_EXPORT int FUNC_EVENT(const MiraiCP::MiraiCPString &c) {
119 static_assert(std::is_same_v<decltype(&FUNC_EVENT), LibLoader::plugin_event_func_ptr>);
120
121 using namespace MiraiCP;
122 std::string content = c;
123 json j;
124
125 try {
126 j = json::parse(content);
127 } catch (json::parse_error &e) {
128 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("消息解析分流：格式化json错误");
129 Logger::logger.error("For debug: " + content);
130 Logger::logger.error(e.what());)
131 return PLUGIN_NORMAL;
132 }
133
134 int type;
135
136 try {
137 if (!j.is_object() || !j.contains("eventId") || !j["eventId"].is_number()) {
138 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("Json格式错误：没有eventId这一field或这一field不是数字类型");)
139 return PLUGIN_NORMAL;
140 }
141 type = j["eventId"].get<int>();
142 if (type != eventTypes::Command && Event::noRegistered(type)) return PLUGIN_NORMAL;
143 } catch (...) {
144 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("Json格式错误：解析json时遇到无法预料的异常");)
145 return PLUGIN_NORMAL;
146 }
147
148 try {
149 // core logic
150 Event::incomingEvent(BaseEventData(std::move(j)), type);
151 } catch (json::type_error &e) {
152 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error("json格式化异常,位置：FUNC_EVENT");
153 Logger::logger.error(e.what());
154 Logger::logger.error(content);)
155 return PLUGIN_NORMAL;
156 } catch (MiraiCPExceptionBase &e) {
157 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Event::broadcast<MiraiCPExceptionEvent>(MiraiCPExceptionEvent(&e));
158 e.raise();)
159 return PLUGIN_NORMAL;
160 } catch (const std::exception &e) {
161 MIRAICP_CRITICAL_NOEXCEPT_BLOCK(Logger::logger.error(e.what());
162 Logger::logger.error("info:", content);)
163 return PLUGIN_NORMAL;
164 } catch (...) {
165 // 如果产生了无法处理的异常，直接退出插件，return 非0值
166 // loader端将处理这个异常并直接unload绑定的插件
167 // 如果存在其他线程，可能导致段错误
168 return PLUGIN_ERROR;
169 }
170 return PLUGIN_NORMAL;
171}
