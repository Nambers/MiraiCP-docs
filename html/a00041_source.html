320 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
321 if (re == "EF")
322 throw FriendException().append(name, line);
323 if (re == "EG")
324 throw GroupException().append(name, line);
325 if (re == "EM")
326 throw MemberException(1).append(name, line);
327 if (re == "EMM")
328 throw MemberException(2).append(name, line);
329 if (re == "EB")
330 throw BotException("找不到bot:" + re).append(name, line);
331 if (re == "EA")
332 throw APIException(ErrorMsg).append(name, line);
333 if (re == "EC")
334 throw EventCancelledException("发送信息被取消").append(name, line);
335 if (re == "ET")
336 throw TimeOutException("发送信息超时").append(name, line);
337 // equal to Tools::start_with
338 if (re.rfind("EBM", 0) == 0)
339 throw BotIsBeingMutedException(std::stoi(re.substr(3))).append(name, line);
340 }
