279 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
280 if (re == "EF")
281 throw FriendException().append(name, line);
282 if (re == "EG")
283 throw GroupException().append(name, line);
284 if (re == "EM")
285 throw MemberException(1).append(name, line);
286 if (re == "EMM")
287 throw MemberException(2).append(name, line);
288 if (re == "EB")
289 throw BotException("找不到bot:" + re).append(name, line);
290 if (re == "EA")
291 throw APIException(ErrorMsg).append(name, line);
292 if (re == "EC")
293 throw EventCancelledException("发送信息被取消").append(name, line);
294 if (re == "ET")
295 throw TimeOutException("发送信息超时").append(name, line);
296 // equal to Tools::start_with
297 if (re.rfind("EBM", 0) == 0)
298 throw BotIsBeingMutedException(std::stoi(re.substr(3))).append(name, line);
299 }
