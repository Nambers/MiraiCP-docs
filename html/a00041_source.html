91 class MiraiCPExceptionCRTP : public MiraiCPExceptionBase {
92 protected:
94 MiraiCPExceptionCRTP(string _filename, int _lineNum) : MiraiCPExceptionBase(std::move(_filename), _lineNum) {
95 exception_broadcast();
96 }
97
98 public:
99 // 构造时传入类型字符串
100 // 最好不要调用该构造函数
101 explicit MiraiCPExceptionCRTP(const string &description, string _filename, int _lineNum) : MiraiCPExceptionCRTP(std::move(_filename), _lineNum) {
102 if (description.empty())
103 this->re = T::exceptionType() + ":MiraiCP异常";
104 else
105 this->re = T::exceptionType() + ":" + description;
106 }
107
108 public:
109 // CRTP类型获取实现
110 string getExceptionType() override { return T::exceptionType(); }
111 };

308 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
309 if (re == "EF")
310 throw FriendException(name, line);
311 if (re == "EG")
312 throw GroupException(name, line);
313 if (re == "EM")
314 throw MemberException(1, name, line);
315 if (re == "EMM")
316 throw MemberException(2, name, line);
317 if (re == "EB")
318 throw BotException("找不到bot:" + re, name, line);
319 if (re == "EA")
320 throw APIException(ErrorMsg, name, line);
321 if (re == "EC")
322 throw EventCancelledException("发送信息被取消", name, line);
323 if (re == "ET")
324 throw TimeOutException("发送信息超时", name, line);
325 if (re == "EP")
326 throw BotException(name, line);
327 // equal to Tools::start_with
328 if (re.rfind("EBM", 0) == 0)
329 throw BotIsBeingMutedException(std::stoi(re.substr(3)), name, line);
330 }
