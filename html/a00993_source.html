39 json ForwardedMessage::nodesToJson() { // NOLINT(misc-no-recursion)
40 auto value = json::array();
41 for (const ForwardedNode &node: nodes) {
42 json temp{{"id", node.id}, {"time", node.time}, {"name", node.name}};
43 if (node.isForwarded()) {
44 temp["isForwardedMessage"] = true;
45 temp["message"] = std::get<std::shared_ptr<ForwardedMessage>>(node.message)->nodesToJson().dump();
46 if (display.has_value())
47 temp["display"] = display->toJson();
48 } else {
49 temp["message"] = std::get<MessageChain>(node.message).toMiraiCode();
50 }
51 value.emplace_back(std::move(temp));
52 }
53 json tmp2 = this->sendmsg;
54 tmp2["value"] = std::move(value);
55 return tmp2;
56 }

71 ForwardedMessage ForwardedMessage::deserializationFromMessageSourceJson(const json &j) {
72 std::vector<ForwardedNode> nodes;
73 try {
74 for (auto &&a: j) {
75 if (a["messageChain"][0].contains("kind") && a["messageChain"][0]["kind"] == "FORWARD") {
76 nodes.emplace_back(a["senderId"], a["senderName"], ForwardedMessage::deserializationFromMessageSourceJson(a["messageChain"][1]["nodeList"]), a["time"], ForwardedMessageDisplayStrategy::defaultStrategy());
77 } else {
78 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageSourceJson(a["messageChain"], false), a["time"]);
79 }
80 }
81 } catch (const json::exception &) {
82 throw APIException("ForwardedMessage格式化异常", MIRAICP_EXCEPTION_WHERE);
83 }
84
85 return ForwardedMessage(std::move(nodes));
86 }
