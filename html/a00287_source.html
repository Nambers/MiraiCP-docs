11 void ThreadPoolBase::worker() {
12 ThreadIdentify::IAmPoolThread();
13 setThreadIndex();
14 std::unique_lock<std::mutex> tasks_lock(tasks_mtx);
15 while (running.load(order_relaxed)) {
16 std::function<void()> task;
17 while (tasks.empty() && running.load(order_relaxed)) {
18 task_available_cv.wait_for(tasks_lock, std::chrono::milliseconds(100));
19 }
20
21 while (!is_paused() && !tasks.empty()) {
22 task = std::move(tasks.front());
23 tasks.pop();
24 tasks_lock.unlock();
25 task();
26 tasks_total.fetch_sub(1, std::memory_order_release);
27 tasks_lock.lock();
28 if (waiting.load(order_relaxed))
29 task_done_cv.notify_one();
30 }
31 }
32 }

47 void ThreadPoolBase::wait_for_tasks() {
48 std::unique_lock<std::mutex> tasks_lock(tasks_mtx);
49 waiting = true;
50 while ((get_tasks_total() != (is_paused() ? tasks.size() : 0))) {
51 // if you require a faster way to know the tasks are over, you should implement it in your tasks,
52 // instead of calling this stupid function to help you.
53 task_done_cv.wait_for(tasks_lock, std::chrono::milliseconds(10));
54 }
55 waiting = false;
