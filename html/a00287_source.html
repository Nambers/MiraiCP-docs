69 std::string SingleMessage::toMiraiCode() const {
70 if (internalType > 0)
71 if (internalType == Types::At_t)
72 return "[mirai:at:" + content + "] ";
73 else if (internalType == Types::AtAll_t)
74 return "[mirai:atall] ";
75 else
76 return "[mirai:" + getTypeString(internalType) + prefix + Tools::escapeToMiraiCode(content) + "]";
77 else
78 return content;

190 RemoteFile RemoteFile::deserializeFromString(const std::string &source) {
191 json j;
192 try {
193 j = json::parse(source);
194 } catch (json::parse_error &e) {
195 Logger::logger.error("格式化json失败，RemoteFile::deserializeFromString");
196 Logger::logger.error(source);
197 Logger::logger.error(e.what());
198 throw e;
199 }
200 try {
201 auto re = RemoteFile(j["id"], j["internalId"], j["name"], j["detailInfo"]["size"]);
202 if (j.contains("downloadInfo")) {
203 auto tmp = Tools::json_jsonmover(j, "downloadInfo");
204 struct Dinfo d {
205 Tools::json_jsonmover(tmp, "url"),
206 Tools::json_jsonmover(tmp, "md5"),
207 Tools::json_jsonmover(tmp, "sha1")
208 };
209 re.dinfo = d;
210 }
211 if (j["detailInfo"].contains("uploaderId")) {
212 auto tmp = Tools::json_jsonmover(j, "detailInfo");
213 struct Finfo f {
214 Tools::json_jsonmover(tmp, "size"),
215 Tools::json_jsonmover(tmp, "uploaderId"),
216 Tools::json_jsonmover(tmp, "expiryTime"),
217 Tools::json_jsonmover(tmp, "uploadTime"),
218 Tools::json_jsonmover(tmp, "lastModifyTime")
219 };
220 re.finfo = f;
221 }
222 if (j.contains("path"))
223 re.path = Tools::json_jsonmover(j, "path");
224 return re;
225 } catch (json::type_error &e) {
226 Logger::logger.error("json格式化失败，位置:RemoteFile");
227 Logger::logger.error(source);
228 Logger::logger.error(e.what());
229 throw e;
230 }
231 }

240 std::string RemoteFile::serializeToString() {
241 json j;
242 if (this->dinfo.has_value()) {
243 j["dinfo"]["url"] = this->dinfo->url;
244 j["dinfo"]["md5"] = this->dinfo->md5;
245 j["dinfo"]["shar1"] = this->dinfo->sha1;
246 }
247 if (this->finfo.has_value()) {
248 j["finfo"]["size"] = this->finfo->size;
249 j["finfo"]["uploaderid"] = this->finfo->uploaderid;
250 j["finfo"]["expirytime"] = this->finfo->expirytime;
251 j["finfo"]["uploadtime"] = this->finfo->uploadtime;
252 j["finfo"]["lastmodifytime"] = this->finfo->lastmodifytime;
253 }
254 j["id"] = this->id;
255 j["internalid"] = this->internalid;
256 j["name"] = this->name;
257 j["size"] = this->size;
258 if (this->path.has_value())
259 j["path"] = this->path.value();
260 return j.dump();
261 }

264 void Image::refreshInfo() {
265 std::string re = KtOperation::ktOperation(KtOperation::QueryImgInfo, toJson());
266 if (re == "E1")
267 throw RemoteAssetException("图片id格式错误", MIRAICP_EXCEPTION_WHERE);
268 json j = json::parse(re);
269 this->url = Tools::json_stringmover(j, "url");
270 this->md5 = Tools::json_stringmover(j, "md5");
271 this->size = j["size"];
272 this->width = j["width"];
273 this->height = j["height"];
274 this->imageType = j["imageType"];
275 this->isEmoji = j["isEmoji"];
276 }
