34 auto GetGroupFromPool(QQID groupid, QQID botid) noexcept {
35 static std::unordered_map<QQID, std::unordered_map<QQID, std::shared_ptr<Group::DataType>>> Pool;
36 auto &Val = Pool[botid][groupid];
37 if (!Val) {
38 Val = std::make_shared<Group::DataType>(groupid);
39 Val->_botId = botid;
40 Val->_id = groupid;
41 Val->_type = MIRAI_GROUP;
42 }
43 return Val;
44 }

158 json j{{"path", path}, {"filePath", filepath}, {"contact", toJson()}};
159 auto re = KtOperation::ktOperation(KtOperation::SendFile, j);
160 if (re == "E2") throw UploadException("找不到" + filepath + "位置:C-uploadfile", MIRAICP_EXCEPTION_WHERE);
161 if (re == "E3")
162 throw UploadException("Upload error:路径格式异常，应为'/xxx.xxx'或'/xx/xxx.xxx'目前只支持群文件和单层路径, path:" + path, MIRAICP_EXCEPTION_WHERE);
163 MIRAICP_ERROR_HANDLE(re, "");
164 return RemoteFile::deserializeFromString(re);
