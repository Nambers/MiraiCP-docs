105 constexpr inline interface_funcs collect_interface_functions(bool admin) {
106 constexpr size_t counter = LOADERAPI_H_GET_COUNTER;
107 static_assert(interface_funcs::api_count == counter);
108 static_assert(sizeof(interface_funcs) == sizeof(void *) * counter); // also check pointer size
109 if (admin) {
110 constexpr size_t line0 = __LINE__;
111 interface_funcs t = {
112 pluginOperation,
113 loggerInterface,
114 showAllPluginId,
115 pushTask,
116 pushTaskWithId,
117 enablePluginById,
118 disablePluginById,
119 enableAllPlugins,
120 disableAllPlugins,
121 loadNewPlugin,
122 unloadPluginById,
123 reloadPluginById,
124 };
125 constexpr size_t line1 = __LINE__;
126 constexpr size_t allapi_construct_number = line1 - line0 - 3;
127 static_assert(allapi_construct_number == counter);
128 return t;
129 } else {
130 constexpr size_t line0 = __LINE__;
131 interface_funcs t2 = {
132 pluginOperation,
133 loggerInterface,
134 showAllPluginId,
135 pushTask,
136 pushTaskWithId,
137 }; // no admin functions
138 constexpr size_t line1 = __LINE__;
139 constexpr size_t adminapi_construct_number = line1 - line0 - 3;
140 static_assert(adminapi_construct_number == interface_funcs::normal_api_count);
141 return t2;
142 }
143 }
