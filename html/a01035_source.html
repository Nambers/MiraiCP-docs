34 auto GetGroupFromPool(QQID groupid, QQID botid) noexcept {
35 static std::unordered_map<QQID, std::unordered_map<QQID, std::shared_ptr<Group::DataType>>> Pool;
36 auto &Val = Pool[botid][groupid];
37 if (!Val) {
38 Val = std::make_shared<Group::DataType>(groupid);
39 Val->_botid = botid;
40 Val->_id = groupid;
41 Val->_type = MIRAI_GROUP;
42 }
43 return Val;
44 }

155 RemoteFile Group::sendFile(const std::string &path, const std::string &filepath) {
156 json source{{"path", path}, {"filepath", filepath}};
157 json tmp{{"source", source.dump()}, {"contactSource", toString()}};
158 std::string result = KtOperation::ktOperation(KtOperation::SendFile, std::move(tmp));
159 if (result == "E2") throw UploadException("找不到" + filepath + "位置:C-uploadfile", MIRAICP_EXCEPTION_WHERE);
160 if (result == "E3")
161 throw UploadException("Upload error:路径格式异常，应为'/xxx.xxx'或'/xx/xxx.xxx'目前只支持群文件和单层路径, path:" + path, MIRAICP_EXCEPTION_WHERE);
162 return RemoteFile::deserializeFromString(result);
163 }
