25* 作用:判断是否连接上本插件，勿改
26* 参数:env 必备，job 必备
27* 返回值:jstring (用str2jstring把string类型转成jsrting) 发送返回的字符串

29using json = nlohmann::json;
30JNIEXPORT jstring
31Verify(JNIEnv *env, jobject) {
32 using namespace MiraiCP;
33 ThreadManager::setEnv(env);
34 MiraiCP::ThreadManager::JNIVersion = env->GetVersion();
35 try {
36 //初始化日志模块
37 Config::construct();
38 Logger::logger.init();
39 enrollPlugin();
40 if (CPPPlugin::plugin == nullptr) {
41 Logger::logger.error("无插件实例加载");
42 } else {
43 CPPPlugin::pluginLogger = new PluginLogger(&Logger::logger);
44 CPPPlugin::plugin->onEnable();
45 }
46 } catch (const MiraiCPException &e) {
47 e.raise();
48 }
49 json j = CPPPlugin::plugin->config.serialize();
50 j["MiraiCPversion"] = MiraiCPVersion;
51 return Tools::str2jstring(j.dump().c_str());
52 //验证机制，返回当前SDK版本
53}

79JNIEXPORT jstring
80Event(JNIEnv *env, jobject, jstring content) {
81 using namespace MiraiCP;
82 ThreadManager::setEnv(env);
83 std::string tmp = Tools::jstring2str(content, env);
84 json j;
85 try {
86 j = json::parse(tmp);
87 } catch (json::parse_error &e) {
88 APIException("格式化json错误").raise();
89 Logger::logger.error("For debug:" + j.dump());
90 Logger::logger.error(e.what(), false);
91 return returnNull();
92 }
93 try {
94 switch (j["type"].get<int>()) {
95 case 1: {
96 //GroupMessage
97 Event::processor.broadcast<GroupMessageEvent>(
98 GroupMessageEvent(j["group"]["botid"],
99 Group(Group::deserialize(j["group"])),
100 Member(Member::deserialize(j["member"])),
101 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
102 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
103 break;
104 }
105 case 2: {
106 //私聊消息
107 Event::processor.broadcast<PrivateMessageEvent>(
108 PrivateMessageEvent(j["friend"]["botid"],
109 Friend(Friend::deserialize(j["friend"])),
110 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
111 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
112 break;
113 }
114 case 3:
115 //群聊邀请
116 Event::processor.broadcast<GroupInviteEvent>(
117 GroupInviteEvent(
118 j["source"]["botid"],
119 j["request"],
120 j["source"]["inviternick"],
121 j["source"]["inviterid"],
122 j["source"]["groupname"],
123 j["source"]["groupid"]));
124 break;
125 case 4:
126 //好友
127 Event::processor.broadcast<NewFriendRequestEvent>(
128 NewFriendRequestEvent(
129 j["source"]["botid"],
130 j["request"],
131 j["source"]["fromid"],
132 j["source"]["fromgroupid"],
133 j["source"]["fromnick"],
134 j["source"]["message"]));
135 break;
136 case 5:
137 //新成员加入
138 Event::processor.broadcast<MemberJoinEvent>(
139 MemberJoinEvent(
140 j["group"]["botid"],
141 j["jointype"],
142 Member(Member::deserialize(j["member"])),
143 Group(Group::deserialize(j["group"])),
144 j["inviterid"]));
145 break;
146 case 6:
147 //群成员退出
148 Event::processor.broadcast<MemberLeaveEvent>(MemberLeaveEvent(
149 j["group"]["botid"],
150 j["leavetype"],
151 j["memberid"],
152 Group(Group::deserialize(j["group"])),
153 j["operatorid"]));
154 break;
155 case 7:
156 Event::processor.broadcast<RecallEvent>(RecallEvent(
157 j["botid"],
158 j["etype"],
159 j["time"],
160 j["authorid"],
161 j["operatorid"],
162 j["ids"],
163 j["internalids"],
164 j["groupid"]));
165 break;
166 case 9:
167 Event::processor.broadcast<BotJoinGroupEvent>(BotJoinGroupEvent(
168 j["group"]["botid"],
169 j["etype"],
170 Group(Group::deserialize(j["group"])),
171 j["inviterid"]));
172 break;
173 case 10:
174 Event::processor.broadcast<GroupTempMessageEvent>(GroupTempMessageEvent(
175 j["group"]["botid"],
176 Group(Group::deserialize(j["group"])),
177 Member(Member::deserialize(j["member"])),
178 MessageChain::deserializationFromMessageSourceJson(json::parse(j["message"].get<std::string>()))
179 .plus(MessageSource::deserializeFromString(j["source"]))));
180 break;
181 case 11:
182 Event::processor.broadcast<BotOnlineEvent>(BotOnlineEvent(j["botid"]));
183 break;
184 case 12:
185 Event::processor.broadcast<TimeOutEvent>(TimeOutEvent(j["msg"]));
186 break;
187 case 13:
188 Event::processor.broadcast<NudgeEvent>(NudgeEvent(Contact::deserialize(j["from"]),
189 Contact::deserialize(j["target"]),
190 j["botid"]));
191 break;
192 case 14:
193 Event::processor.broadcast(BotLeaveEvent(j["groupid"], j["botid"]));
194 break;
195 case 15: {
196 std::optional<Group> a;
197 std::optional<Member> b;
198 Contact temp = Contact::deserialize(j["group"]);
199 if (temp.id() == 0)
200 a = std::nullopt;
201 else
202 a = Group(temp);
203 temp = Contact::deserialize(j["inviter"]);
204 if (temp.id() == 0)
205 b = std::nullopt;
206 else
207 b = Member(temp);
208 Event::processor.broadcast(MemberJoinRequestEvent(a, b, temp.botid(), j["requestData"]));
209 break;
210 }
211 default:
212 MiraiCPThrow(APIException("Unreachable code"));
213 }
214 } catch (json::type_error &e) {
215 Logger::logger.error("json格式化异常,位置C-Handle");
216 Logger::logger.error(e.what(), false);
217 return Tools::str2jstring("ERROR");
218 } catch (const MiraiCPException &e) {
219 e.raise();
220 return Tools::str2jstring("ERROR");
221 } catch (const std::exception &e) {
222 // 这里如果不catch全部exception就会带崩jvm
223 Logger::logger.error(e.what());
224 return Tools::str2jstring("ERROR");
225 }
226 return returnNull();
227}

229int registerMethods(JNIEnv *env, const char *className, JNINativeMethod *gMethods, int numMethods) {
230 jclass clazz = env->FindClass(className);
231 if (clazz == nullptr) {
232 return JNI_FALSE;
233 }
234 //注册native方法
235 if (env->RegisterNatives(clazz, gMethods, numMethods) < 0) {
236 return JNI_FALSE;
237 }
238 return JNI_TRUE;
239}

246extern "C" JNIEXPORT jint JNI_OnLoad(JavaVM *vm, void *) {
247 JNIEnv *env = nullptr;
248 if (vm->GetEnv((void **) &env, JNI_VERSION_1_6) != JNI_OK) {
249 return JNI_ERR;
250 }
251 assert(env != nullptr);
252 MiraiCP::ThreadManager::gvm = vm;
253 // 注册native方法
254 if (!registerMethods(env, "tech/eritquearcus/miraicp/shared/CPPLib", method_table, 3)) {
255 return JNI_ERR;
256 }
257
258 return JNI_VERSION_1_6;
259}
