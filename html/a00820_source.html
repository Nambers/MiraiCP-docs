41 void Logger_interface::log0(const std::string &content, int level, json j, JNIEnv *env) {
42 if (env == nullptr) env = ThreadManager::getEnv();
43 if (this->loggerhandler.enable)
44 this->loggerhandler.action(content, level);
45 env->CallStaticVoidMethod(Config::CPPLib, log, Tools::str2jstring(j.dump().c_str()), (jint) level);
46 }
