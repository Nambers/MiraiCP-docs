34 MessageChain PrivateMessageEvent::nextMessage(long time, bool halt, JNIEnv *env) const {
35 json j;
36 j["contactSource"] = this->sender.toString();
37 j["time"] = time;
38 j["halt"] = halt;
39 std::string r = Config::koperation(Config::NextMsg, j, env);
