36 MessageChain MessageChain::deserializationFromMiraiCode(const std::string &m) {
37 size_t pos = 0;
38 size_t lastPos = -1;
39 MessageChain mc;
40 if (m.length() <= 7) {
41 return MessageChain(PlainText(m));
42 }
43 do {
44 if (m.length() - 7 - pos > 0 && m.substr(pos, 7) == "[mirai:") {
45 if (pos - lastPos > 1)
46 mc.add(PlainText(m.substr(lastPos + 1, pos - lastPos - 1))); // plain text
47 size_t back = MessageChain::findEnd(m, pos);
48 if (back == -1) MiraiCPThrow(IllegalStateException(""));
49 std::string tmp = m.substr(pos, back - pos);
50 tmp = Tools::replace(tmp, "[mirai:", "");
51 size_t i = tmp.find(':'); // first :
52 int t = SingleMessage::getKey(tmp.substr(0, i));
53 switch (t) {
54 case 0:
55 // no miraiCode key is PlainText
56 Logger::logger.error("无法预料的错误, 信息: " + m);
57 break;
58 case 1:
59 mc.add(At(std::stoll(tmp.substr(i + 1, tmp.length() - i - 1))));
60 break;
61 case 2:
62 mc.add(AtAll());
63 break;
64 case 3:
65 mc.add(Image(tmp.substr(i + 1, tmp.length() - i - 1)));
66 break;
67 case 4:
68 mc.add(LightApp(tmp.substr(i + 1, tmp.length() - i - 1)));
69 break;
70 case 5: {
71 size_t comma = tmp.find(',');
72 mc.add(ServiceMessage(std::stoi(tmp.substr(i + 1, comma - i - 1)),
73 tmp.substr(comma + 1, tmp.length() - comma - 1)));
74 break;
75 }
76 case 6: {
77 //[mirai:file:/b53231e8-46dd-11ec-8ba5-5452007bd6c0,102,run.bat,55]
78 size_t comma1 = tmp.find(',');
79 size_t comma2 = tmp.find(',', comma1 + 1);
80 size_t comma3 = tmp.find(',', comma2 + 1);
81 mc.add(RemoteFile(tmp.substr(i + 1, comma1 - i - 1), std::stoi(tmp.substr(comma1 + 1, comma2 - comma1 - 1)), tmp.substr(comma2 + 1, comma3 - comma2 - 1), std::stoll(tmp.substr(comma3 + 1, tmp.length() - comma3 - 1))));
82 break;
83 }
84 case 7:
85 mc.add(Face(std::stoi(tmp.substr(i + 1, tmp.length() - i - 1))));
86 break;
87 default:
88 Logger::logger.error(
89 "MiraiCP碰到了意料之中的错误(原因:部分SimpleMessage在MiraiCode解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MiraiCode:" +
90 m);
91 mc.add(UnSupportMessage("[mirai:" + tmp));
92 break;
93 }
94 pos = back;
95 lastPos = pos;
96 if (t == 1)
97 lastPos++;
98 }
99 pos++;
100 } while (pos < m.length());
101 if (lastPos + 1 < m.length())
102 mc.add(PlainText(m.substr(lastPos + 1, m.length() - lastPos - 1))); // plain text
103 return mc;
104 }

106 MessageChain MessageChain::deserializationFromMessageSourceJson(const json &tmp, bool origin) {
107 json j = tmp;
108 Logger::logger.info("ms: " + tmp.dump());
109 if (origin)
110 j = tmp["originalMessage"];
111 MessageChain mc;
112 if (j[0]["type"] == "MessageOrigin") {
113 mc.add(OnlineForwardedMessage::deserializationFromMessageSourceJson(j));
114 return mc;
115 }
116 for (auto node: j) {
117 if (node["type"] == "SimpleServiceMessage") {
118 mc.add(ServiceMessage(node["serviceId"], node["content"]));
119 continue;
120 }
121 if (node["type"] == "LightApp") {
122 mc.add(LightApp(node["content"]));
123 continue;
124 }
125 if (node["type"] == "OnlineAudio") {
126 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
127 node["urlForDownload"]));
128 continue;
129 }
130 if (node["type"] == "FileMessage") {
131 mc.add(Group(tmp["targetId"].get<QQID>(), tmp["botId"].get<QQID>()).getFileById(node["id"]).plus(node["internalId"]));
132 continue;
133 }
134 if (node["type"] == "MarketFace") {
135 mc.add(MarketFace(node["delegate"]["faceId"]));
136 break;
137 }
138 switch (SingleMessage::getKey(node["type"])) {
139 case -2:
140 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
141 break;
142 case 0:
143 mc.add(PlainText(node["content"].get<std::string>()));
144 break;
145 case 1:
146 mc.add(At(node["target"]));
147 break;
148 case 2:
149 mc.add(AtAll());
150 break;
151 case 3:
152 mc.add(Image(node["imageId"]));
153 break;
154 case 7:
155 mc.add(Face(node["id"]));
156 break;
157 default:
158 Logger::logger.error(
159 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MessageSource:" +
160 j.dump());
161 mc.add(UnSupportMessage(node["content"]));
162 }
163 }
164 return mc;
165 }
