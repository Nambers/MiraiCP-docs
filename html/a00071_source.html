36 MessageChain MessageChain::deserializationFromMiraiCode(const std::string &m) {
37 size_t pos = 0;
38 size_t lastPos = -1;
39 MessageChain mc;
40 if (m.length() <= 7) {
41 return MessageChain(PlainText(m));
42 }
43 do {
44 if (m.length() - 7 - pos > 0 && m.substr(pos, 7) == "[mirai:") {
45 if (pos - lastPos > 1)
46 mc.add(PlainText(m.substr(lastPos + 1, pos - lastPos - 1))); // plain text
47 size_t back = MessageChain::findEnd(m, pos);
48 if (back == -1) MiraiCPThrow(IllegalStateException(""));
49 std::string tmp = m.substr(pos, back - pos);
50 tmp = Tools::replace(tmp, "[mirai:", "");
51 size_t i = tmp.find(':'); // first :
52 int t = SingleMessage::getKey(tmp.substr(0, i));
53 switch (t) {
54 case 0:
55 // no miraiCode key is PlainText
56 Logger::logger.error("无法预料的错误, 信息: " + m);
57 break;
58 case 1:
59 mc.add(At(std::stoll(tmp.substr(i + 1, tmp.length() - i - 1))));
60 break;
61 case 2:
62 mc.add(AtAll());
63 break;
64 case 3:
65 mc.add(Image(tmp.substr(i + 1, tmp.length() - i - 1)));
66 break;
67 case 4:
68 mc.add(LightApp(tmp.substr(i + 1, tmp.length() - i - 1)));
69 break;
70 case 5: {
71 size_t comma = tmp.find(',');
72 mc.add(ServiceMessage(std::stoi(tmp.substr(i + 1, comma - i - 1)),
73 tmp.substr(comma + 1, tmp.length() - comma - 1)));
74 break;
75 }
76 case 6: {
77 //[mirai:file:/b53231e8-46dd-11ec-8ba5-5452007bd6c0,102,run.bat,55]
78 size_t comma1 = tmp.find(',');
79 size_t comma2 = tmp.find(',', comma1 + 1);
80 size_t comma3 = tmp.find(',', comma2 + 1);
81 mc.add(RemoteFile(tmp.substr(i + 1, comma1 - i - 1), std::stoi(tmp.substr(comma1 + 1, comma2 - comma1 - 1)), tmp.substr(comma2 + 1, comma3 - comma2 - 1), std::stoll(tmp.substr(comma3 + 1, tmp.length() - comma3 - 1))));
82 break;
83 }
84 case 7:
85 mc.add(Face(std::stoi(tmp.substr(i + 1, tmp.length() - i - 1))));
86 break;
87 default:
88 Logger::logger.error(
89 "MiraiCP碰到了意料之中的错误(原因:部分SimpleMessage在MiraiCode解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MiraiCode:" +
90 m);
91 mc.add(UnSupportMessage("[mirai:" + tmp));
92 break;
93 }
94 pos = back;
95 lastPos = pos;
96 if (t == 1)
97 lastPos++;
98 }
99 pos++;
100 } while (pos < m.length());
101 if (lastPos + 1 < m.length())
102 mc.add(PlainText(m.substr(lastPos + 1, m.length() - lastPos - 1))); // plain text
103 return mc;
104 }

106 MessageChain MessageChain::deserializationFromMessageSourceJson(const json &tmp, bool origin) {
107 json j = tmp;
108 if (origin)
109 j = tmp["originalMessage"];
110 MessageChain mc;
111 if (j[0]["type"] == "MessageOrigin") {
112 mc.add(OnlineForwardedMessage::deserializationFromMessageSourceJson(j));
113 return mc;
114 }
115 for (auto node: j) {
116 if (node["type"] == "SimpleServiceMessage") {
117 mc.add(ServiceMessage(node["serviceId"], node["content"]));
118 continue;
119 }
120 if (node["type"] == "LightApp") {
121 mc.add(LightApp(node["content"]));
122 continue;
123 }
124 if (node["type"] == "OnlineAudio") {
125 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
126 node["urlForDownload"]));
127 continue;
128 }
129 if (node["type"] == "FileMessage") {
130 mc.add(Group(tmp["targetId"].get<QQID>(), tmp["botId"].get<QQID>()).getFileById(node["id"]).plus(node["internalId"]));
131 continue;
132 }
133 if (node["type"] == "MarketFace") {
134 mc.add(MarketFace(node["delegate"]["faceId"]));
135 break;
136 }
137 switch (SingleMessage::getKey(node["type"])) {
138 case -2:
139 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
140 break;
141 case 0:
142 mc.add(PlainText(node["content"].get<std::string>()));
143 break;
144 case 1:
145 mc.add(At(node["target"]));
146 break;
147 case 2:
148 mc.add(AtAll());
149 break;
150 case 3:
151 mc.add(Image(node["imageId"]));
152 break;
153 case 7:
154 mc.add(Face(node["id"]));
155 break;
156 default:
157 Logger::logger.error(
158 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MessageSource:" +
159 j.dump());
160 mc.add(UnSupportMessage(node["content"]));
161 }
162 }
163 return mc;
164 }
