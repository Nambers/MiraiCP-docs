589 class Event {
590 private:
591 class eventNode {
592 public:
593 bool enable = true;
594 std::function<void(MiraiCPEvent *)> func;
595
596 explicit eventNode(std::function<void(MiraiCPEvent *)> f) : func(std::move(f)) {}
597
598 void run(MiraiCPEvent *a) const {
599 func(a);
600 }
601 };
602
603 using eventNodeTable = std::vector<std::vector<eventNode>>;
604 eventNodeTable vec;
605
606 private:
607 template<typename T>
608 int id() const {
609 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播继承MiraiCPEvent的事件");
610 return int(T::get_event_type());
611 }
612
613 Event() : vec(int(eventTypes::count)){};
614
615 public:
617 class NodeHandle {
618 private:
619 bool *enable;
620
621 public:
622 explicit NodeHandle(bool *a) { this->enable = a; }
623 void stop() { *enable = false; }
624 void resume() { *enable = true; }
625 };
626
627 public:
628 // singleton mode
629 static Event processor;
630
631 public:
632 bool noRegistered(int index) {
633 return vec[index].empty();
634 }
635
637 template<typename T>
638 void broadcast(T &&val) {
639 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播MiraiCPEvent的派生类");
640 MiraiCPEvent *p = &val;
641 for (auto &a: vec[id<T>()]) {
642 a.run(p);
643 }
644 }
645
647 template<typename T>
648 NodeHandle registerEvent(std::function<void(T)> a) {
649 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持注册MiraiCPEvent的派生类事件");
650 std::function<void(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
651 a(*dynamic_cast<T *>(p));
652 };
653 eventNode t(tmp);
654 vec[id<T>()].emplace_back(t);
655 return NodeHandle(&t.enable);
656 }
657 };
658} // namespace MiraiCP
