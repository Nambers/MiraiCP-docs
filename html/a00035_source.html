547 class Event {
548 private:
549 template<typename T>
550 eventTypes id() const {
551 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播继承MiraiCPEvent的事件");
552 return T::getEventType();
553 }
554
555 Event() = default;
556
557 class eventNode {
558 public:
559 bool enable = true;
560 std::function<void(MiraiCPEvent *)> func;
561
562 explicit eventNode(std::function<void(MiraiCPEvent *)> f) : func(std::move(f)) {}
563
564 void run(MiraiCPEvent *a) const {
565 func(a);
566 }
567 };
568 std::vector<std::vector<eventNode>> vec = std::vector<std::vector<eventNode>>(int(eventTypes::count));
569
570 public:
571 const std::vector<std::vector<eventNode>> *const eventNodes() const {
572 return &vec;
573 }
575 class NodeHandle {
576 private:
577 bool *enable;
578
579 public:
580 explicit NodeHandle(bool *a) { this->enable = a; }
581 void stop() { *enable = false; }
582 void resume() { *enable = true; }
583 };
584
585 // singleton mode
586 static Event processor;
587
589 template<typename T>
590 void broadcast(T &&val) {
591 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播MiraiCPEvent的派生类");
592 MiraiCPEvent *p = &val;
593 for (auto &a: vec[int(id<T>())]) {
594 a.run(p);
595 }
596 }
597
599 template<typename T>
600 NodeHandle registerEvent(std::function<void(T)> a) {
601 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持注册MiraiCPEvent的派生类事件");
602 std::function<void(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
603 a(*dynamic_cast<T *>(p));
604 };
605 eventNode t(tmp);
606 vec[int(id<T>())].emplace_back(t);
607 return NodeHandle(&t.enable);
608 }
609 };
