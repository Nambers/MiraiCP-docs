35 ForwardedMessage::ForwardedMessage(Contact *c, std::initializer_list<ForwardedNode> nodes) {
36 json root;
37 json value;
38 root["type"] = c->type();
39 root["id"] = c->id();
40 root["id2"] = c->groupid();
41 for (const ForwardedNode &node: nodes) {
42 json temp;
43 temp["id"] = node.id;
44 temp["time"] = node.time;
45 temp["message"] = node.message.toMiraiCode();
46 temp["name"] = node.name;
47 value.push_back(temp);
48 }
49 root["value"] = value;
50 sendmsg = root;
51 }
52 OnlineForwardedMessage OnlineForwardedMessage::deserializationFromMessageSourceJson(json j) {
53 std::vector<ForwardedNode> nodes;
54 for (json a: j[1]["nodeList"])
55 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageSourceJson(a["messageChain"], false), a["time"]);
56 return OnlineForwardedMessage(j[0]["origin"], j[0]["resourceId"], nodes);
57 }
