59 auto GetMemberFromPool(QQID id, QQID groupId, QQID botid) noexcept {
60 using Tools::idpair;
61 static std::unordered_map<idpair, std::unordered_map<QQID, std::shared_ptr<Member::DataType>>> Pool;
62 idpair pr{botid, groupId};
63 auto &Val = Pool[pr][id];
64 if (!Val) {
65 Val = std::make_shared<Member::DataType>(groupId);
66 Val->_id = id;
67 Val->_botId = botid;
68 Val->_type = MIRAI_MEMBER;
69 }
70 return Val;
71 }

86 Member::Member(nlohmann::json in_json) : Contact(GetMemberFromPool(in_json)) {
87 auto ActualDataPtr = GetDataInternal();
88 assert(ActualDataPtr != nullptr);
89 bool needrefresh = false;
90 if (in_json.contains("nickOrNameCard")) ActualDataPtr->_nickOrNameCard = json_stringmover(in_json, "nickOrNameCard");
91 else
92 needrefresh = true;
93 if (in_json.contains("avatarUrl")) ActualDataPtr->_avatarUrl = json_stringmover(in_json, "avatarUrl");
94 else
95 needrefresh = true;
96 if (in_json.contains("anonymous")) ActualDataPtr->_anonymous = in_json["anonymous"].get<bool>();
97 else
98 needrefresh = true;
99
100 if (needrefresh) forceRefreshNextTime();
101 }

149 void MemberData::refreshInfo() {
150 if (_anonymous) return;
151
152 std::string result = LowLevelAPI::getInfoSource(internalToJson());
153
154 if (result == "E1")
155 throw MemberException(1, MIRAICP_EXCEPTION_WHERE);
156 if (result == "E2")
157 throw MemberException(2, MIRAICP_EXCEPTION_WHERE);
158
159 {
160 LowLevelAPI::info tmp = LowLevelAPI::info0(result);
161 this->_nickOrNameCard = tmp.nickOrNameCard;
162 this->_avatarUrl = tmp.avatarUrl;
163 }
164
165 {
166 result = KtOperation::ktOperation(KtOperation::QueryM, internalToJson());
167 MIRAICP_ERROR_HANDLE(result, "");
168 auto tmp = nlohmann::json::parse(result);
169 _permission = tmp["permission"];
170 _specialTitle = json_stringmover(tmp, "specialTitle");
171 }
172 }
