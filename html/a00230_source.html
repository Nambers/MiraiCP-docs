30using timepoint = decltype(std::chrono::system_clock::now());

32struct scheduleTask {
33 timepoint tp;
34 std::string pluginId;
35 std::string content;
36};
37
38template<>
39struct std::greater<scheduleTask> {
40 bool operator()(const scheduleTask &a, const scheduleTask &b) {
41 return std::greater<>()(a.tp, b.tp);
42 }
43};

50 static std::priority_queue<scheduleTask, std::vector<scheduleTask>, std::greater<scheduleTask>> timerQueue; // NOLINT(modernize-use-transparent-functors)
51 static std::mutex mtx;

53 void timer(std::string pluginId, std::string content, size_t timeInSecond) {
54 auto scheduleTime = std::chrono::system_clock::now() + std::chrono::seconds(timeInSecond);
55 scheduleTask a{scheduleTime, std::move(pluginId), std::move(content)};
56 std::lock_guard lk(mtx);
57 timerQueue.push(std::move(a));
58 loaderWakeCV().notify_one();
59 }

61 inline void sendTimeoutEvent(const std::string &pluginId, const std::string &content) noexcept {
62 nlohmann::json j{{"eventId", 15},
63 {"eventData", nlohmann::json{{"msg", content}}}};
64 PluginListManager::broadcastToOnePlugin(pluginId, j.dump());
65 }
66
67 void popSchedule() noexcept {
68 std::lock_guard lk(mtx);
69 while (!timerQueue.empty() && timerQueue.top().tp < std::chrono::system_clock::now()) {
70 auto task = timerQueue.top();
71 timerQueue.pop();
72 sendTimeoutEvent(task.pluginId, task.content);
73 }
74 }
75
76 bool empty() noexcept {
77 return timerQueue.empty();
78 }
