24 using json = nlohmann::json;
25 json ForwardedMessage::nodesToJson() {
26 json value;
27 for (const ForwardedNode &node: nodes) {
28 json temp;
29 temp["id"] = node.id;
30 temp["time"] = node.time;
31 if (node.forwardedMsg != nullptr) {
32 temp["isForwardedMessage"] = true;
33 temp["message"] = node.forwardedMsg->nodesToJson().dump();
34 } else
35 temp["message"] = node.message.toMiraiCode();
36 temp["name"] = node.name;
37 value.push_back(temp);
38 }
39 json tmp = this->sendmsg;
40 tmp["value"] = value;
41 return tmp;
42 }

44 MessageSource ForwardedMessage::sendTo(Contact *c, JNIEnv *env) {
45 json temp;
46 json text;
47 text["id"] = c->id();
48 text["groupid"] = c->groupid();
49 text["type"] = c->type();
50 text["content"] = this->nodesToJson();
51 temp["text"] = text.dump();
52 temp["botid"] = c->botid();
53 std::string re = Config::koperation(Config::Buildforward, temp, env);
54 ErrorHandle(re, "");
55 return MessageSource::deserializeFromString(re);
56 }

66 std::vector<ForwardedNode> nodes;
67 for (json a: j[1]["nodeList"])
68 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageSourceJson(a["messageChain"], false), a["time"]);
69 return OnlineForwardedMessage(j[0]["origin"], j[0]["resourceId"], nodes);
