308 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
309 if (re == "EF")
310 throw FriendException(name, line);
311 if (re == "EG")
312 throw GroupException(name, line);
313 if (re == "EM")
314 throw MemberException(1, name, line);
315 if (re == "EMM")
316 throw MemberException(2, name, line);
317 if (re == "EB")
318 throw BotException("找不到bot:" + re, name, line);
319 if (re == "EA")
320 throw APIException(ErrorMsg, name, line);
321 if (re == "EC")
322 throw EventCancelledException("发送信息被取消", name, line);
323 if (re == "ET")
324 throw TimeOutException("发送信息超时", name, line);
325 if (re == "EP")
326 throw BotException(name, line);
327 // equal to Tools::start_with
328 if (re.rfind("EBM", 0) == 0)
329 throw BotIsBeingMutedException(std::stoi(re.substr(3)), name, line);
330 }
