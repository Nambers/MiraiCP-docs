23 using json = nlohmann::json;
24 void MessageSource::recall(JNIEnv *env) const {
25 json j;
26 j["source"] = this->serializeToString();
27 std::string re = Config::koperation(Config::Recall, j, env);
28 if (re == "Y") return;
29 if (re == "E1") MiraiCPThrow(BotException());
30 if (re == "E2") MiraiCPThrow(RecallException());
31 }

44 MessageSource MessageSource::deserializeFromString(const std::string &source) {
45 json j = json::parse(source);
46 try {
47 return {j["ids"].dump(), j["internalIds"].dump(), source};
48 } catch (json::type_error &e) {
49 Logger::logger.error("消息源序列化出错，格式不符合(MessageSource::deserializeFromString)");
50 Logger::logger.error(source);
51 Logger::logger.error(e.what());
52 MiraiCPThrow(IllegalArgumentException(std::string("消息源序列化出错，格式不符合(MessageSource::deserializeFromString), ") + e.what()));
53 }
54 }

56 MessageSource MessageSource::quoteAndSendMiraiCode(const std::string &content, QQID groupid, JNIEnv *env) const {
57 json obj;
58 json sign;
59 obj["messageSource"] = this->serializeToString();
60 obj["msg"] = content;
61 sign["MiraiCode"] = true;
62 sign["groupid"] = groupid;
63 obj["sign"] = sign.dump();
64 std::string re = Config::koperation(Config::SendWithQuote, obj, env);
65 return MessageSource::deserializeFromString(re);
66 }
