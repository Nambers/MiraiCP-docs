41 MessageSource MessageSource::deserializeFromString(const std::string &source) {
42 json j = json::parse(source);
43 try {
44 return {j["ids"].dump(), j["internalIds"].dump(), source};
45 } catch (json::type_error &e) {
46 Logger::logger.error("消息源序列化出错，格式不符合(MessageSource::deserializeFromString)");
47 Logger::logger.error(source);
48 Logger::logger.error(e.what());
49 MiraiCPThrow(IllegalArgumentException(std::string("消息源序列化出错，格式不符合(MessageSource::deserializeFromString), ") + e.what()));
50 }
51 }

77 MessageSource Contact::sendMsg0(const std::string &msg, int retryTime, bool miraicode, JNIEnv *env) {
78 if (msg.empty()) {
79 MiraiCPThrow(IllegalArgumentException("不能发送空信息, 位置: Contact::SendMsg"));
80 }
81 std::string re = LowLevelAPI::send0(msg, this, retryTime, miraicode, env,
82 "reach a error area, Contact::SendMiraiCode");
83 if (re == "ET")
84 MiraiCPThrow(TimeOutException("发送消息过于频繁导致的tx服务器未能即使响应, 位置: Contact::SendMsg"));
85 if (Tools::starts_with(re, "EBM"))
86 MiraiCPThrow(BotIsBeingMutedException(std::stoi(re.substr(3))));
87 return MessageSource::deserializeFromString(re);
88 }
