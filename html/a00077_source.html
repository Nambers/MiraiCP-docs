669 class Event {
670 private: // typedefs
671 class eventNode {
672 private:
674 std::shared_ptr<NodeHandle> _handle;
675
676 public:
677 std::function<bool(MiraiCPEvent *)> func;
678
679 explicit eventNode(std::function<bool(MiraiCPEvent *)> f) : _handle(new NodeHandle(true)), func(std::move(f)) {}
680
681 eventNode(eventNode &&_o) noexcept : _handle(std::move(_o._handle)), func(std::move(_o.func)) {}
682
683 public:
685 bool run(MiraiCPEvent *a) const {
686 return _handle->isEnable() && func(a);
687 }
688
689 std::shared_ptr<NodeHandle> getHandle() {
690 return _handle;
691 }
692 };
693
694 using priority_level = unsigned char;
695 using event_vector = std::vector<eventNode>;
696 using eventNodeTable = std::vector<std::map<priority_level, event_vector>>;
697
698 private: // member
699 eventNodeTable _all_events_;
700
701 private:
702 Event() : _all_events_(int(eventTypes::count)){};
703
704 public: // singleton mode
705 static Event processor;
706
707 private:
708 template<typename EventClass>
709 constexpr static int id() {
710 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持广播继承MiraiCPEvent的事件");
711 return static_cast<int>(EventClass::get_event_type());
712 }
713
714 public:
715 static bool noRegistered(int index) {
716 return processor._all_events_[index].empty();
717 }
718
720 template<typename EventClass>
721 static void broadcast(EventClass &&val) {
722 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持广播MiraiCPEvent的派生类");
723 MiraiCPEvent *p = &val;
724 for (auto &&[k, v]: processor._all_events_[id<EventClass>()]) {
725 for (auto &&a: v) {
726 if (a.run(p)) return;
727 }
728 }
729 }
730
738 template<typename EventClass>
739 static std::shared_ptr<NodeHandle> registerEvent(std::function<void(EventClass)> callback, priority_level level = 100) {
740 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持注册MiraiCPEvent的派生类事件");
741 std::function<bool(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
742 callback(*dynamic_cast<EventClass *>(p));
743 return false;
744 };
745 auto t = eventNode(tmp);
746 auto ans = t.getHandle();
747 // 先获得shared_ptr才可以emplace_back
748 processor._all_events_[id<EventClass>()][level].emplace_back(std::move(t));
749 return ans;
750 }
751
760 template<typename EventClass>
761 static std::shared_ptr<NodeHandle> registerBlockingEvent(std::function<bool(EventClass)> callback, priority_level level = 100) {
762 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持注册MiraiCPEvent的派生类事件");
763 std::function<bool(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
764 return callback(*dynamic_cast<EventClass *>(p));
765 };
766 auto t = eventNode(tmp);
767 auto ans = t.getHandle();
768 // 先获得shared_ptr才可以emplace_back
769 processor._all_events_[id<EventClass>()][level].emplace_back(std::move(t));
770 return ans;
771 }
