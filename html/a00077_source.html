77 class MiraiCPExceptionCRTP : public MiraiCPExceptionBase {
78 protected:
80 MiraiCPExceptionCRTP(string _filename, int _lineNum) : MiraiCPExceptionBase(std::move(_filename), _lineNum) {

82 }
83
84 public:
85 // 构造时传入类型字符串
86 // 最好不要调用该构造函数
87 explicit MiraiCPExceptionCRTP(const string &description, string _filename, int _lineNum) : MiraiCPExceptionCRTP(std::move(_filename), _lineNum) {
88 if (description.empty())
89 this->re = T::exceptionType() + ":MiraiCP异常";
90 else
91 this->re = T::exceptionType() + ":" + description;
92 }
93
94 public:
95 // CRTP类型获取实现
96 string getExceptionType() override { return T::exceptionType(); }

294 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
295 if (re == "EF")
296 throw FriendException(name, line);
297 if (re == "EG")
298 throw GroupException(name, line);
299 if (re == "EM")
300 throw MemberException(1, name, line);
301 if (re == "EMM")
302 throw MemberException(2, name, line);
303 if (re == "EB")
304 throw BotException("找不到bot:" + re, name, line);
305 if (re == "EA")
306 throw APIException(ErrorMsg, name, line);
307 if (re == "EC")
308 throw EventCancelledException("发送信息被取消", name, line);
309 if (re == "ET")
310 throw TimeOutException("发送信息超时", name, line);
311 if (re == "EP")
312 throw BotException(name, line);
313 // equal to Tools::start_with
314 if (re.rfind("EBM", 0) == 0)
315 throw BotIsBeingMutedException(std::stoi(re.substr(3)), name, line);
316 }
