52 void timer(std::string pluginId, std::string content, size_t timeInSecond) {
53 auto timenow = std::chrono::system_clock::now();
54 auto scheduleTime = timenow + std::chrono::seconds(timeInSecond);
55 scheduleTask a{timenow, scheduleTime, std::move(pluginId), std::move(content)};
56 {
57 std::lock_guard lk(mtx);
58 timerQueue.push(std::move(a));
59 }
60 {
61 loaderWakeCV().notify_one();
62 }
63 }

71 bool checkValidTimeStamp(const scheduleTask &task) {
72 return task.createTime > PluginListManager::getPluginTimeStamp(task.pluginId);
73 }
74
75 void popSchedule() noexcept {
76 std::lock_guard lk(mtx);
77 while (timeup()) {
78 auto &task = timerQueue.top();
79 try {
80 if (checkValidTimeStamp(task)) {
81 sendTimeoutEvent(task.pluginId, task.content);
82 }
83 } catch (const PluginNotLoadedException &e) {
84 logger.error("未加载的插件传入了定时任务，任务失效");
85 e.raise();
86 }
87 timerQueue.pop();
88 }
89 }
90
91 bool timeup() noexcept {
92 return !timerQueue.empty() && timerQueue.top().tp < std::chrono::system_clock::now();
93 }
94} // namespace LibLoader::Scheduler
