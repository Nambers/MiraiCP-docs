33 // env != null, call from jni
34 JNIEXPORT jstring Verify(JNIEnv *env, jobject, jstring id) {
35 using namespace MiraiCP;
36 ThreadManager::setEnv(env);
37 MiraiCP::ThreadManager::JNIVersion = env->GetVersion();
38 try {
39 //初始化日志模块
40 Config::construct();
41 Logger::logger.init();
42 enrollPlugin();
43 // plugin == nullptr 无插件实例加载
44 if (CPPPlugin::plugin != nullptr) {
45 CPPPlugin::pluginLogger = new PluginLogger(&Logger::logger);
46 CPPPlugin::plugin->onEnable();
47 }
48 } catch (const MiraiCPExceptionBase &e) {
49 e.raise();
50 }
51 json j = (CPPPlugin::plugin != nullptr) ? CPPPlugin::plugin->config.serialize() : json::parse("{}");
52 j["MiraiCPversion"] = MiraiCPVersion;
53 Config::pluginId = std::stoi(Tools::jstring2str(id));
54 return Tools::str2jstring(j.dump().c_str());
55 //验证机制，返回当前SDK版本
56 }

63 if (CPPPlugin::plugin == nullptr) return job;

66 try {
67 CPPPlugin::plugin->onDisable();
68 } catch (const MiraiCPExceptionBase &e) {
69 e.raise();
70 }
71 CPPPlugin::plugin = nullptr;
72 return job;
73 }

80 /*
81 * 消息解析分流
82 * env != null, call from jni
83 */
84 JNIEXPORT jstring Event(JNIEnv *env, jobject, jstring content) {
85 using namespace MiraiCP;
86 ThreadManager::setEnv(env);
87 std::string tmp = Tools::jstring2str(content, env);
88 json j;
89 try {
90 j = json::parse(tmp);
91 } catch (json::parse_error &e) {
92 APIException("格式化json错误", MIRAICP_EXCEPTION_WHERE).raise();
93 Logger::logger.error("For debug:" + j.dump());
94 Logger::logger.error(e.what(), false);
95 return returnNull();
96 }
97 int type = j["type"].get<int>();
98
99 if (eventTypes(type) != eventTypes::Command && Event::processor.noRegistered(type)) return returnNull();
100 try {
101 switch (eventTypes(type)) {
102 case eventTypes::GroupMessageEvent: {
103 //GroupMessage
104 Event::processor.broadcast<GroupMessageEvent>(
105 GroupMessageEvent(j["group"]["botid"],
106 Group(Group::deserialize(j["group"])),
107 Member(Member::deserialize(j["member"])),
108 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
109 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
110 break;
111 }
112 case eventTypes::PrivateMessageEvent: {
113 //私聊消息
114 Event::processor.broadcast<PrivateMessageEvent>(
115 PrivateMessageEvent(j["friend"]["botid"],
116 Friend(Friend::deserialize(j["friend"])),
117 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
118 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
119 break;
120 }
121 case eventTypes::GroupInviteEvent:
122 //群聊邀请
123 Event::processor.broadcast<GroupInviteEvent>(
124 GroupInviteEvent(
125 j["source"]["botid"],
126 j["request"],
127 j["source"]["inviternick"],
128 j["source"]["inviterid"],
129 j["source"]["groupname"],
130 j["source"]["groupid"]));
131 break;
132 case eventTypes::NewFriendRequestEvent:
133 //好友
134 Event::processor.broadcast<NewFriendRequestEvent>(
135 NewFriendRequestEvent(
136 j["source"]["botid"],
137 j["request"],
138 j["source"]["fromid"],
139 j["source"]["fromgroupid"],
140 j["source"]["fromnick"],
141 j["source"]["message"]));
142 break;
143 case eventTypes::MemberJoinEvent:
144 //新成员加入
145 Event::processor.broadcast<MemberJoinEvent>(
146 MemberJoinEvent(
147 j["group"]["botid"],
148 j["jointype"],
149 Member(Member::deserialize(j["member"])),
150 Group(Group::deserialize(j["group"])),
151 j["inviterid"]));
152 break;
153 case eventTypes::MemberLeaveEvent:
154 //群成员退出
155 Event::processor.broadcast<MemberLeaveEvent>(MemberLeaveEvent(
156 j["group"]["botid"],
157 j["leavetype"],
158 j["memberid"],
159 Group(Group::deserialize(j["group"])),
160 j["operatorid"]));
161 break;
162 case eventTypes::RecallEvent:
163 Event::processor.broadcast<RecallEvent>(RecallEvent(
164 j["botid"],
165 j["etype"],
166 j["time"],
167 j["authorid"],
168 j["operatorid"],
169 j["ids"],
170 j["internalids"],
171 j["groupid"]));
172 break;
173 case eventTypes::BotJoinGroupEvent:
174 Event::processor.broadcast<BotJoinGroupEvent>(BotJoinGroupEvent(
175 j["group"]["botid"],
176 j["etype"],
177 Group(Group::deserialize(j["group"])),
178 j["inviterid"]));
179 break;
180 case eventTypes::GroupTempMessageEvent:
181 Event::processor.broadcast<GroupTempMessageEvent>(GroupTempMessageEvent(
182 j["group"]["botid"],
183 Group(Group::deserialize(j["group"])),
184 Member(Member::deserialize(j["member"])),
185 MessageChain::deserializationFromMessageSourceJson(json::parse(j["message"].get<std::string>()))
186 .plus(MessageSource::deserializeFromString(j["source"]))));
187 break;
188 case eventTypes::TimeOutEvent:
189 Event::processor.broadcast(TimeOutEvent(j["msg"]));
190 break;
191 case eventTypes::BotOnlineEvent:
192 Event::processor.broadcast(BotOnlineEvent(j["botid"]));
193 break;
194 case eventTypes::NudgeEvent:
195 Event::processor.broadcast(NudgeEvent(Contact::deserialize(j["from"]),
196 Contact::deserialize(j["target"]),
197 Contact::deserialize(j["subject"]),
198 j["botid"]));
199 break;
200 case eventTypes::BotLeaveEvent:
201 Event::processor.broadcast(BotLeaveEvent(j["groupid"], j["botid"]));
202 break;
203 case eventTypes::MemberJoinRequestEvent: {
204 std::optional<Group> a;
205 std::optional<Member> b;
206 Contact temp = Contact::deserialize(j["group"]);
207 if (temp.id() == 0)
208 a = std::nullopt;
209 else
210 a = Group(temp);
211 temp = Contact::deserialize(j["inviter"]);
212 if (temp.id() == 0)
213 b = std::nullopt;
214 else
215 b = Member(temp);
216 Event::processor.broadcast(MemberJoinRequestEvent(a, b, temp.botid(), j["requester"], j["requestData"]));
217 break;
218 }
219 case eventTypes::MessagePreSendEvent: {
220 Event::processor.broadcast(MessagePreSendEvent(Contact::deserialize(j["target"]), MessageChain::deserializationFromMessageSourceJson(j["message"].get<std::string>(), false), j["botid"]));
221 break;
222 }
223 case eventTypes::Command: {
224 // command
225 std::optional<Contact> c = std::nullopt;
226 if (j.contains("contact")) c = Contact::deserialize(j["contact"]);
227 CommandManager::commandManager[j["bindId"]]->onCommand(c, Bot(j["botid"]), MessageChain::deserializationFromMessageSourceJson((j.contains("message") ? j["message"].get<std::string>() : ""), false));
228 break;
229 }
230 default:
231 throw APIException("Unreachable code", MIRAICP_EXCEPTION_WHERE);
232 }
233 } catch (json::type_error &e) {
234 Logger::logger.error("json格式化异常,位置C-Handle");
235 Logger::logger.error(e.what());
236 Logger::logger.error("info:", tmp);
237 return Tools::str2jstring("ERROR");
238 } catch (const MiraiCPExceptionBase &e) {
239 e.raise();
240 return Tools::str2jstring("ERROR");
241 } catch (const std::exception &e) {
242 // 这里如果不catch全部exception就会带崩jvm
243 Logger::logger.error(e.what());
244 Logger::logger.error("info:", tmp);
245 return Tools::str2jstring("ERROR");
246 }
247 return returnNull();
248 }
249 // env != null
250 int registerMethods(JNIEnv *env, const char *className, JNINativeMethod *gMethods, int numMethods) {
251 jclass clazz = env->FindClass(className);
252 if (clazz == nullptr) {
253 return JNI_FALSE;
254 }
255 //注册native方法
256 if (env->RegisterNatives(clazz, gMethods, numMethods) < 0) {
257 return JNI_FALSE;
258 }
259 return JNI_TRUE;
260 }
261 JNINativeMethod method_table[]{
262 {(char *) "Verify", (char *) "(Ljava/lang/String;)Ljava/lang/String;", (jstring *) Verify},
263 {(char *) "Event", (char *) "(Ljava/lang/String;)Ljava/lang/String;", (jstring *) Event},
264 {(char *) "PluginDisable", (char *) "()Ljava/lang/Void;", (jobject *) PluginDisable}};
265} // namespace MiraiCP::JNIApi
266extern "C" JNIEXPORT jint JNI_OnLoad(JavaVM *vm, void *) {
267 JNIEnv *env = nullptr;
268 if (vm->GetEnv((void **) &env, JNI_VERSION_1_6) != JNI_OK) {
269 return JNI_ERR;
270 }
271 assert(env != nullptr);
272 MiraiCP::ThreadManager::gvm = vm;
273 // 注册native方法
274 if (!MiraiCP::JNIApi::registerMethods(env, "tech/eritquearcus/miraicp/shared/CPPLib", MiraiCP::JNIApi::method_table, 3)) {
275 return JNI_ERR;
276 }
277 return JNI_VERSION_1_6;
278}
