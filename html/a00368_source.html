46 MessageSource MessageSource::deserializeFromString(const std::string &source) {
47 json j = json::parse(source);
48 try {
49 return {j["ids"].dump(), j["internalIds"].dump(), source};
50 } catch (json::type_error &e) {
51 Logger::logger.error("消息源序列化出错，格式不符合(MessageSource::deserializeFromString)");
52 Logger::logger.error(source);
53 Logger::logger.error(e.what());
54 throw IllegalArgumentException(std::string("消息源序列化出错，格式不符合(MessageSource::deserializeFromString), ") + e.what(), MIRAICP_EXCEPTION_WHERE);
