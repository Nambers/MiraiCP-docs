688 class eventNode {
689 private:
691 std::shared_ptr<NodeHandle> _handle;
692
693 public:
694 std::function<bool(MiraiCPEvent *)> func;
695
696 public:
697 eventNode() : _handle(), func(nullptr) {}
698
699 explicit eventNode(std::function<bool(MiraiCPEvent *)> f) : _handle(new NodeHandle(true)), func(std::move(f)) {}
700
701 eventNode(const eventNode &_o) = default; // for MSVC compatible, or you will get an error
702 eventNode(eventNode &&_o) noexcept : _handle(std::move(_o._handle)), func(std::move(_o.func)) {}
703
704 public:
706 bool run(MiraiCPEvent *a) const {
707 return _handle->isEnable() && func(a);
708 }

710 std::shared_ptr<NodeHandle> getHandle() {
711 return _handle;
712 }

730 constexpr static int id() {
731 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持广播继承MiraiCPEvent的事件");
732 return static_cast<int>(EventClass::get_event_type());
733 }

735 public:
736 static bool noRegistered(int index) {
737 return processor._all_events_[index].empty();
738 }

742 for (auto &a: processor._all_events_) a.clear();

747 template<typename EventClass>
748 static void broadcast(EventClass &&val) {
749 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持广播MiraiCPEvent的派生类");
750 MiraiCPEvent *p = &val;
751 for (auto &&[k, v]: processor._all_events_[id<EventClass>()]) {
752 for (auto &&a: v) {
753 if (a.run(p)) return;
754 }
755 }

765 template<typename EventClass>
766 static std::shared_ptr<NodeHandle> registerEvent(std::function<void(EventClass)> callback, priority_level level = 100) {
767 static_assert(std::is_base_of_v<MiraiCPEvent, EventClass>, "只支持注册MiraiCPEvent的派生类事件");
768 std::function<bool(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
769 callback(*static_cast<EventClass *>(p));
770 return false;
771 };
772 auto t = eventNode(tmp);
773 auto ans = t.getHandle();
774 // 先获得shared_ptr才可以emplace_back
775 processor._all_events_[id<EventClass>()][level].emplace_back(std::move(t));
776 return ans;
777 }
