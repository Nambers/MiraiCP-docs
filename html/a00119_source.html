71 template<class T>
72 T get() const {
73 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
74 if (T::type() != this->type())
75 throw IllegalArgumentException("cannot convert from " + SingleMessage::messageType[this->type()] + " to " + SingleMessage::messageType[T::type()], MIRAICP_EXCEPTION_WHERE);

261 static size_t findEnd(const std::string &s, size_t start) {
262 size_t pos = start;
263 while (pos < s.length()) {
264 switch (s[pos]) {
265 case '\\':
266 pos += 2;
267 continue;
268 case ']':
269 return pos;
270 }
271 pos++;
272 }
273 return -1;
274 }
