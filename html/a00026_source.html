28class Main : public CPPPlugin {
29public:
30 Main() : CPPPlugin() {}
31 void onEnable() override {
32 Event::registerEvent<GroupMessageEvent>([](GroupMessageEvent e) {
33 // 发送MessageChain
34 e.group.sendMessage(e.message);
35 // 发送SingleMessage
36 e.group.sendMessage(PlainText("a"));
37 e.group.sendMessage(e.sender.at(), PlainText(""), Face(5));
38 // 回复信息
39 e.group.quoteAndSendMessage(PlainText("a"), e.message.source.value());
40 // 撤回信息
41 e.message.source.value().recall();
42 e.group.sendMessage("hi").recall();
43 });
44 }
45};
