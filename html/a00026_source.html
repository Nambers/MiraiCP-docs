538 class Event {
539 private:
540 template<typename T>
541 eventTypes id() const {
542 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播继承MiraiCPEvent的事件");
543 return T::getEventType();
544 }
545
546 Event() = default;
547
548 class eventNode {
549 public:
550 bool enable = true;
551 std::function<void(MiraiCPEvent *)> func;
552
553 explicit eventNode(std::function<void(MiraiCPEvent *)> f) : func(std::move(f)) {}
554
555 void run(MiraiCPEvent *a) const {
556 func(a);
557 }
558 };
559
560 std::vector<std::vector<eventNode>> vec = std::vector<std::vector<eventNode>>(int(eventTypes::count));
561
562 public:
564 class NodeHandle {
565 private:
566 bool *enable;
567
568 public:
569 explicit NodeHandle(bool *a) { this->enable = a; }
570 void stop() { *enable = false; }
571 void resume() { *enable = true; }
572 };
573
574 // singleton mode
575 static Event processor;
576
578 template<typename T>
579 void broadcast(T &&val) {
580 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播MiraiCPEvent的派生类");
581 MiraiCPEvent *p = &val;
582 for (auto &a: vec[int(id<T>())]) {
583 a.run(p);
584 }
585 }
586
588 template<typename T>
589 NodeHandle registerEvent(std::function<void(T)> a) {
590 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持注册MiraiCPEvent的派生类事件");
591 std::function<void(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
592 a(*dynamic_cast<T *>(p));
593 };
594 eventNode t(tmp);
595 vec[int(id<T>())].emplace_back(t);
596 return NodeHandle(&t.enable);
597 }
598 };
