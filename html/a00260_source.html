117 PluginFuncAddrData Plugin::testSymbolExistance(plugin_handle handle, const std::string &path) {
118 auto enter = (plugin_entrance_func_ptr) LoaderApi::libSymbolLookup(handle, STRINGIFY(FUNC_ENTRANCE));
119 if (!enter) throw SymbolResolveException(path, SymbolResolveException::Entrance, MIRAICP_EXCEPTION_WHERE);
120
121 auto exiting = (plugin_func_ptr) LoaderApi::libSymbolLookup(handle, STRINGIFY(FUNC_EXIT));
122 if (!exiting) throw SymbolResolveException(path, SymbolResolveException::Exit, MIRAICP_EXCEPTION_WHERE);
123
124 auto event_addr = (plugin_event_func_ptr) LoaderApi::libSymbolLookup(handle, STRINGIFY(FUNC_EVENT));
125 if (!event_addr) throw SymbolResolveException(path, SymbolResolveException::Event, MIRAICP_EXCEPTION_WHERE);
126
127 auto pluginInfo = (plugin_info_func_ptr) LoaderApi::libSymbolLookup(handle, STRINGIFY(PLUGIN_INFO));
128 if (!pluginInfo) throw SymbolResolveException(path, SymbolResolveException::Config, MIRAICP_EXCEPTION_WHERE);
129
130 if (!checkPluginInfoValid(pluginInfo)) throw InfoNotCompleteException(path, MIRAICP_EXCEPTION_WHERE);
131
132 return {enter, event_addr, exiting, pluginInfo};
133 }

171 plugin_handle Plugin::loadNewPlugin(const std::string &path) noexcept {
172 try {
173 auto actualPath = path; // NOLINT(performance-unnecessary-copy-initialization)
174#if MIRAICP_WINDOWS
175 auto from = std::filesystem::path(path);
176 if (!exists(from) || !from.has_extension()) {
177 logger.error("path don't exist or invalid " + path);
178 return nullptr;
179 }
180 auto actualFile = std::filesystem::temp_directory_path().append(
181 std::to_string(std::hash<std::string>()(path)) + ".dll");
182 actualPath = actualFile.string();
183 // try to del it anyway
184 // works around with https://github.com/msys2/MSYS2-packages/issues/1937
185 std::error_code ec;
186 std::filesystem::remove(actualFile, ec);
187 ec.clear();
188 std::filesystem::copy(path, actualFile, std::filesystem::copy_options::overwrite_existing, ec);
189 if (ec) {
190 logger.error("无法复制dll(" + path + ")到缓存目录(" + actualPath + "), 原因: " + ec.message() + "，请检查是否有别的进程在占用该缓存文件！");
191 return nullptr;
192 }
193#endif
194 return LoaderApi::libOpen(actualPath);
195 } catch (...) {}
196 return nullptr;
197 }

201 void Plugin::enableInternal() {
202 if (handle == nullptr) {
203 throw PluginNotLoadedException(path, MIRAICP_EXCEPTION_WHERE);
204 }
205
206 if (enabled) {
207 assert(infoFunc);
208 throw PluginAlreadyEnabledException(_getId(), MIRAICP_EXCEPTION_WHERE);
209 }
210
211 if (infoFunc()->getMVersion() != MiraiCP::MiraiCPVersion) {
212 LibLoader::logger.warning("MiraiCP依赖库版本(" + infoFunc()->getMVersion() + ")和libLoader版本(" + MiraiCP::MiraiCPVersion + ")不一致, 建议更新到最新");
213 }
214
215 _enable();
216 updateTimeStamp();
217
218 Antares::pool->push_task([this] {
219 std::shared_lock lk(_mtx);
220
221 if (nullptr == entrance) {
222 // 并发过快，插件在启用前被卸载了。直接return
223 return;
224 }
225
226 int ret = -1;
227
228 {
229 ThreadIdentify::setThreadWorkingName(_getId());
230 MIRAICP_DEFER(ThreadIdentify::unsetThreadWorkingName(););
231 try {
232 ret = callEntranceByAuthority();
233 } catch (...) {
234 }
235 }
236
237 if (ret != 0) {
238 logger.error("插件：" + _getId() + "启用时出现错误");
239 }
240 });
241 }

243 std::shared_ptr<std::future<void>> Plugin::disableInternal(bool lockedAndWait) {
244 if (exit == nullptr) return nullptr;
245 _disable();
246 updateTimeStamp();
247
248 return std::make_shared<std::future<void>>(Antares::pool->submit([this, lockedAndWait] {
249 std::shared_lock lk(_mtx, std::defer_lock);
250
251 if (!lockedAndWait) lk.lock();
252 if (!checkValid() || exit == nullptr) {
253 // 任务分派过慢
254 return;
255 }
256
257 int ret = -1;
258
259 {
260 ThreadIdentify::setThreadWorkingName(_getId());
261 MIRAICP_DEFER(ThreadIdentify::unsetThreadWorkingName(););
262 try {
263 ret = exit();
264 } catch (...) {}
265 }
266
267 assert(infoFunc);
268
269 if (ret != 0) logger.error("插件：" + _getId() + "退出时出现错误");
270 }));
271 }

273 void Plugin::unloadInternal() {
274 if (nullptr == handle) {
275 // DON'T CALL getIdSafe() if plugin is not valid!!!
276 logger.warning("plugin at path: " + path + " is already unloaded");
277 return;
278 }
279
280 // first disable it
281 if (enabled) {
282 auto fu = disableInternal(true);
283 if (fu) {
284 fu->wait();
285 }
286 }
287
288 // then unload it
289 LoaderApi::libClose(handle);
290
291 _unload();
292 }

294 void Plugin::loadInternal(bool alsoEnablePlugin) {
295 if (handle != nullptr) {
296 assert(infoFunc);
297 throw PluginAlreadyLoadedException(_getId(), MIRAICP_EXCEPTION_WHERE);
298 }
299
300 handle = loadNewPlugin(path); // nothrow
301
302 if (handle == nullptr) {
303 logger.error(LoaderApi::libError());
304 throw PluginLoadException(path, MIRAICP_EXCEPTION_WHERE);
305 }
306
307 auto symbols = testSymbolExistance(handle, path); // throws
308
309 _load(handle, symbols);
310 _disable();
311
312 if (alsoEnablePlugin) enableInternal();
313 }

329 void Plugin::pushEvent_worker(const MiraiCP::MiraiCPString &event) {
330 std::shared_lock lk(_mtx);
331 if (!enabled) {
332 // 线程池任务分派过慢。插件此时已经被卸载，直接return
333 return;
334 }
335
336 int ret = -1;
337
338 {
339 ThreadIdentify::setThreadWorkingName(_getId());
340 MIRAICP_DEFER(ThreadIdentify::unsetThreadWorkingName(););
341 try {
342 ret = eventFunc(event);
343 } catch (...) {
344 }
345 }
346
347 assert(infoFunc);
348
349 if (ret != 0) {
350 logger.error("插件：" + _getId() + "出现严重错误");
351 logger.error(event);
352 }
353 }

362 void Plugin::loadsAll(const std::vector<std::string> &paths, const std::vector<PluginAuthority> &authorities) noexcept {
363 // std::lock_guard lk(PluginListManager::getLock());
364
365 for (size_t i = 0; i < paths.size(); ++i) {
366 auto &path = paths[i];
367 auto authority = authorities[i];
368
369 auto newPlugin = std::make_shared<Plugin>(path);
370
371 auto timestamp = std::chrono::system_clock::now();
372
373 try {
374 newPlugin->loadPlugin(false);
375 } catch (LoaderBaseException &e) {
376 e.raise();
377 continue;
378 }
379
380 auto timestamp2 = std::chrono::system_clock::now();
381
382 auto timedelta = std::chrono::duration_cast<std::chrono::milliseconds>(timestamp2 - timestamp).count();
383 logger.info("loaded plugin " + path + " in " + std::to_string(timedelta) + "ms");
384
385 newPlugin->authority = authority;
386
387 try {
388 PluginListManager::addNewPlugin(newPlugin);
389 } catch (LoaderBaseException &e) {
390 e.raise();
391 }
392 }
393 }
