32class EventHandlerPitch {
33public:
34 static long __stdcall eventHandler(PEXCEPTION_POINTERS pExceptionPointers) {
35 if (alreadyInHandler) {
36 return EXCEPTION_CONTINUE_EXECUTION;
37 }
38 auto pluginName = ThreadIdentify::getThreadWorkingName();
39
40 if (pluginName.empty()) {
41 // test the thread is from jvm
42 if (ThreadIdentify::isMeLoaderThread()) {
43 LibLoader::logger.error("libLoader线程遇到致命错误，请向MiraiCP仓库提交您的报错信息以及堆栈信息");
44 exit(1);
45 } else if (ThreadIdentify::isMePoolThread()) {
46 // 非插件导致的工作线程致命错误
47 LibLoader::logger.error("libLoader工作线程遇到致命错误，请向MiraiCP仓库提交您的报错信息以及堆栈信息");
48 exit(1);
49 }
50
51 std::string maybeId = ThreadIdentify::identifyMe();
52 // 插件列表中查询是否有这个线程名的插件，如果没有则放弃处理，有则卸载
53 if (LibLoader::PluginListManager::pluginNameLookup(maybeId)) {
54 pluginName = std::move(maybeId);
55 } else {
56 return EXCEPTION_CONTINUE_EXECUTION;
57 }
58 }
59
60 // 插件导致的崩溃，卸载该插件
61 alreadyInHandler = true;
62 MIRAICP_DEFER(alreadyInHandler = false;);
63 char hexTemp[20];
64 sprintf_s(hexTemp, 20, "%lX", pExceptionPointers->ExceptionRecord->ExceptionCode);
65 LibLoader::logger.error("插件 " + pluginName + " 遇到致命错误！工作线程异常终止，ExceptionCode: 0x" + hexTemp);
66 LibLoader::logger.error("请检查您的插件代码，崩溃发生后的任何行为均未定义，MiraiCP将尽可能尝试继续运行");
67 LibLoader::sendPluginException(std::move(pluginName));
68
69 size_t threadIndex = BS::thread_pool::getCurrentThreadIndexView();
70 if (threadIndex != (std::numeric_limits<size_t>::max)()) {
71 // 是线程池线程
72 LibLoader::sendThreadReset(threadIndex);
73 }
74 //
75 TerminateThread(GetCurrentThread(), 1);
76 return EXCEPTION_CONTINUE_EXECUTION;
77 }
78
79private:
80 EventHandlerPitch() {
81 preHandler = SetUnhandledExceptionFilter(EventHandlerPitch::eventHandler);
82 }
83
84 ~EventHandlerPitch() {
85 SetUnhandledExceptionFilter(preHandler);
86 }
87
88 static EventHandlerPitch pitch;
89
90private:
91 LPTOP_LEVEL_EXCEPTION_FILTER preHandler;
92};

106class [[maybe_unused]] SignalHandle {
107 SignalHandle() noexcept {
108 auto &act = getAct();
109 act.sa_flags = SA_SIGINFO;
110 sigemptyset(&act.sa_mask);
111 act.sa_sigaction = &SignalHandle::sigsegv_handler;
112 sigaction(SIGSEGV, &act, &getOact());
113 }
114
115 ~SignalHandle() {
116 sigaction(SIGSEGV, &getOact(), &getAct());
117 }
118
119private:
120 static struct sigaction &getAct() {
121 static struct sigaction act;
122 return act;
123 }
124
125 static struct sigaction &getOact() {
126 static struct sigaction oact;
127 return oact;
128 }
129
130 static void sigsegv_handler(int a, siginfo_t *si, void *unused) {
131 if (alreadyInHandler) {
132 return;
133 }
134 auto pluginName = ThreadIdentify::getThreadWorkingName();
135 if (pluginName.empty()) {
136 // test the thread is from jvm
137 if (ThreadIdentify::isMeLoaderThread()) {
138 LibLoader::logger.error("libLoader线程遇到致命错误，请向MiraiCP仓库提交您的报错信息以及堆栈信息");
139 exit(1);
140 } else if (ThreadIdentify::isMePoolThread()) {
141 // 非插件导致的工作线程致命错误
142 LibLoader::logger.error("libLoader工作线程遇到致命错误，请向MiraiCP仓库提交您的报错信息以及堆栈信息");
143 exit(1);
144 }
145
146 std::string maybeId = ThreadIdentify::identifyMe();
147 // 插件列表中查询是否有这个线程名的插件，如果没有则放弃处理，有则卸载
148 if (LibLoader::PluginListManager::pluginNameLookup(maybeId)) {
149 pluginName = std::move(maybeId);
150 } else {
151 getOact().sa_sigaction(a, si, unused);
152 return;
153 }
154 }
155 // 插件导致的崩溃，卸载该插件
156 alreadyInHandler = true;
157 MIRAICP_DEFER(alreadyInHandler = false;);
158 LibLoader::logger.error(
159 "插件" + pluginName + "遇到致命错误! 插件运行终止，信号： SIG" +
160#if __GLIBC__ >= 2 && __GLIBC_MINOR__ >= 32
161 sigabbrev_np(si->si_signo)
162#else
163 "NAL" + std::to_string(si->si_signo)
164#endif
165 );
166 LibLoader::logger.error("请检查您的插件代码，崩溃发生后的任何行为均未定义，MiraiCP将尽可能尝试继续运行");
167 LibLoader::sendPluginException(std::move(pluginName));
168
169 size_t threadIndex = BS::thread_pool::getCurrentThreadIndexView();
170 if (threadIndex != (std::numeric_limits<size_t>::max)()) {
171 // 是线程池线程
172 LibLoader::sendThreadReset(threadIndex);
173 }
174#ifdef PTHREAD_CANCEL_SUPPORTED
175 pthread_cancel(pthread_self());
176#endif
177 }
178
179private:
180 static SignalHandle Handler;
181};
