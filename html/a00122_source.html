168 class MIRAICP_EXPORT MemberException : public MiraiCPExceptionCRTP<MemberException> {
169 public:
170 enum MemberExceptionType : int {
171 OtherType,
172 NoSuchGroup,
173 NoSuchMember
174 };
175 MemberExceptionType type = OtherType;
176
177 /*
178 * "1" - 找不到群
179 * "2" - 找不到群成员
180 */
181 explicit MemberException(int _type, string _filename, int _lineNum) : MiraiCPExceptionCRTP(
182 [&]() -> string {
183 type = MemberExceptionType(_type);
184 switch (type) {
185 case NoSuchGroup:
186 return "找不到群";
187 case NoSuchMember:
188 return "找不到群成员";
189 default:
190 return "";
191 }
192 }(),
193 std::move(_filename), _lineNum) {}
194
195 static string exceptionType() { return "MemberException"; }
196 };
