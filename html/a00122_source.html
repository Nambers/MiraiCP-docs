32using json = nlohmann::json;
33// env != null, call from jni
34JNIEXPORT jstring Verify(JNIEnv *env, jobject, jstring id) {
35 using namespace MiraiCP;
36 ThreadManager::setEnv(env);
37 MiraiCP::ThreadManager::JNIVersion = env->GetVersion();
38 try {
39 //初始化日志模块
40 Config::construct();
41 Logger::logger.init();
42 enrollPlugin();
43 if (CPPPlugin::plugin == nullptr) Logger::logger.error("无插件实例加载");
44 else {
45 CPPPlugin::pluginLogger = new PluginLogger(&Logger::logger);
46 CPPPlugin::plugin->onEnable();
47 }
48 } catch (const MiraiCPException &e) {
49 e.raise();
50 }
51 json j = CPPPlugin::plugin->config.serialize();
52 j["MiraiCPversion"] = MiraiCPVersion;
53 Config::pluginId = std::stoi(Tools::jstring2str(id));
54 return Tools::str2jstring(j.dump().c_str());
55 //验证机制，返回当前SDK版本
56}

83JNIEXPORT jstring Event(JNIEnv *env, jobject, jstring content) {
84 using namespace MiraiCP;
85 ThreadManager::setEnv(env);
86 std::string tmp = Tools::jstring2str(content, env);
87 json j;
88 try {
89 j = json::parse(tmp);
90 } catch (json::parse_error &e) {
91 APIException("格式化json错误").raise();
92 Logger::logger.error("For debug:" + j.dump());
93 Logger::logger.error(e.what(), false);
94 return returnNull();
95 }
96 int type = j["type"].get<int>();
97
98 if (eventTypes(type) != eventTypes::Command && Event::processor.noRegistered(type)) return returnNull();
99 try {
100 switch (eventTypes(type)) {
101 case eventTypes::GroupMessageEvent: {
102 //GroupMessage
103 Event::processor.broadcast<GroupMessageEvent>(
104 GroupMessageEvent(j["group"]["botid"],
105 Group(Group::deserialize(j["group"])),
106 Member(Member::deserialize(j["member"])),
107 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
108 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
109 break;
110 }
111 case eventTypes::PrivateMessageEvent: {
112 //私聊消息
113 Event::processor.broadcast<PrivateMessageEvent>(
114 PrivateMessageEvent(j["friend"]["botid"],
115 Friend(Friend::deserialize(j["friend"])),
116 MessageChain::deserializationFromMessageSourceJson(json::parse(j["source"].get<std::string>()))
117 .plus(MessageSource::deserializeFromString(j["source"].get<std::string>()))));
118 break;
119 }
120 case eventTypes::GroupInviteEvent:
121 //群聊邀请
122 Event::processor.broadcast<GroupInviteEvent>(
123 GroupInviteEvent(
124 j["source"]["botid"],
125 j["request"],
126 j["source"]["inviternick"],
127 j["source"]["inviterid"],
128 j["source"]["groupname"],
129 j["source"]["groupid"]));
130 break;
131 case eventTypes::NewFriendRequestEvent:
132 //好友
133 Event::processor.broadcast<NewFriendRequestEvent>(
134 NewFriendRequestEvent(
135 j["source"]["botid"],
136 j["request"],
137 j["source"]["fromid"],
138 j["source"]["fromgroupid"],
139 j["source"]["fromnick"],
140 j["source"]["message"]));
141 break;
142 case eventTypes::MemberJoinEvent:
143 //新成员加入
144 Event::processor.broadcast<MemberJoinEvent>(
145 MemberJoinEvent(
146 j["group"]["botid"],
147 j["jointype"],
148 Member(Member::deserialize(j["member"])),
149 Group(Group::deserialize(j["group"])),
150 j["inviterid"]));
151 break;
152 case eventTypes::MemberLeaveEvent:
153 //群成员退出
154 Event::processor.broadcast<MemberLeaveEvent>(MemberLeaveEvent(
155 j["group"]["botid"],
156 j["leavetype"],
157 j["memberid"],
158 Group(Group::deserialize(j["group"])),
159 j["operatorid"]));
160 break;
161 case eventTypes::RecallEvent:
162 Event::processor.broadcast<RecallEvent>(RecallEvent(
163 j["botid"],
164 j["etype"],
165 j["time"],
166 j["authorid"],
167 j["operatorid"],
168 j["ids"],
169 j["internalids"],
170 j["groupid"]));
171 break;
172 case eventTypes::BotJoinGroupEvent:
173 Event::processor.broadcast<BotJoinGroupEvent>(BotJoinGroupEvent(
174 j["group"]["botid"],
175 j["etype"],
176 Group(Group::deserialize(j["group"])),
177 j["inviterid"]));
178 break;
179 case eventTypes::GroupTempMessageEvent:
180 Event::processor.broadcast<GroupTempMessageEvent>(GroupTempMessageEvent(
181 j["group"]["botid"],
182 Group(Group::deserialize(j["group"])),
183 Member(Member::deserialize(j["member"])),
184 MessageChain::deserializationFromMessageSourceJson(json::parse(j["message"].get<std::string>()))
185 .plus(MessageSource::deserializeFromString(j["source"]))));
186 break;
187 case eventTypes::TimeOutEvent:
188 Event::processor.broadcast(TimeOutEvent(j["msg"]));
189 break;
190 case eventTypes::BotOnlineEvent:
191 Event::processor.broadcast(BotOnlineEvent(j["botid"]));
192 break;
193 case eventTypes::NudgeEvent:
194 Event::processor.broadcast(NudgeEvent(Contact::deserialize(j["from"]),
195 Contact::deserialize(j["target"]),
196 j["botid"]));
197 break;
198 case eventTypes::BotLeaveEvent:
199 Event::processor.broadcast(BotLeaveEvent(j["groupid"], j["botid"]));
200 break;
201 case eventTypes::MemberJoinRequestEvent: {
202 std::optional<Group> a;
203 std::optional<Member> b;
204 Contact temp = Contact::deserialize(j["group"]);
205 if (temp.id() == 0)
206 a = std::nullopt;
207 else
208 a = Group(temp);
209 temp = Contact::deserialize(j["inviter"]);
210 if (temp.id() == 0)
211 b = std::nullopt;
212 else
213 b = Member(temp);
214 Event::processor.broadcast(MemberJoinRequestEvent(a, b, temp.botid(), j["requestData"]));
215 break;
216 }
217 case eventTypes::MessagePreSendEvent: {
218 Event::processor.broadcast(MessagePreSendEvent(Contact::deserialize(j["target"]), MessageChain::deserializationFromMessageSourceJson(j["message"].get<std::string>(), false), j["botid"]));
219 break;
220 }
221 case eventTypes::Command: {
222 // command
223 std::optional<Contact> c = std::nullopt;
224 if (j.contains("contact")) c = Contact::deserialize(j["contact"]);
225 CommandManager::commandManager[j["bindId"]]->onCommand(c, Bot(j["botid"]), MessageChain::deserializationFromMessageSourceJson((j.contains("message") ? j["message"].get<std::string>() : ""), false));
226 break;
227 }
228 default:
229 MiraiCPThrow(APIException("Unreachable code"));
230 }
231 } catch (json::type_error &e) {
232 Logger::logger.error("json格式化异常,位置C-Handle");
233 Logger::logger.error(e.what());
234 Logger::logger.error("info:", tmp);
235 return Tools::str2jstring("ERROR");
236 } catch (const MiraiCPException &e) {
237 e.raise();
238 return Tools::str2jstring("ERROR");
239 } catch (const std::exception &e) {
240 // 这里如果不catch全部exception就会带崩jvm
241 Logger::logger.error(e.what());
242 Logger::logger.error("info:", tmp);
243 return Tools::str2jstring("ERROR");
244 }
245 return returnNull();
246}

248int registerMethods(JNIEnv *env, const char *className, JNINativeMethod *gMethods, int numMethods) {
249 jclass clazz = env->FindClass(className);
250 if (clazz == nullptr) {
251 return JNI_FALSE;
252 }
253 //注册native方法
254 if (env->RegisterNatives(clazz, gMethods, numMethods) < 0) {
255 return JNI_FALSE;
256 }
257 return JNI_TRUE;
258}

265extern "C" JNIEXPORT jint JNI_OnLoad(JavaVM *vm, void *) {
266 JNIEnv *env = nullptr;
267 if (vm->GetEnv((void **) &env, JNI_VERSION_1_6) != JNI_OK) {
268 return JNI_ERR;
269 }
270 assert(env != nullptr);
271 MiraiCP::ThreadManager::gvm = vm;
272 // 注册native方法
273 if (!registerMethods(env, "tech/eritquearcus/miraicp/shared/CPPLib", method_table, 3)) {
274 return JNI_ERR;
275 }
276 return JNI_VERSION_1_6;
277}
