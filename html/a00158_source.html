69 std::string SingleMessage::toMiraiCode() const {
70 // Logger::logger.info("base");
71 if (type > 0)
72 if (type == 1)
73 return "[mirai:at:" + content + "] ";
74 else if (type == 2)
75 return "[mirai:atall] ";
76 else
77 return "[mirai:" + messageType[type] + this->prefix + Tools::escapeToMiraiCode(content) + "]";
78 else
79 return content;
80 }

120 bool Image::isUploaded(QQID botid) {
121 if (!this->md5.has_value()) this->refreshInfo();
122 if (this->size == 0) throw IllegalArgumentException("size不能为0", MIRAICP_EXCEPTION_WHERE);
123 nlohmann::json tmp = this->toJson();
124 tmp["botid"] = botid;
125 std::string re = KtOperation::ktOperation(KtOperation::ImageUploaded, std::move(tmp));
126 return re == "true";
127 }

201 RemoteFile RemoteFile::deserializeFromString(const std::string &source) {
202 json j;
203 try {
204 j = json::parse(source);
205 } catch (json::parse_error &e) {
206 Logger::logger.error("格式化json失败，RemoteFile::deserializeFromString");
207 Logger::logger.error(source);
208 Logger::logger.error(e.what());
209 throw e;
210 }
211 try {
212 auto re = RemoteFile(j["id"], j["internalid"], j["name"], j["finfo"]["size"]);
213 if (j.contains("dinfo")) {
214 struct Dinfo d {
215 j["dinfo"]["url"],
216 j["dinfo"]["md5"],
217 j["dinfo"]["sha1"]
218 };
219 re.dinfo = d;
220 }
221 if (j["finfo"].contains("uploaderid")) {
222 struct Finfo f {
223 j["finfo"]["size"],
224 j["finfo"]["uploaderid"],
225 j["finfo"]["expirytime"],
226 j["finfo"]["uploadtime"],
227 j["finfo"]["lastmodifytime"]
228 };
229 re.finfo = f;
230 }
231 if (j.contains("path"))
232 re.path = j["path"];
233 return re;
234 } catch (json::type_error &e) {
235 Logger::logger.error("json格式化失败，位置:RemoteFile");
236 Logger::logger.error(source);
237 Logger::logger.error(e.what());
238 throw e;
239 }
240 }

249 std::string RemoteFile::serializeToString() {
250 json j;
251 if (this->dinfo.has_value()) {
252 j["dinfo"]["url"] = this->dinfo->url;
253 j["dinfo"]["md5"] = this->dinfo->md5;
254 j["dinfo"]["shar1"] = this->dinfo->sha1;
255 }
256 if (this->finfo.has_value()) {
257 j["finfo"]["size"] = this->finfo->size;
258 j["finfo"]["uploaderid"] = this->finfo->uploaderid;
259 j["finfo"]["expirytime"] = this->finfo->expirytime;
260 j["finfo"]["uploadtime"] = this->finfo->uploadtime;
261 j["finfo"]["lastmodifytime"] = this->finfo->lastmodifytime;
262 }
263 j["id"] = this->id;
264 j["internalid"] = this->internalid;
265 j["name"] = this->name;
266 j["size"] = this->size;
267 if (this->path.has_value())
268 j["path"] = this->path.value();
269 return j.dump();
270 }

273 void Image::refreshInfo() {
274 std::string re = KtOperation::ktOperation(KtOperation::QueryImgInfo, toJson());
275 if (re == "E1")
276 throw RemoteAssetException("图片id格式错误", MIRAICP_EXCEPTION_WHERE);
277 json j = json::parse(re);
278 this->url = Tools::json_stringmover(j, "url");
279 this->md5 = Tools::json_stringmover(j, "md5");
280 this->size = j["size"];
281 this->width = j["width"];
282 this->height = j["height"];
283 this->imageType = j["type"];
284 }
