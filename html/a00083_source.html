263 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
264 if (re == "EF")
265 throw FriendException(name, line);
266 if (re == "EG")
267 throw GroupException(name, line);
268 if (re == "EM")
269 throw MemberException(1, name, line);
270 if (re == "EMM")
271 throw MemberException(2, name, line);
272 if (re == "EB")
273 throw BotException("找不到bot:" + re, name, line);
274 if (re == "EA")
275 throw APIException(ErrorMsg, name, line);
276 if (re == "EC")
277 throw EventCancelledException("发送信息被取消", name, line);
278 if (re == "ET")
279 throw TimeOutException("发送信息超时", name, line);
280 if (re == "EP")
281 throw BotException(name, line);
282 // equal to Tools::start_with
283 if (re.rfind("EBM", 0) == 0)
284 throw BotIsBeingMutedException(std::stoi(re.substr(3)), name, line);
285 }
