52 template<class T>
53 T get() const {
54 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
55 if (T::type() != this->type())
56 throw IllegalArgumentException("cannot convert from " + SingleMessage::messageType[this->type()] + " to " + SingleMessage::messageType[T::type()], MIRAICP_EXCEPTION_WHERE);
57 T *re = static_cast<T *>(this->content.get());
58 if (re == nullptr)
59 throw IllegalArgumentException("cannot convert from " + SingleMessage::messageType[this->type()] + " to " + SingleMessage::messageType[T::type()], MIRAICP_EXCEPTION_WHERE);
60 return *re;
61 }

139 static size_t findEnd(const std::string &s, size_t start) {
140 size_t pos = start;
141 while (pos < s.length()) {
142 switch (s[pos]) {
143 case '\\':
144 pos += 2;
145 continue;
146 case ']':
147 return pos;
148 }
149 pos++;
150 }
151 return -1;
152 }
