258 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
259 if (re == "EF")
260 throw FriendException(name, line);
261 if (re == "EG")
262 throw GroupException(name, line);
263 if (re == "EM")
264 throw MemberException(1, name, line);
265 if (re == "EMM")
266 throw MemberException(2, name, line);
267 if (re == "EB")
268 throw BotException("找不到bot:" + re, name, line);
269 if (re == "EA")
270 throw APIException(ErrorMsg, name, line);
271 if (re == "EC")
272 throw EventCancelledException("发送信息被取消", name, line);
273 if (re == "ET")
274 throw TimeOutException("发送信息超时", name, line);
275 if (re == "EP")
276 throw BotException(name, line);
277 // equal to Tools::start_with
278 if (re.rfind("EBM", 0) == 0)
279 throw BotIsBeingMutedException(std::stoi(re.substr(3)), name, line);
280 }
