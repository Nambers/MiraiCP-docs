25 json ForwardedMessage::nodesToJson() {
26 json value;
27 for (const ForwardedNode &node: nodes) {
28 json temp;
29 temp["id"] = node.id;
30 temp["time"] = node.time;
31 if (node.isForwarded()) {
32 temp["isForwardedMessage"] = true;
33 temp["message"] = std::get<std::shared_ptr<ForwardedMessage>>(node.message)->nodesToJson().dump();
34 if (display.has_value())
35 temp["display"] = display->toJson();
36 } else
37 temp["message"] = std::get<MessageChain>(node.message).toMiraiCode();
38 temp["name"] = node.name;
39 value.emplace_back(std::move(temp));
40 }
41 json tmp = this->sendmsg;
42 tmp["value"] = std::move(value);
43 return tmp;
44 }

63 ForwardedMessage ForwardedMessage::deserializationFromMessageSourceJson(const json &j) {
64 std::vector<ForwardedNode> nodes;
65 try {
66 for (auto &&a: j) {
67 if (a["messageChain"][0].contains("kind") && a["messageChain"][0]["kind"] == "FORWARD") {
68 nodes.emplace_back(a["senderId"], a["senderName"], ForwardedMessage::deserializationFromMessageSourceJson(a["messageChain"][1]["nodeList"]), a["time"], ForwardedMessageDisplayStrategy::defaultStrategy());
69 } else {
70 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageSourceJson(a["messageChain"], false), a["time"]);
71 }
72 }
73 } catch (json::exception &e) {
74 throw APIException("ForwardedMessage格式化异常", MIRAICP_EXCEPTION_WHERE);
75 }
76
77 return {std::move(nodes), ForwardedMessageDisplayStrategy::defaultStrategy()};
78 }
