28 template<typename F, typename... Args, typename R = std::invoke_result_t<std::decay_t<F>, std::decay_t<Args>...>>
29 std::future<R> promiseTask(F &&func, Args &&...args) {
30 auto promise = std::make_shared<std::promise<R>>();
31 auto future = promise->get_future();
32 std::function<R()> taskfunction = std::bind(std::forward<F>(func), std::forward<Args>(args)...);
33 size_t function_id = internal::get_auto_incr_id();
34 auto to_push_func = std::make_shared<std::function<void()>>(
35 [promise = std::move(promise), function_id, taskfunction = std::move(taskfunction)]() {
36 try {
37 if constexpr (std::is_void_v<R>) {
38 std::invoke(taskfunction);
39 promise->set_value();
40 } else {
41 promise->set_value(std::invoke(taskfunction));
42 }
43 } catch (...) {
44 try {
45 promise->set_exception(std::current_exception());
46 Logger::logger.error("Failed to run task: " + std::to_string(function_id));
47 } catch (...) {
48 }
49 }
50 try {
51 internal::remove_task(function_id);
52 } catch (...) {
53 }
54 });
55 internal::push_task(function_id, std::move(to_push_func));
56 return future;
57 }
