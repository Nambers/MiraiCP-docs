48 MiraiCPString pluginOperation(const MiraiCPString &s) {
49#ifdef LOADER_NATIVE
50 auto nCharPtr = s.copyToCharPtr();
51 MIRAICP_DEFER(delete[] nCharPtr;);
52 return {LoaderAPIs::oper(nCharPtr)};
53#else
54 auto env = JNIEnvManager::getEnv();
55 auto tmp = jstring2str((jstring) env->CallStaticObjectMethod(JNIEnvs::Class_cpplib,
56 JNIEnvs::koper,
57 str2jstring(s.copyToCharPtr())));
58 return {tmp};
59#endif
60 }

82 void enablePluginById(const MiraiCPString &id) {
83 std::lock_guard lk(task_mtx);
84 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::ADD_THREAD, id));
85 loaderWakeCV().notify_one();
86 }

88 void disablePluginById(const MiraiCPString &id) {
89 std::lock_guard lk(task_mtx);
90 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::END_THREAD, id));
91 loaderWakeCV().notify_one();
92 }

106 void loadNewPlugin(const MiraiCPString &path, bool activateNow) {
107 std::lock_guard lk(task_mtx);
108 if (activateNow)
109 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::LOAD_NEW_ACTIVATENOW, path));
110 else
111 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::LOAD_NEW_DONTACTIVATE, path));
112 loaderWakeCV().notify_one();
113 }

118 loaderWakeCV().notify_one();
119 }

121 void reloadPluginById(const MiraiCPString &id) {

123 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::RELOAD, id));
