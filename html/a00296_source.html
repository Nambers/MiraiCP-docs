34 std::shared_ptr<Contact> Contact::deserializeToPointer(nlohmann::json j) {
35 uint8_t thistype = j["type"];
36 switch (thistype) {
37 case MIRAI_FRIEND: {
38 return std::make_shared<Friend>(std::move(j));
39 }
40 case MIRAI_MEMBER: {
41 return std::make_shared<Member>(std::move(j));
42 }
43 case MIRAI_GROUP: {
44 return std::make_shared<Group>(std::move(j));
45 }
46 default:
47 assert(0);
48 break;
49 }
50
51 return {};
52 }

54 MessageSource Contact::sendVoiceImpl(std::string path) const {
55 json j{{"path", std::move(path)},
56 {"contact", toJson()}};
57 std::string re = KtOperation::ktOperation(KtOperation::Voice, j);
58 if (re == "E1") {
59 throw UploadException("上传语音文件格式不对(必须为.amr/.silk)或文件不存在", MIRAICP_EXCEPTION_WHERE);
60 } else if (re == "E2") {
61 throw UploadException("上传语音文件大小超过服务器限制，一般限制在1MB上下", MIRAICP_EXCEPTION_WHERE);
62 }
63 return MessageSource::deserializeFromString(re);
64 }

116 MessageSource Contact::sendMsgImpl(std::string msg) const {
117 if (msg.empty()) {
118 throw IllegalArgumentException("不能发送空信息, 位置: Contact::SendMsg", MIRAICP_EXCEPTION_WHERE);
119 }
120 nlohmann::json j{{"message", std::move(msg)}, {"contact", toJson()}};
121 auto re = KtOperation::ktOperation(KtOperation::Send, j, true, "reach a error area, Contact::sendMsgImpl");
122 return MessageSource::deserializeFromString(re);
123 }
