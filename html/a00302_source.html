61 void MessageProcessor::processMessage(const MessageProxy &msg) {
62 bool invalid = true;
63 auto payload = msg.getPayload();
64 auto type = payload.payload_id;
65 MIRAICP_DEFER(if (invalid) logger.error("MessageProcessor::processMessage: invalid message type:" + std::to_string(type)););
66
67 if (type < 0 || type >= MsgType::MESSAGE_TYPE_COUNT) return;
68
69 auto handler = handlers[type];
70 if (!handler) return;
71
72 invalid = false;
73 handler(payload.payload, msg.getPlugin());
74 }
