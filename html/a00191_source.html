42// detect platform
43// ref: https://stackoverflow.com/questions/5919996/how-to-detect-reliably-mac-os-x-ios-linux-windows-in-c-preprocessor
44#if defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
45#undef MIRAICP_WINDOWS
46#define MIRAICP_WINDOWS 1
47#elif __APPLE__
48#include <TargetConditionals.h>
49#if TARGET_IPHONE_SIMULATOR || TARGET_OS_MACCATALYST || TARGET_OS_IPHONE
50#undef MIRAICP_IOS
51#define MIRAICP_IOS 1
52#elif TARGET_OS_MAC
53#undef MIRAICP_MACOS
54#define MIRAICP_MACOS 1
55#else
56static_assert(false, "Unknown apple platform");
57#endif
58#elif __ANDROID__
59#ifdef __TERMUX__
60#undef MIRAICP_TERMUX
61#define MIRAICP_TERMUX 1
62#else
63#undef MIRAICP_ANDROID
64#define MIRAICP_ANDROID 1
65#endif
66#elif __linux__
67#undef MIRAICP_LINUX
68#define MIRAICP_LINUX 1
69#elif __unix__
70#undef MIRAICP_UNIX
71#define MIRAICP_UNIX 1
72#else
73static_assert(false, "Unsupported platform");
74#endif

132 auto TOKEN_PASTE(_defered_statement_wrapper_, __LINE__) = [&]() { code }; \
133 CommonTools::MiraiCPDefer TOKEN_PASTE(_defered_object_, __LINE__)(std::move(TOKEN_PASTE(_defered_statement_wrapper_, __LINE__)))

176#define DECL_GETTER(attr) decltype(DataType::_##attr) attr();
177#define IMPL_GETTER(attr) \
178 decltype(LOC_CLASS_NAMESPACE::DataType::_##attr) LOC_CLASS_NAMESPACE::attr() { \
179 InternalData->requestRefresh(); \
180 MIRAICP_DATALOCK; \
181 return GetDataInternal()->_##attr; \
182 }
