61 void builtInCommand(const std::string &cmd) {
62 using nlohmann::json;
63 json j = json::parse(cmd);
64 auto it = actions.find(j["name"]);
65 if (it == actions.end()) {
66 logger.error("builtInCommand: unknown command: " + j["name"].get<std::string>());
67 return;
68 }
69 it->second(get_or_empty(j, "content"));
70 }
