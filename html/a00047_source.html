83 template<class T>
84 bool registerCommand(T command) {
85 static_assert(std::is_base_of_v<IRawCommand, T>, "只支持IRawCommand的派生类");
86 nlohmann::json j;
87 j["pluginId"] = Config::pluginId;
88 j["usage"] = command.config().usage;
89 j["primaryName"] = command.config().primaryName;
90 j["secondName"] = command.config().secondNames;
91 j["description"] = command.config().description;
92 j["override"] = command.config().override;
93 j["preFixOption"] = command.config().preFixOption;
94 size_t before = commandList.size();
95 std::shared_ptr<IRawCommand> c;
96 c.reset(new T(command));
97 commandList.push_back(c);
98 size_t now = commandList.size();
99 if (now - before == 1)
100 j["bindId"] = now - 1;
101 else {
102 auto i = std::find(commandList.begin(), commandList.end(), c);
103 if (i != commandList.end())
104 j["bindId"] = i - commandList.begin();
105 else
106 throw IllegalArgumentException("找不到合适的bindId", MIRAICP_EXCEPTION_WHERE);
107 }
108 nlohmann::json rej;
109 rej["command"] = j.dump();
110 std::string re = Config::koperation(Config::CommandReg, rej);
111 return re == "true";
112 }
113 static CommandManager commandManager;
114 };
