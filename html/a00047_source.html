76 template<class T>
77 bool registerCommand(T command) {
78 static_assert(std::is_base_of_v<IRawCommand, T>, "只支持IRawCommand的派生类");
79 nlohmann::json j;
80 j["pluginId"] = CPPPlugin::config.id;
81 j["usage"] = command.config().usage;
82 j["primaryName"] = command.config().primaryName;
83 j["secondName"] = command.config().secondNames;
84 j["description"] = command.config().description;
85 j["override"] = command.config().overrideOrigin;
86 j["preFixOption"] = command.config().preFixOption;
87 size_t before = commandList.size();
88 std::shared_ptr<IRawCommand> c;
89 c.reset(new T(command));
90 commandList.push_back(c);
91 size_t now = commandList.size();
92 if (now - before == 1)
93 j["bindId"] = now - 1;
94 else {
95 auto i = std::find(commandList.begin(), commandList.end(), c);
96 if (i != commandList.end())
97 j["bindId"] = i - commandList.begin();
98 else
99 throw IllegalArgumentException("找不到合适的bindId", MIRAICP_EXCEPTION_WHERE);
100 }
101 nlohmann::json rej{{"command", j.dump()}};
102 std::string re = KtOperation::ktOperation(KtOperation::CommandReg, std::move(rej));
103 return re == "true";
104 }
105 static CommandManager commandManager;
106 };
