31 void Group::OnlineAnnouncement::deleteThis() {
32 json j, i;
33 i["botid"] = this->botid;
34 i["groupid"] = this->groupid;
35 i["fid"] = this->fid;
36 i["type"] = 1;
37 j["identify"] = i.dump();
38 std::string re = Config::koperation(Config::Announcement, j);
39 if (re == "E1")
40 MiraiCPThrow(IllegalArgumentException("无法根据fid找到群公告(群公告不存在)"));
41 if (re == "E2")
42 MiraiCPThrow(BotException());
43 if (re == "E3")
44 MiraiCPThrow(IllegalStateException("群公告状态异常"));
45 }

140 RemoteFile Group::sendFile(const std::string &path, const std::string &filepath, JNIEnv *env) {
141 json tmp;
142 json source;
143 source["path"] = path;
144 source["filepath"] = filepath;
145 tmp["source"] = source.dump();
146 tmp["contactSource"] = this->serializationToString();
147 std::string callback = Config::koperation(Config::SendFile, tmp, env);
148 if (callback == "E2") MiraiCPThrow(UploadException("找不到" + filepath + "位置:C-uploadfile"));
149 if (callback == "E3")
150 MiraiCPThrow(UploadException("Upload error:路径格式异常，应为'/xxx.xxx'或'/xx/xxx.xxx'目前只支持群文件和单层路径, path:" + path));
151 return RemoteFile::deserializeFromString(callback);
152 }
