123 timer,

126 enableAllPlugins,
127 disableAllPlugins,
128 loadNewPlugin,
129 unloadPluginById,
130 reloadPluginById,
131 };
132 constexpr size_t line1 = __LINE__;
133 constexpr size_t allapi_construct_number = line1 - line0 - 3;
134 static_assert(allapi_construct_number == counter);
135 return t;
136 } else {
137 constexpr size_t line0 = __LINE__;
138 interface_funcs t2 = {
139 pluginOperation,
140 loggerInterface,
141 showAllPluginId,
142 pushTask,
143 pushTaskWithId,
144 timer,
145 }; // no admin functions
146 constexpr size_t line1 = __LINE__;

148 static_assert(adminapi_construct_number == interface_funcs::normal_api_count);
149 return t2;
150 }
