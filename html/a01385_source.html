59 bool ForwardedMessage::operator==(const ForwardedMessage &m) const {
60 if (this->nodes.size() != m.nodes.size()) return false;
61 int i = 0;
62 if (!(this->display == m.display)) return false;
63 return std::all_of(this->nodes.begin(), this->nodes.end(), [&i, &m](const auto &n) {
64 return n.message == m[i++].message;
65 });
66 }

91 ForwardedMessage ForwardedMessage::deserializationFromMessageJson(const json &msg) {
92 if (!(msg.contains("type") && msg["type"] == "ForwardMessage") && msg.contains("nodeList"))
93 throw std::invalid_argument("json is not a ForwardMessage");
94 ForwardedMessageDisplayStrategy display;
95 if (msg.contains("title") && msg.contains("brief") && msg.contains("source") && msg.contains("summary") && msg.contains("preview"))
96 display = ForwardedMessageDisplayStrategy::fromJson(msg);
97 else
98 display = ForwardedMessageDisplayStrategy::defaultStrategy();
99 std::vector<ForwardedNode> nodes;
100 auto &j = msg["nodeList"];
101 if (!j.is_array())
102 throw IllegalArgumentException("OnlineForwardedMessage参数异常, 应为 array", MIRAICP_EXCEPTION_WHERE);
103 try {
104 for (auto &&a: j) {
105 if (a["messageChain"][0].contains("kind") && a["messageChain"][0]["kind"] == "FORWARD") {
106 nodes.emplace_back(a["senderId"], a["senderName"], ForwardedMessage::deserializationFromMessageJson(a["messageChain"][1]["nodeList"]), a["time"]);
107 } else {
108 nodes.emplace_back(a["senderId"], a["senderName"], MessageChain::deserializationFromMessageJson(a["messageChain"]), a["time"]);
109 }
110 }
111
112 } catch (const json::parse_error &) {
113 throw APIException("OnlineForwardedMessage格式化异常", MIRAICP_EXCEPTION_WHERE);
114 }
115 return ForwardedMessage(std::move(nodes), display);
116 }
