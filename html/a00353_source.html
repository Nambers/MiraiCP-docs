263 void Event::incomingEvent(BaseEventData j, int type) {
264 switch (type) {
265 case eventTypes::GroupMessageEvent: {
266 Event::broadcast(GroupMessageEvent(std::move(j)));
267 break;
268 }
269 case eventTypes::FriendMessageEvent: {
270 Event::broadcast(PrivateMessageEvent(std::move(j)));
271 break;
272 }
273 case eventTypes::GroupTempMessageEvent: {
274 Event::broadcast(GroupTempMessageEvent(std::move(j)));
275 break;
276 }
277 // case IMessageEvent::MessageEventType::StrangerMessageEvent: {
278 // // todo Implement StrangerMessageEvent (ea)
279 // }
280 case eventTypes::GroupInviteEvent: {
281 //群聊邀请
282 Event::broadcast(GroupInviteEvent(std::move(j)));
283 break;
284 }
285 case eventTypes::NewFriendRequestEvent: {
286 //好友
287 Event::broadcast(NewFriendRequestEvent(std::move(j)));
288 break;
289 }
290 case eventTypes::MemberJoinEvent: { //新成员加入
291 Event::broadcast(MemberJoinEvent(std::move(j)));
292 break;
293 }
294 case eventTypes::MemberLeaveEvent: {
295 //群成员退出
296 Event::broadcast(MemberLeaveEvent(std::move(j)));
297 break;
298 }
299 case eventTypes::FriendRecallEvent: {
300 Event::broadcast(RecallEvent::FriendRecallEvent(std::move(j)));
301 break;
302 }
303 case eventTypes::MemberRecallEvent: {
304 Event::broadcast(RecallEvent::MemberRecallEvent(std::move(j)));
305 break;
306 }
307 case eventTypes::BotJoinGroupEvent: {
308 Event::broadcast(BotJoinGroupEvent(std::move(j)));
309 break;
310 }
311 case eventTypes::TimeOutEvent: {
312 Event::broadcast(TimeOutEvent(json_stringmover(j.eventData, "msg")));
313 break;
314 }
315 case eventTypes::BotOnlineEvent: {
316 Event::broadcast(BotOnlineEvent(j.botId));
317 break;
318 }
319 case eventTypes::NudgeEvent: {
320 Event::broadcast(NudgeEvent(std::move(j)));
321 break;
322 }
323 case eventTypes::BotLeaveEvent: {
324 Event::broadcast(BotLeaveEvent(std::move(j)));
325 break;
326 }
327 case eventTypes::MemberJoinRequestEvent: {
328 Event::broadcast(MemberJoinRequestEvent(std::move(j)));
329 break;
330 }
331 case eventTypes::MessagePreSendEvent: {
332 Event::broadcast(MessagePreSendEvent(std::move(j)));
333 break;
334 }
335 case eventTypes::Command: {
336 // command
337 if (CommandManager::commandList.find(j.eventData["primaryName"]) != CommandManager::commandList.end()) {
338 CommandManager::commandList[j.eventData["primaryName"]]->onCommand(
339 j.eventData.contains("contact") ? Contact::deserializeToPointer(json_jsonmover(j.eventData, "contact")) : nullptr,
340 Bot(j.botId),
341 MessageChain::deserializationFromMessageJson(json::parse(json_stringmover(j.eventData, "message"))));
342 }
343 break;
344 }
345 default: {
346 throw APIException("Unreachable code", MIRAICP_EXCEPTION_WHERE);
347 }
348 }
349 }

418 std::shared_ptr<Contact> BaseEventData::BuiltInContact::toContactPointer() {
419 switch (this->type) {
420 case ContactType::TypeFriend:
421 return std::make_shared<Friend>(this->id, this->botId);
422 case ContactType::TypeGroup:
423 return std::make_shared<Group>(this->id, this->botId);
424 case ContactType::TypeMember:
425 return std::make_shared<Member>(this->id, this->groupId, this->botId);
426 case ContactType::TypeBot:
427 // bot
428 return std::make_shared<Bot>(this->id);
429 case ContactType::TypeStranger:
430 // todo Implement Stranger (ea)
431 case ContactType::TypeAnonymousMember:
432 // todo anonymous member
433 default:
434 throw APIException("Type of builtInContact doesn't match or implement", MIRAICP_EXCEPTION_WHERE);
435 }
