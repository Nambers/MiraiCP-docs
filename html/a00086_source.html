24 using json = nlohmann::json;
25 void MessageSource::recall(JNIEnv *env) const {
26 json j;
27 j["source"] = this->serializeToString();
28 std::string re = Config::koperation(Config::Recall, j, env);
29 if (re == "Y") return;
30 if (re == "E1") MiraiCPThrow(BotException());
31 if (re == "E2") MiraiCPThrow(RecallException());
32 }

45 MessageSource MessageSource::deserializeFromString(const std::string &source) {
46 json j = json::parse(source);
47 try {
48 return {j["ids"].dump(), j["internalIds"].dump(), source};
49 } catch (json::type_error &e) {
50 Logger::logger.error("消息源序列化出错，格式不符合(MessageSource::deserializeFromString)");
51 Logger::logger.error(source);
52 Logger::logger.error(e.what());
53 MiraiCPThrow(IllegalArgumentException(std::string("消息源序列化出错，格式不符合(MessageSource::deserializeFromString), ") + e.what()));
54 }
55 }

57 MessageSource MessageSource::quoteAndSendMiraiCode(const std::string &content, QQID groupid, JNIEnv *env) const {
58 json obj;
59 json sign;
60 obj["messageSource"] = this->serializeToString();
61 obj["msg"] = content;
62 sign["MiraiCode"] = true;
63 sign["groupid"] = groupid;
64 obj["sign"] = sign.dump();
65 std::string re = Config::koperation(Config::SendWithQuote, obj, env);
66 return MessageSource::deserializeFromString(re);
67 }
