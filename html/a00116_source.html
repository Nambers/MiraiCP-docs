43 MessageChain MessageChain::deserializationFromMiraiCode(const std::string &m) {
44 size_t pos = 0;
45 size_t lastPos = -1;
46 MessageChain mc;
47 if (m.length() <= 7) {
48 return MessageChain(PlainText(m));
49 }
50 do {
51 if (m.length() - 7 - pos > 0 && m.substr(pos, 7) == "[mirai:") {
52 if (pos - lastPos > 1)
53 mc.add(PlainText(m.substr(lastPos + 1, pos - lastPos - 1))); // plain text
54 size_t back = MessageChain::findEnd(m, pos);
55 if (back == -1) throw IllegalStateException("", MIRAICP_EXCEPTION_WHERE);
56 std::string tmp = m.substr(pos, back - pos);
57 tmp = Tools::replace(tmp, "[mirai:", "");
58 size_t i = tmp.find(':'); // first :
59 int t = SingleMessage::getKey(tmp.substr(0, i));
60 switch (t) {
61 case 0:
62 // no miraiCode key is PlainText
63 Logger::logger.error("无法预料的错误, 信息: " + m);
64 break;
65 case 1:
66 mc.add(At(std::stoll(tmp.substr(i + 1, tmp.length() - i - 1))));
67 break;
68 case 2:
69 mc.add(AtAll());
70 break;
71 case 3:
72 mc.add(Image(tmp.substr(i + 1, tmp.length() - i - 1)));
73 break;
74 case 4:
75 mc.add(LightApp(tmp.substr(i + 1, tmp.length() - i - 1)));
76 break;
77 case 5: {
78 size_t comma = tmp.find(',');
79 mc.add(ServiceMessage(std::stoi(tmp.substr(i + 1, comma - i - 1)),
80 tmp.substr(comma + 1, tmp.length() - comma - 1)));
81 break;
82 }
83 case 6: {
84 //[mirai:file:/b53231e8-46dd-11ec-8ba5-5452007bd6c0,102,run.bat,55]
85 size_t comma1 = tmp.find(',');
86 size_t comma2 = tmp.find(',', comma1 + 1);
87 size_t comma3 = tmp.find(',', comma2 + 1);
88 mc.add(RemoteFile(tmp.substr(i + 1, comma1 - i - 1), std::stoi(tmp.substr(comma1 + 1, comma2 - comma1 - 1)), tmp.substr(comma2 + 1, comma3 - comma2 - 1), std::stoll(tmp.substr(comma3 + 1, tmp.length() - comma3 - 1))));
89 break;
90 }
91 case 7:
92 mc.add(Face(std::stoi(tmp.substr(i + 1, tmp.length() - i - 1))));
93 break;
94 case 8:
95 mc.add(FlashImage(tmp.substr(i + 1, tmp.length() - i - 1)));
96 break;
97 case 9: {
98 //[mirai:musicshare:name,title,summary,jUrl,pUrl,mUrl,brief]
99 auto temp = Tools::split(tmp.substr(i + 1, tmp.length() - i - 1), ",");
100 mc.add(MusicShare(temp[0], temp[1], temp[2], temp[3], temp[4], temp[5], temp[6]));
101 break;
102 }
103 default:
104 Logger::logger.error(
105 "MiraiCP碰到了意料之中的错误(原因:部分SimpleMessage在MiraiCode解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MiraiCode:" +
106 m);
107 mc.add(UnSupportMessage("[mirai:" + tmp));
108 break;
109 }
110 pos = back;
111 lastPos = pos;
112 if (t == 1)
113 lastPos++;
114 }
115 pos++;
116 } while (pos < m.length());
117 if (lastPos + 1 < m.length())
118 mc.add(PlainText(m.substr(lastPos + 1, m.length() - lastPos - 1))); // plain text
119 return mc;
120 }

122 MessageChain MessageChain::deserializationFromMessageSourceJson(const json &tmp, bool origin) {
123 json j = tmp;
124 if (origin)
125 j = tmp["originalMessage"];
126 MessageChain mc;
127 if (j.empty()) return mc;
128 if (j[0]["type"] == "MessageOrigin") {
129 if (j[0]["kind"] == "MUSIC_SHARE") {
130 mc.add(MusicShare(j[1]["kind"], j[1]["title"], j[1]["summary"], j[1]["jumpUrl"], j[1]["pictureUrl"], j[1]["musicUrl"], j[1]["brief"]));
131 return mc;
132 }
133 mc.add(OnlineForwardedMessage::deserializationFromMessageSourceJson(j));
134 return mc;
135 }
136 for (auto node: j) {
137 if (node["type"] == "SimpleServiceMessage") {
138 mc.add(ServiceMessage(node["serviceId"], node["content"]));
139 continue;
140 }
141 if (node["type"] == "LightApp") {
142 mc.add(LightApp(node["content"]));
143 continue;
144 }
145 if (node["type"] == "OnlineAudio") {
146 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
147 node["urlForDownload"]));
148 continue;
149 }
150 if (node["type"] == "FileMessage") {
151 mc.add(Group(tmp["targetId"].get<QQID>(), tmp["botId"].get<QQID>()).getFileById(node["id"]).plus(node["internalId"]));
152 continue;
153 }
154 if (node["type"] == "MarketFace") {
155 mc.add(MarketFace(node["delegate"]["faceId"]));
156 break;
157 }
158 switch (SingleMessage::getKey(node["type"])) {
159 case -2:
160 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
161 break;
162 case 0:
163 mc.add(PlainText(node["content"].get<std::string>()));
164 break;
165 case 1:
166 mc.add(At(node["target"]));
167 break;
168 case 2:
169 mc.add(AtAll());
170 break;
171 case 3:
172 mc.add(Image(node["imageId"]));
173 break;
174 case 7:
175 mc.add(Face(node["id"]));
176 break;
177 case 8:
178 mc.add(FlashImage(node["imageId"]));
179 break;
180 default:
181 Logger::logger.warning(
182 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在MessageSource解析支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段信息使MiraiCP可以支持这种消息: MessageSource:" +
183 j.dump());
184 mc.add(UnSupportMessage(node.dump()));
185 }
186 }
187 return mc;
188 }
