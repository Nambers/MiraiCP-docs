58 template<class T>
59 T getVal() const {
60 // for dev: 不用 get 为了不和shared_ptr重叠
61 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
62 if (T::type() != getType())
63 throw IllegalArgumentException(
64 "cannot convert from " + SingleMessage::getTypeString(getType()) + " to " +
65 SingleMessage::getTypeString(T::type()),
66 MIRAICP_EXCEPTION_WHERE);
67 T *re = static_cast<T *>(std::shared_ptr<SingleMessage>::get());
68 if (re == nullptr)
69 throw IllegalArgumentException(
70 "cannot convert from " + SingleMessage::getTypeString(getType()) + " to " +
71 SingleMessage::getTypeString(T::type()),
72 MIRAICP_EXCEPTION_WHERE);
73 return *re;
74 }

258 static size_t findEnd(const std::string &s, size_t start) {
259 size_t pos = start;
260 while (pos < s.length()) {
261 switch (s[pos]) {
262 case '\\':
263 pos += 2;
264 continue;
265 case ']':
266 return pos;
267 }
268 pos++;
269 }
270 return -1;
271 }
