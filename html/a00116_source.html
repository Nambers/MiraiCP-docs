25 std::string jstring2str(jstring jStr, JNIEnv *env) {
26 if (env == nullptr) env = ThreadManager::getEnv();
27 if (!jStr) {
28 Logger::logger.error("警告:kotlin部分返回空字符串, 位置:Tools::jstring2str");
29 return "";
30 }
31 std::u16string s = reinterpret_cast<const char16_t *>(env->GetStringChars(jStr, nullptr));
32 if (s.length() == 0) {
33 Logger::logger.error("警告:kotlin部分返回空字符串, 位置:Tools::jstring2str");
34 return "";
35 }
36 std::string x;
37 utf8::utf16to8(s.begin(), s.end(), std::back_inserter(x));
38 return x;
39 }

41 jstring str2jstring(const char *stra, JNIEnv *env) {
42 if (env == nullptr) env = ThreadManager::getEnv();
43 if (!stra) {
44 Logger::logger.error("警告:C++部分传入空字符串，位置:Tools::str2jstring");
45 }
46 std::string str(stra);
47 std::vector<unsigned short> utf16line;
48 utf8::utf8to16(str.begin(), str.end(), std::back_inserter(utf16line));
49 auto *c = new jchar[utf16line.size()];
50 for (int i = 0; i < utf16line.size(); i++) {
51 c[i] = utf16line[i];
52 }
53 return env->NewString((jchar *) c, (jsize) utf16line.size());
54 }
55 std::string replace(std::string str, std::string_view from, std::string_view to) {
56 size_t start_pos = 0;
57 while ((start_pos = str.find(from, start_pos)) != std::string::npos) {
58 str.replace(start_pos, from.length(), to);
59 start_pos += to.length(); // Handles case where 'to' is a substd::string of 'from'
60 }
61 return str;
62 }
63 std::vector<unsigned long long> StringToVector(std::string temp) {
64 std::vector<unsigned long long> result;
65 temp.erase(temp.begin());
66 temp.pop_back();
67 std::regex ws_re("[,]+");
68 std::vector<std::string> v(std::sregex_token_iterator(temp.begin(), temp.end(), ws_re, -1),
69 std::sregex_token_iterator());
70 result.reserve(v.size());
71 for (auto &&s: v)
72 result.push_back(std::stoull(s));
73 return result;
74 }
