61 MessageChain GroupMessageEvent::nextMessage(long time, bool halt) const {
62 json j{{"contact", this->group.toJson()},
63 {"time", time},
64 {"halt", halt}};
65 std::string r = KtOperation::ktOperation(KtOperation::NextMsg, j);
66 if (r == "E1")
67 throw TimeOutException("取下一条信息超时", MIRAICP_EXCEPTION_WHERE);
68 json re = json::parse(r);
69 return MessageChain::deserializationFromMessageJson(nlohmann::json::parse(json_stringmover(re, "message"))).plus(MessageSource::deserializeFromString(re["messageSource"]));

264 void Event::incomingEvent(BaseEventData j, int type) {
265 switch (type) {
266 case eventTypes::GroupMessageEvent: {
267 Event::broadcast(GroupMessageEvent(std::move(j)));
268 break;
269 }
270 case eventTypes::FriendMessageEvent: {
271 Event::broadcast(PrivateMessageEvent(std::move(j)));
272 break;
273 }
274 case eventTypes::GroupTempMessageEvent: {
275 Event::broadcast(GroupTempMessageEvent(std::move(j)));
276 break;
277 }
278 // case IMessageEvent::MessageEventType::StrangerMessageEvent: {
279 // // todo Implement StrangerMessageEvent (ea)
280 // }
281 case eventTypes::GroupInviteEvent: {
282 //群聊邀请
283 Event::broadcast(GroupInviteEvent(std::move(j)));
284 break;
285 }
286 case eventTypes::NewFriendRequestEvent: {
287 //好友
288 Event::broadcast(NewFriendRequestEvent(std::move(j)));
289 break;
290 }
291 case eventTypes::MemberJoinEvent: { //新成员加入
292 Event::broadcast(MemberJoinEvent(std::move(j)));
293 break;
294 }
295 case eventTypes::MemberLeaveEvent: {
296 //群成员退出
297 Event::broadcast(MemberLeaveEvent(std::move(j)));
298 break;
299 }
300 case eventTypes::FriendRecallEvent: {
301 Event::broadcast(RecallEvent::FriendRecallEvent(std::move(j)));
302 break;
303 }
304 case eventTypes::MemberRecallEvent: {
305 Event::broadcast(RecallEvent::MemberRecallEvent(std::move(j)));
306 break;
307 }
308 case eventTypes::BotJoinGroupEvent: {
309 Event::broadcast(BotJoinGroupEvent(std::move(j)));
310 break;
311 }
312 case eventTypes::TimeOutEvent: {
313 Event::broadcast(TimeOutEvent(json_stringmover(j.eventData, "msg")));
314 break;
315 }
316 case eventTypes::BotOnlineEvent: {
317 Event::broadcast(BotOnlineEvent(j.botId));
318 break;
319 }
320 case eventTypes::NudgeEvent: {
321 Event::broadcast(NudgeEvent(std::move(j)));
322 break;
323 }
324 case eventTypes::BotLeaveEvent: {
325 Event::broadcast(BotLeaveEvent(std::move(j)));
326 break;
327 }
328 case eventTypes::MemberJoinRequestEvent: {
329 Event::broadcast(MemberJoinRequestEvent(std::move(j)));
330 break;
331 }
332 case eventTypes::MessagePreSendEvent: {
333 Event::broadcast(MessagePreSendEvent(std::move(j)));
334 break;
335 }
336 case eventTypes::Command: {
337 // command
338 CommandManager::commandList[j.eventData["bindId"]]->onCommand(
339 j.eventData.contains("contact") ? Contact::deserializeToPointer(json_jsonmover(j.eventData, "contact")) : nullptr,
340 Bot(j.botId),
341 MessageChain::deserializationFromMessageJson(json::parse(json_stringmover(j.eventData, "message"))));
342 break;
343 }
344 default: {
345 throw APIException("Unreachable code", MIRAICP_EXCEPTION_WHERE);
346 }
347 }

384 std::shared_ptr<Contact> BaseEventData::BuiltInContact::toContactPointer() {
385 switch (this->type) {
386 case ContactType::TypeFriend:
387 return std::make_shared<Friend>(this->id, this->botId);
388 case ContactType::TypeGroup:
389 return std::make_shared<Group>(this->id, this->botId);
390 case ContactType::TypeMember:
391 return std::make_shared<Member>(this->id, this->groupId, this->botId);
392 case ContactType::TypeBot:
393 // bot
394 return std::make_shared<Bot>(this->id);
395 case ContactType::TypeStranger:
396 // todo Implement Stranger (ea)
397 case ContactType::TypeAnonymousMember:
398 // todo anonymous member
399 default:
400 throw APIException("Type of builtInContact doesn't match or implement", MIRAICP_EXCEPTION_WHERE);
401 }
