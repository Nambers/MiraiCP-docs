72 template<class T>
73 bool registerCommand(T command) {
74 static_assert(std::is_base_of_v<IRawCommand, T>, "只支持IRawCommand的派生类");
75 nlohmann::json j;
76 j["pluginId"] = CPPPlugin::config.id;
77 j["usage"] = command.config().usage;
78 j["primaryName"] = command.config().primaryName;
79 j["secondName"] = command.config().secondNames;
80 j["description"] = command.config().description;
81 j["override"] = command.config().overrideOrigin;
82 j["preFixOption"] = command.config().preFixOption;
83 size_t before = commandList.size();
84 std::shared_ptr<IRawCommand> c;
85 c.reset(new T(command));
86 commandList.push_back(c);
87 size_t now = commandList.size();
88 if (now - before == 1)
89 j["bindId"] = now - 1;
90 else {
91 auto i = std::find(commandList.begin(), commandList.end(), c);
92 if (i != commandList.end())
93 j["bindId"] = i - commandList.begin();
94 else
95 throw IllegalArgumentException("找不到合适的bindId", MIRAICP_EXCEPTION_WHERE);
96 }
97 nlohmann::json rej;
98 rej["command"] = j.dump();
99 std::string re = KtOperation::ktOperation(KtOperation::CommandReg, rej);
100 return re == "true";
101 }
102 static CommandManager commandManager;
103 };
