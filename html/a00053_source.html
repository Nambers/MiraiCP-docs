318 struct ContactDataHelper {

330 DataType *GetDataInternal() const {
331 auto clz_ptr = static_cast<const ClassType *>(this);
332 assert(clz_ptr->InternalData != nullptr);
333 return static_cast<DataType *>(clz_ptr->InternalData.get());
334 }
335
336 public:
338 std::string nickOrNameCard() const {
339 auto dataPtr = GetDataInternal();
340 dataPtr->requestRefresh();

343 };

346 std::string avatarUrl() const {
347 auto dataPtr = GetDataInternal();
