215 void Event::incomingEvent(BaseEventData j, int type) {
216 switch (type) {
217 case eventTypes::GroupMessageEvent: {
218 Event::broadcast(GroupMessageEvent(std::move(j)));
219 break;
220 }
221 case eventTypes::FriendMessageEvent: {
222 Event::broadcast(PrivateMessageEvent(std::move(j)));
223 break;
224 }
225 case eventTypes::GroupTempMessageEvent: {
226 Event::broadcast(GroupTempMessageEvent(std::move(j)));
227 break;
228 }
229// case IMessageEvent::MessageEventType::StrangerMessageEvent: {
230// // todo Implement StrangerMessageEvent (ea)
231// }
232 case eventTypes::GroupInviteEvent: {
233 //群聊邀请
234 Event::broadcast(GroupInviteEvent(std::move(j)));
235 break;
236 }
237 case eventTypes::NewFriendRequestEvent: {
238 //好友
239 Event::broadcast(NewFriendRequestEvent(std::move(j)));
240 break;
241 }
242 case eventTypes::MemberJoinEvent: { //新成员加入
243 Event::broadcast(MemberJoinEvent(std::move(j)));
244 break;
245 }
246 case eventTypes::MemberLeaveEvent: {
247 //群成员退出
248 Event::broadcast(MemberLeaveEvent(std::move(j)));
249 break;
250 }
251 case eventTypes::FriendRecallEvent: {
252 Event::broadcast(RecallEvent::FriendRecallEvent(std::move(j)));
253 break;
254 }
255 case eventTypes::MemberRecallEvent: {
256 Event::broadcast(RecallEvent::MemberRecallEvent(std::move(j)));
257 break;
258 }
259 case eventTypes::BotJoinGroupEvent: {
260 Event::broadcast(BotJoinGroupEvent(std::move(j)));
261 break;
262 }
263 case eventTypes::TimeOutEvent: {
264 Event::broadcast(TimeOutEvent(Tools::json_stringmover(j.eventData, "msg")));
265 break;
266 }
267 case eventTypes::BotOnlineEvent: {
268 Event::broadcast(BotOnlineEvent(j.botId));
269 break;
270 }
271 case eventTypes::NudgeEvent: {
272 Event::broadcast(NudgeEvent(std::move(j)));
273 break;
274 }
275 case eventTypes::BotLeaveEvent: {
276 Event::broadcast(BotLeaveEvent(std::move(j)));
277 break;
278 }
279 case eventTypes::MemberJoinRequestEvent: {
280 Event::broadcast(MemberJoinRequestEvent(std::move(j)));
281 break;
282 }
283 case eventTypes::MessagePreSendEvent: {
284 Event::broadcast(MessagePreSendEvent(std::move(j)));
285 break;
286 }
287 case eventTypes::Command: {
288 // command
289 CommandManager::commandManager[j.eventData["bindId"]]->onCommand(
290 j.eventData.contains("contact") ? Contact::deserializeToPointer(Tools::json_jsonmover(j.eventData, "contact")) : nullptr,
291 Bot(j.botId),
292 MessageChain::deserializationFromMessageJson(
293 j.eventData.contains("message") ? Tools::json_jsonmover(j.eventData, "message") : ""));
294 break;
295 }
296 default: {
297 throw APIException("Unreachable code", MIRAICP_EXCEPTION_WHERE);
298 }
299 }

327 std::shared_ptr<Contact> BaseEventData::BuiltInContact::toContactPointer() {
328 switch (this->type) {
329 case ContactType::TypeFriend:
330 return std::make_shared<Friend>(this->id, this->botId);
331 case ContactType::TypeGroup:
332 return std::make_shared<Group>(this->id, this->botId);
333 case ContactType::TypeMember:
334 return std::make_shared<Member>(this->id, this->groupId, this->botId);
335 case ContactType::TypeBot:
336 // bot
337 return std::make_shared<Bot>(this->id);
338 case ContactType::TypeStranger:
339 // todo Implement Stranger (ea)
340 case ContactType::TypeAnonymousMember:
341 // todo anonymous member
342 default:
343 throw APIException("Type of builtInContact doesn't match or implement", MIRAICP_EXCEPTION_WHERE);
344 }
