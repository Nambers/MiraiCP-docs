49 MiraiCPString pluginOperation(const MiraiCPString &s) {
50#ifdef LOADER_NATIVE
51 auto nCharPtr = s.copyToCharPtr();
52 MIRAICP_DEFER(delete[] nCharPtr;);
53 return {LoaderAPIs::oper(nCharPtr)};
54#else
55 auto env = JNIEnvManager::getEnv();
56 auto tmp = jstring2str((jstring) env->CallStaticObjectMethod(JNIEnvs::Class_cpplib,
57 JNIEnvs::koper,
58 str2jstring(s.copyToCharPtr())));
59 return {tmp};
60#endif
61 }

123 void loadNewPlugin(const MiraiCPString &path, bool activateNow) {
124 {
125 std::lock_guard lk(task_mtx);
126 if (activateNow)
127 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::LOAD_NEW_ACTIVATENOW, path));
128 else
129 loader_thread_task_queue.push(std::make_pair(LOADER_TASKS::LOAD_NEW_DONTACTIVATE, path));
