52 inline std::shared_ptr<IContactData> get_bot(QQID id) {
53 static std::unordered_map<QQID, std::shared_ptr<IContactData>> BotPool;
54 static std::mutex mtx;
55 std::lock_guard<std::mutex> lck(mtx);
56 auto &Ptr = BotPool.try_emplace(id).first->second;
57 if (!Ptr){
58 Ptr = std::make_shared<IContactData>();
59 Ptr->_id = id;
60 Ptr->_type = MIRAI_BOT;
61 Ptr->_botId = id;
62 Ptr->forceRefreshNextTime();
63 }
64 return Ptr;
65 }
