26 auto GetFriendPool(QQID id, QQID botid) noexcept {
27 static std::unordered_map<QQID, std::unordered_map<QQID, std::shared_ptr<Friend::DataType>>> Pool;
28 auto &val = Pool[botid][id];
29 if (!val) {
30 val = std::make_shared<Friend::DataType>();
31 val->_id = id;
32 val->_botid = botid;
33 val->_type = MIRAI_FRIEND;
34 }
35 return val;
36 }

38 auto GetFriendPool(const json &in_json) {

42 throw IllegalArgumentException("构造Friend时传入的json异常", MIRAICP_EXCEPTION_WHERE);
