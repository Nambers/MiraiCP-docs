584 class Event {
585 private: // typedefs
586 class eventNode {
587 public:
588 bool enable = true;
589 std::function<bool(MiraiCPEvent *)> func;
590
591 explicit eventNode(std::function<bool(MiraiCPEvent *)> f) : func(std::move(f)) {}
592
593 bool run(MiraiCPEvent *a) const {
594 return func(a);
595 }
596 };
597
599 class NodeHandle {
600 private:
601 bool *enable;
602
603 public:
604 explicit NodeHandle(bool *a) { this->enable = a; }
605 void stop() { *enable = false; }
606 void resume() { *enable = true; }
607 };
608
609 using priority_level = unsigned char;
610 using event_vector = std::vector<eventNode>;
611 using eventNodeTable = std::vector<std::map<priority_level, event_vector>>;
612
613 private: // member
614 eventNodeTable _all_events_;
615
616 private:
617 Event() : _all_events_(int(eventTypes::count)){};
618
619 public: // singleton mode
620 static Event processor;
621
622 private:
623 template<typename T>
624 int id() const {
625 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播继承MiraiCPEvent的事件");
626 return int(T::get_event_type());
627 }
628
629 public:
630 bool noRegistered(int index) {
631 return _all_events_[index].empty();
632 }
633
635 template<typename T>
636 void broadcast(T &&val) {
637 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持广播MiraiCPEvent的派生类");
638 MiraiCPEvent *p = &val;
639 for (auto &&[k, v]: _all_events_[id<T>()]) {
640 for (auto &&a: v) {
641 if (a.run(p)) return;
642 }
643 }
644 }
645
652 template<typename T>
653 NodeHandle registerEvent(std::function<void(T)> callback, priority_level level = 100) {
654 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持注册MiraiCPEvent的派生类事件");
655 std::function<bool(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
656 callback(*dynamic_cast<T *>(p));
657 return false;
658 };
659 eventNode t(tmp);
660 _all_events_[id<T>()][level].emplace_back(t);
661 return NodeHandle(&t.enable);
662 }
663
671 template<typename T>
672 NodeHandle registerBlockingEvent(std::function<bool(T)> callback, priority_level level = 100) {
673 static_assert(std::is_base_of_v<MiraiCPEvent, T>, "只支持注册MiraiCPEvent的派生类事件");
674 std::function<bool(MiraiCPEvent *)> tmp = [=](MiraiCPEvent *p) {
675 return callback(*dynamic_cast<T *>(p));
676 };
677 eventNode t(tmp);
678 _all_events_[id<T>()][level].emplace_back(t);
679 return NodeHandle(&t.enable);
680 }
681 };
