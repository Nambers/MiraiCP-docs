57 template<class T>
58 T get() const {
59 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
60 if (T::type() != this->type())
61 MiraiCPThrow(IllegalArgumentException("cannot convert from " + SingleMessage::messageType[this->type()] + " to " + SingleMessage::messageType[T::type()]));
62 T *re = static_cast<T *>(this->content.get());
63 if (re == nullptr) MiraiCPThrow(IllegalArgumentException("x"));
64 return *re;
65 }

137 static size_t findEnd(const std::string &s, size_t start) {
138 size_t pos = start;
139 while (pos < s.length()) {
140 switch (s[pos]) {
141 case '\\':
142 pos += 2;
143 continue;
144 case ']':
145 return pos;
146 }
147 pos++;
148 }
149 return -1;
150 }

252 return false;
253 for (size_t i = 0; i < this->content.size(); i++) {
254 if (this->content[i] != mc[i])

256 }
