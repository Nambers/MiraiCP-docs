34 void Member::refreshInfo(JNIEnv *env) {
35 if (isAnonymous)
36 return;
37 std::string temp = LowLevelAPI::getInfoSource(this->toString(), env);
38 if (temp == "E1")
39 throw MemberException(1, MIRAICP_EXCEPTION_WHERE);
40 if (temp == "E2")
41 throw MemberException(2, MIRAICP_EXCEPTION_WHERE);
42 LowLevelAPI::info tmp = LowLevelAPI::info0(temp);
43 this->_nickOrNameCard = tmp.nickornamecard;
44 this->_avatarUrl = tmp.avatarUrl;
45 this->permission = getPermission();
46 if (temp == "E1") {
47 throw MemberException(1, MIRAICP_EXCEPTION_WHERE);
48 }
49 if (temp == "E2") {
50 throw MemberException(2, MIRAICP_EXCEPTION_WHERE);
51 }
52 }
