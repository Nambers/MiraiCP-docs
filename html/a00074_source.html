56 template<class T>
57 T get() const {
58 static_assert(std::is_base_of_v<SingleMessage, T>, "只支持SingleMessage的派生类");
59 if (T::type() != this->type())
60 MiraiCPThrow(IllegalArgumentException("cannot convert from " + SingleMessage::messageType[this->type()] + " to " + SingleMessage::messageType[T::type()]));
61 T *re = static_cast<T *>(this->content.get());
62 if (re == nullptr) MiraiCPThrow(IllegalArgumentException("x"));

64 }

142 static size_t findEnd(const std::string &s, size_t start) {
143 size_t pos = start;
144 while (pos < s.length()) {
145 switch (s[pos]) {
146 case '\\':
147 pos += 2;
148 continue;
149 case ']':
150 return pos;
151 }
152 pos++;
153 }
154 return -1;
155 }
