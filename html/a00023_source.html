25 MessageChain PrivateMessageEvent::nextMessage(long time, bool halt, JNIEnv *env) {
26 json j;
27 j["contactSource"] = this->sender.serializationToString();
28 j["time"] = time;
29 j["halt"] = halt;
30 std::string r = Config::koperation(Config::NextMsg, j, env);
31 if (r == "-1")
32 MiraiCPThrow(TimeOutException("取下一条信息超时"));
33 json re = json::parse(r);
34 return MessageChain::deserializationFromMiraiCode(re["message"]).plus(MessageSource::deserializeFromString(re["messageSource"]));
35 }
