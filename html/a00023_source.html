24 MessageChain PrivateMessageEvent::nextMessage(long time, bool halt, JNIEnv *env) {
25 json j;
26 j["contactSource"] = this->sender.serializationToString();
27 j["time"] = time;
28 j["halt"] = halt;
29 std::string r = Config::koperation(Config::NextMsg, j, env);
30 if (r == "-1")
31 MiraiCPThrow(TimeOutException("取下一条信息超时"));
32 json re = json::parse(r);
33 return MessageChain::deserializationFromMiraiCode(re["message"]).plus(MessageSource::deserializeFromString(re["messageSource"]));
34 }
