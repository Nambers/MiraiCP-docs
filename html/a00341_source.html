29 auto GetFriendPool(QQID id, QQID botid) noexcept {
30 static std::unordered_map<QQID, std::unordered_map<QQID, std::shared_ptr<Friend::DataType>>> Pool;
31 auto &val = Pool[botid][id];
32 if (!val) {
33 val = std::make_shared<Friend::DataType>();
34 val->_id = id;
35 val->_botId = botid;
36 val->_type = MIRAI_FRIEND;
37 }
38 return val;
39 }

54 Friend::Friend(nlohmann::json in_json) : Contact(GetFriendPool(in_json)) {
55 auto ActualDataPtr = GetDataInternal();

58 bool needRefresh = false;

61 else
62 needRefresh = true;
63 if (in_json.contains("avatarUrl")) ActualDataPtr->_avatarUrl = json_stringmover(in_json, "avatarUrl");
64 else
65 needRefresh = true;
66
67 if (needRefresh) forceRefreshNextTime();
