41 class ThreadPoolBase {
42 protected:
43 template<typename T1, typename T2, typename T = std::common_type_t<T1, T2>>
44 class [[nodiscard]] blocks {
45 size_t block_size = 0;
46 size_t num_blocks = 0;
47 size_t total_size = 0;
48 T first_index = 0;
49 T index_after_last = 0;
50
51 public:
52 blocks(const T1 first_index_, const T2 index_after_last_, const size_t num_blocks_)
53 : num_blocks(num_blocks_),
54 first_index(static_cast<T>(first_index_)),
55 index_after_last(static_cast<T>(index_after_last_)) {
56 if (index_after_last < first_index)
57 std::swap(index_after_last, first_index);
58 total_size = static_cast<size_t>(index_after_last - first_index);
59 block_size = static_cast<size_t>(total_size / num_blocks);
60 if (block_size == 0) {
61 block_size = 1;
62 num_blocks = (total_size > 1) ? total_size : 1;
63 }
64 }
65
66 [[nodiscard]] T start(const size_t i) const {
67 return static_cast<T>(i * block_size) + first_index;
68 }
69
70 [[nodiscard]] T end(const size_t i) const {
71 return (i == num_blocks - 1) ? index_after_last : (static_cast<T>((i + 1) * block_size) + first_index);
72 }
73
74 [[nodiscard]] size_t get_num_blocks() const {
75 return num_blocks;
76 }
77
78 [[nodiscard]] size_t get_total_size() const {
79 return total_size;
80 }
81 };

131 class ThreadPool : public ThreadPoolBase {
132 protected:
133 std::vector<std::thread> threads;
134
135 public:
136 ThreadPool(concurrency_t thread_count_ = 0) // NOLINT(google-explicit-constructor)
137 : ThreadPoolBase(),
138 threads(determine_thread_count(thread_count_)) {
139 create_threads();
140 }
141
142 ~ThreadPool() {
143 wait_for_tasks();
144 destroy_threads();
145 }

147 template<typename T>
148 static T *Create() {
149 auto ptr = (T *) Traits::malloc(sizeof(T));
150 new (ptr) T;
151 return ptr;
152 }
153
154 template<typename T>
155 static void Destroy(T *promise) {
156 promise->~T();
157 Traits::free(promise);
158 }
159
160 template<typename F, typename... A>
161 void push_task(F &&task, A &&...args) {
162 std::function<void()> task_function = std::bind(std::forward<F>(task), std::forward<A>(args)...);
163 {
164 std::unique_lock tasks_lock(tasks_mtx);
165 tasks.push(std::move(task_function));
166 }
167 tasks_total.fetch_add(1, std::memory_order_acq_rel);
168 task_available_cv.notify_one();
169 }
170
171 template<typename F, typename... A, typename R = std::invoke_result_t<std::decay_t<F>, std::decay_t<A>...>>
172 [[nodiscard]] std::future<R> submit(F &&task, A &&...args) {
173 auto task_promise = Create<std::promise<R>>();
174 auto future = task_promise->get_future();
175 push_task(
176 [task_promise](auto &&taskInner, auto &&...argss) {
177 try {
178 if constexpr (std::is_void_v<R>) {
179 std::invoke(std::forward<decltype(taskInner)>(taskInner),
180 std::forward<decltype(argss)>(argss)...);
181 task_promise->set_value();
182 } else {
183 task_promise->set_value(std::invoke(std::forward<decltype(taskInner)>(taskInner),
184 std::forward<decltype(argss)>(argss)...));
185 }
186 } catch (...) {
187 try {
188 task_promise->set_exception(std::current_exception());
189 } catch (...) {
190 }
191 }
192 Destroy(task_promise);
193 },
194 std::forward<F>(task), std::forward<A>(args)...);
195 return future;
196 }
197
198 size_t get_thread_count() {
199 return threads.size();
200 }
201
202 void reset(concurrency_t thread_count_ = 0) {
203 bool was_paused = paused.exchange(true);
204 wait_for_tasks();
205 destroy_threads();
206 auto thread_count = determine_thread_count(thread_count_);
207 threads.resize(thread_count);
208 paused.exchange(was_paused);
209 create_threads();
210 }

216 void resetThreadByIndex(size_t index) {
217 static std::mutex resetThreadMtx;
218 std::lock_guard lk(resetThreadMtx);
219 size_t oldThCounter = threadIndexCounter;
220 threadIndexCounter = index;
221 LibLoader::logger.warning("resetting thread in pool at index " + std::to_string(index) + " due to a user caused error");
222 new (&threads[index]) std::thread(&ThreadPool::worker, this);
223 while (threadIndexCounter != index + 1) {
224 std::this_thread::yield();
225 }
226 threadIndexCounter = oldThCounter;
227 }
228
229 private:
230 void create_threads() {
231 running = true;
232 for (auto &th: threads) {
233 th = std::thread(&ThreadPool::worker, this);
234 }
235 }
236
237 void destroy_threads() {
238 running = false;
239 task_available_cv.notify_all();
240 for (auto &thread: threads) {
241 thread.join();
242 }
243 }
244 };
