33 std::shared_ptr<Contact> Contact::deserializeToPointer(nlohmann::json j) {
34 uint8_t thistype = j["type"];
35 switch (thistype) {
36 case MIRAI_FRIEND: {
37 return std::make_shared<Friend>(std::move(j));
38 }
39 case MIRAI_MEMBER: {
40 return std::make_shared<Member>(std::move(j));
41 }
42 case MIRAI_GROUP: {
43 return std::make_shared<Group>(std::move(j));
44 }
45 default:
46 assert(0);
47 break;
48 }
49
50 return {};
51 }

53 MessageSource Contact::sendVoiceImpl(std::string path) const {
54 json j{{"path", std::move(path)},
55 {"contact", toJson()}};
56 std::string re = KtOperation::ktOperation(KtOperation::Voice, j);
57 if (re == "E1") {
58 throw UploadException("上传语音文件格式不对(必须为.amr/.silk)或文件不存在", MIRAICP_EXCEPTION_WHERE);
59 } else if (re == "E2") {
60 throw UploadException("上传语音文件大小超过服务器限制，一般限制在1MB上下", MIRAICP_EXCEPTION_WHERE);
61 }
62 return MessageSource::deserializeFromString(re);
63 }
