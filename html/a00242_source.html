32 std::shared_ptr<Contact> Contact::deserializeToPointer(nlohmann::json j) {
33 uint8_t thistype = j["type"];
34 switch (thistype) {
35 case MIRAI_FRIEND: {
36 return std::make_shared<Friend>(std::move(j));
37 }
38 case MIRAI_MEMBER: {
39 return std::make_shared<Member>(std::move(j));
40 }
41 case MIRAI_GROUP: {
42 return std::make_shared<Group>(std::move(j));
43 }
44 default:
45 assert(0);
46 break;
47 }
48
49 return {};
50 }

52 MessageSource Contact::sendVoiceImpl(std::string path) const {
53 json j{{"path", std::move(path)},
54 {"contact", toJson()}};
55 std::string re = KtOperation::ktOperation(KtOperation::Voice, j);
56 if (re == "E1") {
57 throw UploadException("上传语音文件格式不对(必须为.amr/.silk)或文件不存在", MIRAICP_EXCEPTION_WHERE);
58 } else if (re == "E2") {
59 throw UploadException("上传语音文件大小超过服务器限制，一般限制在1MB上下", MIRAICP_EXCEPTION_WHERE);
60 }
61 return MessageSource::deserializeFromString(re);
62 }
