108 void LoaderMain::mainloop() noexcept {
109 std::unique_lock cvuniquelock(loaderCVLock());
110 loaderWakeCV().wait_for(
111 cvuniquelock,
112 std::chrono::milliseconds(100),
113 []() { return is_loader_exited() || shouldTick() || !loader_thread_task_queue.empty(); });
114 cvuniquelock.unlock();
115
116 tick();
117
118 if (!loader_thread_task_queue.empty()) {
119 loadertask task;
120 {
121 std::lock_guard lk(task_mtx);
122 std::swap(task, loader_thread_task_queue.front());
123 loader_thread_task_queue.pop();
124 }
125 try {
126 switch (task.first) {
127 case LOADER_TASKS::ADD_THREAD:
128 loader_enablePluginById(task.second);
129 break;
130 case LOADER_TASKS::END_THREAD:
131 loader_disablePluginById(task.second);
132 break;
133 case LOADER_TASKS::ENABLE_ALL:
134 loader_enableAllPlugins();
135 break;
136 case LOADER_TASKS::DISABLE_ALL:
137 loader_disableAllPlugins();
138 break;
139 case LOADER_TASKS::LOAD_NEW_ACTIVATENOW:
140 loader_loadNewPlugin(task.second, true);
141 break;
142 case LOADER_TASKS::LOAD_NEW_DONTACTIVATE:
143 loader_loadNewPlugin(task.second, false);
144 break;
145 case LOADER_TASKS::UNLOAD:
146 loader_unloadPluginById(task.second);
147 break;
148 case LOADER_TASKS::RELOAD:
149 loader_reloadPluginById(task.second);
150 break;
151 case LOADER_TASKS::EXCEPTION_PLUGINEND:
152 loader_pluginEndByException(task.second);
153 break;
154 case LOADER_TASKS::RESET_THREAD:
155 loader_resetThreadByIndex(task.second);
156 break;
157 default:
158 throw std::exception();
159 }
160 } catch (LoaderBaseException &e) {
161 e.raise();
162 } catch (std::exception &e) {
163 logger.error("libLoader线程遇到无法预料的错误：" + std::string(e.what()) + "，程序将继续执行，如果发生意料之外的状况，请提交issue");
164 }
165 }
166 }
