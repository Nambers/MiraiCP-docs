26 class MiraiCPException : public std::exception {
27 protected:
29 const std::string exceptionType;
30 std::string re;
31
32 public:
33 int lineNum = 0;
34 std::string filename;
35
36 //构造时传入类型字符串
37 explicit MiraiCPException(const std::string &&type, const std::string &description = "") : exceptionType(type) {
38 if (description.empty())
39 this->re = type + ":MiraiCP异常";
40 else
41 this->re = type + ":" + description;
42 }

45 class ExceptionBroadcasting {
46 public:
47 MiraiCPException *e;
48 explicit ExceptionBroadcasting(MiraiCPException *ex) : e(ex) {}
49 ~ExceptionBroadcasting();
50 };
51
53 MiraiCPException append(const std::string &name, int line) {
54 lineNum = line;
55 filename = name;
56 ExceptionBroadcasting(this);
57 return *this;
58 }
59
61 std::string getExceptionType() const { return exceptionType; }
62
64 const char *what() const noexcept override { return re.c_str(); }
65
67 virtual void basicRaise() const {
68 Logger::logger.error(this->what());
69 };
70
72 void raise() const {
73 this->basicRaise();
74 if (!filename.empty() && lineNum != 0)
75 Logger::logger.error("文件名:" + filename + "\n行号:" + std::to_string(lineNum));
76 }
77 };

270 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
271 if (re == "EF")
272 throw FriendException().append(name, line);
273 if (re == "EG")
274 throw GroupException().append(name, line);
275 if (re == "EM")
276 throw MemberException(1).append(name, line);
277 if (re == "EMM")
278 throw MemberException(2).append(name, line);
279 if (re == "EB")
280 throw BotException("找不到bot:" + re).append(name, line);
281 if (re == "EA")
282 throw APIException(ErrorMsg).append(name, line);
283 }
