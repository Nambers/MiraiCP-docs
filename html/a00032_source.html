26 class MiraiCPException : public std::exception {
27 protected:
29 const std::string exceptionType;
30 std::string re;
31
32 public:
33 int lineNum = 0;
34 std::string filename;
35
36 //构造时传入类型字符串
37 explicit MiraiCPException(const std::string &&type, const std::string &description = "") : exceptionType(type) {
38 if (description.empty())
39 this->re = type + ":MiraiCP异常";
40 else
41 this->re = type + ":" + description;
42 }

45 class ExceptionBroadcasting {
46 public:
47 MiraiCPException *e;
48 explicit ExceptionBroadcasting(MiraiCPException *ex) : e(ex) {}
49 ~ExceptionBroadcasting();
50 };
51
52 MiraiCPException append(const std::string &name, int line) {
53 lineNum = line;
54 filename = name;
55 ExceptionBroadcasting(this);
56 return *this;
57 }
58
60 std::string getExceptionType() const { return exceptionType; }
61
63 const char *what() const noexcept override { return re.c_str(); }
64
66 virtual void basicRaise() const {
67 Logger::logger.error(this->what());
68 };
69
71 void raise() const {
72 this->basicRaise();
73 if (!filename.empty() && lineNum != 0)
74 Logger::logger.error("文件名:" + filename + "\n行号:" + std::to_string(lineNum));
75 }
76 };

269 inline void ErrorHandle0(const std::string &name, int line, const std::string &re, const std::string &ErrorMsg = "") {
270 if (re == "EF")
271 throw FriendException().append(name, line);
272 if (re == "EG")
273 throw GroupException().append(name, line);
274 if (re == "EM")
275 throw MemberException(1).append(name, line);
276 if (re == "EMM")
277 throw MemberException(2).append(name, line);
278 if (re == "EB")
279 throw BotException("找不到bot:" + re).append(name, line);
280 if (re == "EA")
281 throw APIException(ErrorMsg).append(name, line);
282 }
