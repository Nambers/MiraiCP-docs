47 std::string libError() {
48#if MIRAICP_WINDOWS
49 // https://stackoverflow.com/questions/1387064/how-to-get-the-error-message-from-the-error-code-returned-by-getlasterror
50 //Returns the last Win32 error, in string format. Returns an empty string if there is no error.
51
52 //Get the error message ID, if any.
53 DWORD errorMessageID = ::GetLastError();
54 if (errorMessageID == 0) {
55 return {}; //No error message has been recorded
56 }
57
58 LPWSTR messageBuffer = nullptr;
59
60 //Ask Win32 to give us the string version of that message ID.
61 //The parameters we pass in, tell Win32 to create the buffer that holds the message for us (because we don't yet know how long the message string will be).
62 size_t size = FormatMessageW(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
63 nullptr, errorMessageID, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), (LPWSTR) &messageBuffer, 0, nullptr);
64
65 //Copy the error message into a std::string.
66 std::wstring message(messageBuffer, size);
67
68 //Free the Win32's string's buffer.
69 LocalFree(messageBuffer);
70
71 // Translate to utf8 encoded std::string
72 std::string ans;
73 utf8::utf16to8(message.begin(), message.end(), std::back_inserter(ans));
74
75 return ans;
76#else
77 return dlerror();
78#endif
79 }
