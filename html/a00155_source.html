30 void ThreadManager::setEnv(JNIEnv *e) {
31 mtx.lock();
32 if (!ThreadManager::included(ThreadManager::getThreadId())) {
33 ThreadInfo tmp{e, false};
34 ThreadManager::threads.insert(std::pair<std::string, ThreadInfo>(ThreadManager::getThreadId(), tmp));
35 } else {
36 ThreadManager::threads[ThreadManager::getThreadId()].e = e;
37 }
38 mtx.unlock();
39 }

41 void ThreadManager::newEnv(const char *threadName) {
42 JNIEnv *env = nullptr;
43 JavaVMAttachArgs args{JNIVersion,
44 const_cast<char *>(threadName),
45 nullptr};
46 gvm->AttachCurrentThread((void **) &env, &args);
47 ThreadInfo tmp{env, true};
48 ThreadManager::threads.insert(std::pair<std::string, ThreadInfo>(ThreadManager::getThreadId(), tmp));
49 Logger::logger.info("refresh env");
50 };

52 void ThreadManager::detach() {
53 mtx.lock();
54 if (ThreadManager::included(ThreadManager::getThreadId())) {
55 bool att = ThreadManager::threads[ThreadManager::getThreadId()].attach;
56 ThreadManager::threads.erase(ThreadManager::getThreadId());
57 if (att)
58 gvm->DetachCurrentThread();
59 }
60 mtx.unlock();
61 }

63 bool ThreadManager::included(const std::string &id) {
64 if (ThreadManager::threads.empty() || ThreadManager::threads.count(id) == 0)
65 return false;
66 return true;
67 }

69 JNIEnv *ThreadManager::getEnv() {
70 mtx.lock();
71 if (!ThreadManager::included(getThreadId())) {
72 ThreadManager::newEnv();
73 }
74 JNIEnv *tmp = ThreadManager::threads[ThreadManager::getThreadId()].e;
75 mtx.unlock();
76 return tmp;
77 }
