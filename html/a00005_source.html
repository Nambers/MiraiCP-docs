56 void ThreadManager::setEnv(JNIEnv *e) {
57 mtx.lock();
58 if (!ThreadManager::included(ThreadManager::getThreadId())) {
59 ThreadInfo tmp{e, false};
60 ThreadManager::threads.insert(std::pair<std::string, ThreadInfo>(ThreadManager::getThreadId(), tmp));
61 } else {
62 ThreadManager::threads[ThreadManager::getThreadId()].e = e;
63 }
64 mtx.unlock();
65 }

67 void ThreadManager::newEnv(const char *threadName) {
68 JNIEnv *env = nullptr;
69 JavaVMAttachArgs args{JNIVersion,
70 const_cast<char *>(threadName),
71 nullptr};
72 gvm->AttachCurrentThread((void **) &env, &args);
73 ThreadInfo tmp{env, true};
74 ThreadManager::threads.insert(std::pair<std::string, ThreadInfo>(ThreadManager::getThreadId(), tmp));
75 Logger::logger.info("refresh env");
76 };
77
78 void ThreadManager::detach() {
79 mtx.lock();
80 if (ThreadManager::included(ThreadManager::getThreadId())) {
81 bool att = ThreadManager::threads[ThreadManager::getThreadId()].attach;
82 ThreadManager::threads.erase(ThreadManager::getThreadId());
83 if (att)
84 gvm->DetachCurrentThread();
85 }
86 mtx.unlock();
87 }

89 bool ThreadManager::included(const std::string &id) {
90 if (ThreadManager::threads.empty() || ThreadManager::threads.count(id) == 0)
91 return false;
92 return true;
93 }

95 JNIEnv *ThreadManager::getEnv(const std::string &file, int loc, const std::string &func) {
96 mtx.lock();
97 if (!ThreadManager::included(getThreadId())) {
98 ThreadManager::newEnv();
99 }
100 JNIEnv *tmp = ThreadManager::threads[ThreadManager::getThreadId()].e;
101 ThreadManager::threads[ThreadManager::getThreadId()].stack.push(file, loc, func);
102 mtx.unlock();
103 return tmp;

216 //message chain
217 MessageChain MessageChain::deserializationFromMiraiCode(const std::string &m) {
218 size_t pos = 0;
219 size_t lastPos = -1;
220 MessageChain mc;
221 if (m.length() <= 7) {
222 return MessageChain(PlainText(m));
223 }
224 do {
225 if (m.length() - 7 - pos > 0 && m.substr(pos, 7) == "[mirai:") {
226 if (pos - lastPos > 1)
227 mc.add(PlainText(m.substr(lastPos + 1, pos - lastPos - 1))); // plain text
228 size_t back = MessageChain::findEnd(m, pos);
229 if (back == -1) throw IllegalStateException("");
230 std::string tmp = m.substr(pos, back - pos);
231 tmp = Tools::replace(tmp, "[mirai:", "");
232 size_t i = tmp.find(':'); // first :
233 int t = SingleMessage::getKey(tmp.substr(0, i));
234 switch (t) {
235 case 0:
236 // no miraiCode key is PlainText
237 break;
238 case 1:
239 mc.add(At(std::stoll(tmp.substr(i + 1, tmp.length() - i - 1))));
240 break;
241 case 2:
242 mc.add(Image(tmp.substr(i + 1, tmp.length() - i - 1)));
243 break;
244 case 3:
245 mc.add(AtAll());
246 break;
247 case 4:
248 mc.add(LightApp(tmp.substr(i + 1, tmp.length() - i - 1)));
249 break;
250 case 5: {
251 size_t comma = tmp.find(',');
252 mc.add(ServiceMessage(std::stoi(tmp.substr(i + 1, comma - i - 1)),
253 tmp.substr(comma + 1, tmp.length() - comma - 1)));
254 break;
255 }
256 default:
257 mc.add(UnSupportMessage("[mirai:" + tmp));
258 break;
259 }
260 pos = back;
261 lastPos = pos;
262 if (t == 1)
263 lastPos++;
264 }
265 pos++;
266 } while (pos < m.length());
267 if (lastPos + 1 < m.length())
268 mc.add(PlainText(m.substr(lastPos + 1, m.length() - lastPos - 1))); // plain text
269 return mc;
270 }

272 MessageChain MessageChain::deserializationFromMessageSourceJson(const json &tmp) {
273 json j = tmp["originalMessage"];
274 MessageChain mc;
275 for (auto node: j) {
276 if (node["type"] == "SimpleServiceMessage") {
277 mc.add(ServiceMessage(node["serviceId"], node["content"]));
278 continue;
279 }
280 if (node["type"] == "LightApp") {
281 mc.add(LightApp(node["content"]));
282 continue;
283 }
284 if (node["type"] == "OnlineAudio") {
285 mc.add(OnlineAudio(node["filename"], node["fileMd5"], node["fileSize"], node["codec"], node["length"],
286 node["urlForDownload"]));
287 continue;
288 }
289 if (node["type"] == "FileMessage") {
290 mc.add(Group(tmp["targetId"], tmp["botId"]).getFileById(node["id"]).plus(node["internalId"]));
291 continue;
292 }
293 int type = SingleMessage::getKey(node["type"]);
294 switch (type) {
295 case -2:
296 mc.add(QuoteReply(MessageSource::deserializeFromString(node["source"].dump())));
297 break;
298 case 0:
299 mc.add(PlainText(node["content"].get<std::string>()));
300 break;
301 case 1:
302 mc.add(At(node["target"]));
303 break;
304 case 2:
305 mc.add(AtAll());
306 break;
307 case 3:
308 mc.add(Image(node["imageId"]));
309 break;
310 case 4:
311 Logger::logger.error(
312 "MiraiCP碰到了预料之外的错误(原因:匹配到了LightApp)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段话使MiraiCP可以修复: MessageSource:" +
313 j.dump());
314 break;
315 case 5:
316 Logger::logger.error(
317 "MiraiCP碰到了预料之外的错误(原因:匹配到了ServiceMessage)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段话使MiraiCP可以修复: MessageSource:" +
318 j.dump());
319 break;
320 default:
321 mc.add(UnSupportMessage(node["content"]));
322 Logger::logger.error(
323 "MiraiCP碰到了意料之中的错误(原因:接受到的SimpleMessage在支持之外)\n请到MiraiCP(github.com/Nambers/MiraiCP)发送issue并复制本段话使MiraiCP可以支持这种消息: MessageSource:" +
324 j.dump());
325 }
326 }
327 return mc;
328 }

331 RemoteFile RemoteFile::deserializeFromString(const std::string &source) {
332 json j;
333 try {
334 j = json::parse(source);
335 } catch (json::parse_error &e) {
336 Logger::logger.error("格式化json失败，RemoteFile::deserializeFromString");
337 Logger::logger.error(source);
338 Logger::logger.error(e.what());
339 throw e;
340 }
341 try {
342 struct Dinfo d {
343 j["dinfo"]["url"],
344 j["dinfo"]["md5"],
345 j["dinfo"]["sha1"]
346 };
347 struct Finfo f {
348 j["finfo"]["size"],
349 j["finfo"]["uploaderid"],
350 j["finfo"]["downloadtime"],
351 j["finfo"]["uploadtime"],
352 j["finfo"]["lastmodifytime"]
353 };
354 return RemoteFile(j["id"], j["internalid"], j["name"], j["finfo"]["size"], j["path"], d, f);
355 } catch (json::type_error &e) {
356 Logger::logger.error("json格式化失败，位置:RemoteFile");
357 Logger::logger.error(source);
358 Logger::logger.error(e.what());
359 throw e;
360 }
361 }
