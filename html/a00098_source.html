31 struct PluginFuncAddrData {
32 plugin_entrance_func_ptr entrance = nullptr;
33 plugin_event_func_ptr eventFunc = nullptr;
34 plugin_func_ptr exit = nullptr;
35 plugin_info_func_ptr infoFunc = nullptr;

37 void _resetAddrData() {
38 PluginFuncAddrData tmp;
39 std::swap(*this, tmp);
40 }
41 };
42
43 class PluginData : protected PluginFuncAddrData {
44 public:
45 const std::string path;
46
47 protected:
48 plugin_handle handle = nullptr;
49
50 PluginAuthority authority = PLUGIN_AUTHORITY_NORMAL;
51 bool enabled = false;
52
53 protected:
54 explicit PluginData(std::string inPath) : path(std::move(inPath)) {}
55
56 protected:
57 [[nodiscard]] std::string _getId() const {
58 assert(infoFunc);
59 return infoFunc()->getId();
60 }
61
62 void _load(plugin_handle _handle, const PluginFuncAddrData &funcAddrs) {
63 handle = _handle;
64 *static_cast<PluginFuncAddrData *>(this) = funcAddrs;
65 }
66
67 void _unload() {
68 // cannot modify path
69 handle = nullptr;
70 _resetAddrData();
71 }
72
73 void _enable() {
74 enabled = true;
75 }
76
77 void _disable() {
78 enabled = false;
79 }
80
81 public:
82 [[nodiscard]] bool isEnabled() const { return enabled; }
83
84 [[nodiscard]] bool isLoaded() const { return handle != nullptr; }
85 };
